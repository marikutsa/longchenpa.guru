༄༅། །མན་ངག་ལྟ་བའི་ཕྲེང་བའི་ཚིག་དོན་གྱི་འགྲེལ་ཟིན་མདོར་བསྡུས་པ་ཟབ་དོན་པད་ཚལ་འབྱེད་པའི་ཉི་འོད་ཅེས་བྱ་བ་བཞུགས་སོ། །
༄༅། །ན་མོ་གུ་རུ་ཤྲཱི་མ་ཧཱ་སུ་ཁ་པཱ་དཱ་ཡ། གཉིས་རྟོག་ཆོས་ཀུན་དབྱིངས་སུ་རབ་ཞི་ཤེས་དང་ཤེས་བྱ་རོ་མཉམ་པས། །བརྗོད་མེད་ངང་ལས་བརྗོད་པའི་ཆོ་འཕྲུལ་མོས་པ་དུ་མའི་རྗེས་སུ་འཇུག །སྲིད་པའི་འཁྲུལ་འཁོར་གཞི་རུ་ཟློག་ལ་ཀུན་ཀྱང་དགོངས་པ་གཅིག་ཏུ་འབབས། །དུས་གསུམ་རྒྱལ་ལས་ཕྲིན་ལས་ཁྱད་འཕགས་པདྨ་སཾ་བྷར་གུས་ཕྱག་འཚལ། །འཇ༵མ་དཔལ་གཞོན་ནུ་བཤེས་གཉེན་ཚུལ་འཛིན་པ། །ཆོས་ཆེན་དབྱ༵ངས་ཀྱིས་བསྟན་འགྲོའི་སྣང་ཆེན་སྤེལ། །མཁྱེན༵་པ་གཉིས་ལྡན་རྗེས་བརྩེ༵འི་འཛུམ་ཟེར་ཅན། །མཁས་གྲུབ་དབ༵ང་པོ༵་དཔལ་ལྡན་བླ་མར་འདུད། །ཟབ་དོན་འབྱེད་ལ་རྣམ་དཔྱོད་མཐུ་དམན་ཀྱང༌། །ཐམས་ཅད་གཟིགས་པའི་ཞལ་ལུང་བདུད་རྩིའི་ཆབ། །རྙོགས་མེད་ཡི་གེའི་ལམ་དུ་དྲང་བ་ལ། །བློ་ལྡན་སྐལ་པ་བཟང་རྣམས་དོན་གཉེར་བྱོས། །དེའང་བཅོམ་ལྡན་འདས་ཁྱབ་བདག་རྡོ་རྗེ་འཆང་ཆེན་པོས་སྒྲ་ཐལ་འགྱུར་རྩ་བའི་རྒྱུད་ལས་ལུང་བསྟན་པ་ལྟར་རིགས་གསུམ་མགོན་པོ་མི་བདག་གི་རྣམ་པར་སྣང་བའི་ཆོས་རྒྱལ་མེས་དབོན་རྣམ་པ་གསུམ་གྱི་ནང་ནས་འཇམ་དབྱངས་རྣམ་འཕྲུལ་ལྷ་བཙད་པོ་ཁྲི་སྲོང་ལྡེའུ་བཙན་གྱི་སྐུ་རིང་ལ། འཕགས་ཡུལ་པཎ་ཆེན་དུ་མ་བོད་དུ་སྤྱན་དྲངས་ནས་དམ་པ་ཆོས་ཀྱི་ཉི་མ་ཁོར་ཡུག་ཏུ་བརྡལ་བར་མཛད་པའི་ནང་ནས་གངས་ཅན་པའི་འགྲོ་བ་ཡོངས་ལ་བཀའ་དྲིན་ཁྱད་པར་དུ་ཆེ་བ་རྡོ་རྗེ་ཐེག་པ་ཐུན་མོང་མ་ཡིན་པའི་ཤིང་རྟའི་སྲོལ་འབྱེད། སྔགས་ཀྱི་སྟོན་མཆོག་དཔལ་བདེ་བ་ཆེན་པོ་པདྨ་འབྱུང་གནས་ཞབས་ཀྱིས་ཟབ་ཅིང་རྒྱ་ཆེ་བའི་ཆོས་ཀྱི་འཁོར་ལོ་བསམ་ལས་འདས་པ་བསྐོར་ཏེ་གདུལ་བྱ་རྣམས་གདོད་མའི་ས་ལ་དབུགས་དབྱུང་བར་མཛད་པ་ལས༑ མན་ངག་ལྟ་བའི་ཕྲེང་བ་དང་མ་མོ་གསང་བ་ལས་ཀྱི་ཐིག་ལེ་གཉིས་ཆོས་རྒྱལ་ལ་དངོས་སུ་གདམས་པ་གདུལ་བྱ་ཐུན་མོང་བ་རྣམས་ལའང་ཡོངས་སུ་གྲགས་ཤིང་ཡི་གེའི་ཕྲེང་བར་བར་མ་ཆད་དུ་བྱོན་པས་མཁས་པ་མང་པོས་ཁུངས་སུ་དྲངས་པའི་གཞུང་རྩོད་མེད་དུ་གྲུབ་པ་སྟེ། ལྟ་ཕྲེང་ལ་རོང་ཟོམ་པཎྜི་ཏས་མཛད་པའི་འགྲེལ་པ་ཁྱད་པར་ཅན་སོགས་བཞུགས་མོད། བར་སྐབས་སུ་རྒྱུན་སྲ་བ་ལ་བཀའ་བབ་བདུན་གྱི་མངའ་བདག་རྗེ་བླ་མ་རིན་པོ་ཆེའི་ཐུགས་བསྐྱེད་ཀྱི་མཐུ་ལས་བསྟན་པའི་མེ་རོ་བསླངས་པའི་ཕྲིན་ལས་ཀྱི་རྒྱུན་ལ་ཕན་པར་འདོད་པའི་ལྷག་བསམ་གྱིས་དྲངས་ཏེ། ཕྱོགས་མཚོན་ཙམ་གྱི་རྟོགས་སླའི་བསྡུས་འགྲེལ་ཟིན་ཐུན་གྱི་ཚུལ་དུ་དགོད་པ་ལ་གསུམ། ཀླད་ཀྱི་དོན། གཞུང་གི་དོན། མཇུག་གི་དོན་ནོ། །དང་པོ་ལ་གཉིས། མཚན་བསྟན་པ། མཆོད་པར་བརྗོད་པའོ། །དང་པོ་ནི། འཇིག་རྟེན་དང་འཇིག་རྟེན་ལས་འདས་པའི་ལྟ༵་བ༵་དང༵་མདོ་སྔགས་ཀྱི་ཐེག༵་པ༵་ལ༵་སོ༵གས་པའི༵་ཁོངས་ནས་ལམ་གྱི་གྲོགས་སྤྱོད་པ་དང་འབྲས་བུའི་རྣམ་བཞག་སོ་སོའི་ཁྱད༵་པར༵་ཕྱོགས་གཅིག་ཏུ་སྙིང་པོར་བསྡུ༵་ས་ནས་རྗེ་འབངས་རྣམས་ཀྱི་དོན་ཏུ་གྲུབ་པའི་སློབ་དཔོན་ཆེན་པོ་ཉིད་ཀྱིས་མཛད་པའི༵་བསྐྱུ༵ད་བྱང༵་སྟེ་བརྗེད་བྱང་གི་ཡི་གེའོ། །མཚན་དེ་ལྟར་བརྗོད་པའི་དགོས་པ་ནི། གཞུང་གི་བསྟན་དོན་ངེས་སླ་བ་དང༌། གླེགས་བམ་ཚོལ་བ་ལ་འཇུག་བདེ་བའི་དོན་ཏུ་བཤད་དོ། །གཉིས་པ་ནི། བྷ་ག་ཝན་ཞེས་པའི་སྒྲ་ལས་ལེགས་པ་དྲུག་དང་ལྡན་པའི་བཅོ༵མ་ལྡན༵་འད༵ས་ཏེ་མཚན་དེ་གང་ལ་བརྗོད་པ་དོན་ཡོད་པའི་རྗེ་བཙུན་འཇ༵མ་དཔ༵ལ་གཞོ༵ན་ནུ༵་རྒྱལ་བ་ཐམས་ཅད་ཀྱི་མཁྱེན་རབ་ཀྱི་བདག་པོ་ལ་ཐེག་པ་ཐུན་མོང་ལྟར་མཆོད་བརྗོད་མཛད་པ་ད༵ང༌། སྔགས་ཀྱི་ཐུན་མོང་མ་ཡིན་པའི་སྟོན་པ་དང་སྡུད་མཛད་རྡོ༵་རྗེ༵་ཆོས༵་སམ་རྡོ་རྗེ་འཛིན་པ་ལ༵་གུས་པ་ཆེན་པོས་ཕྱག༵་འཚལ༵་ལོ༵་ཞེས་རང་གི་མངོན་པར་འདོད་པའི་ལྷ་ལ་ཕྱག་བྱ་བ་སྔོན་ཏུ་བཏང་སྟེ་དོན་ལྡན་གྱི་ལས་ལ་འཇུག་པ་སྐྱེ་བོ་སྐྱེས་བུ་དམ་པའི་ངང་ཚུལ་ཡིན་པས་ན་དེ་སྐད་གསུངས་པའོ། །འོན་ཀྱང་འདུས་པ་ཡེ་ཤེས་ཞབས་ལུགས་པ་རྣམས་རྡོ་རྗེ་ཆོས་ཞེས་པ་པདྨ་གར་དབང་ལ་མཛད་པ་མང་ངོ༌། །གཉིས་པ་ལ། ལྟ་བའི་ཁྱད་པར་དང༌། བརྟུལ་ཞུགས་ཀྱི་ཁྱད་པར་བཤད་པ་གཉིས་ལས། དང་པོ་ལའང་། ཕྱི་རོལ་པ་དང༌། ཆོས་འདི་པ་རྣམས་ཀྱི་ལྟ་བ་བཤད་པ་གཉིས། དང་པོ་ལ༑ བསྟན་བཤད་གཉིས་ལས། དང་པོ་ནི། འཇི༵ག་རྟེན༵་གྱི༵་ཁམ༵ས་ན༵་སེམ༵ས་ཅན༵་ཕྱིན༵་ཅི༵་ལོག༵་གི༵་ལྟ༵་བ༵་འཛིན་པ་གྲངས༵་མེད༵་པ༵་རྣམས་མདོ༵ར་བསྡུས་ན་རྣམ༵་པ༵་བཞིར༵་འདུ༵ས་ཏེ༵་། གཙོ་བོར་ཏེ་སྤྱད་པ་ཕྱལ༵་བ༵་དང༵་། རྒྱང༵་འཕེ༵་ན་གཉིས་དང༵་། ལྟ་སྤྱད་མུར༵་ཐུག༵་དང༵་། མུ༵་སྟེག༵་ས་པའོ༵། །དེ་བཞིའི་དང་པོ་གཉིས་ལ་ཞར་ལས་བྱུང་བའི་མ་རིག་པ་ཞེས་ཁ་ཅིག་འཆད་ཅིང༌། ཡང་ན་དེ་དག་གི་རྒྱུད་ལའང་འཇིག་ལྟ་ལྷན་སྐྱེས་ཡོད་པར་འཆད་པ་དང༌། དེ་འདྲའི་ལྟ་ངན་གྱིས་གཞི་བཟུང་སྟེ། ཀུན་བཏགས་ཀྱི་བདག་འཛིན་སྐྱེ་བ་དེ་བཞིའི་བསྟན་རིམ་ལ་བརྟེན་ནས་གོ་བའི་ཆེད་དུ་ཡིན་པར་བྱས་ཀྱང་ཆོག་པ་འདྲའོ། །གཉིས་པ་ལ་གཉིས་ལས། དང་པོ་ཕྱལ་རྒྱང་གཉིས་བཤད་པ་ནི། དེ༵་ལ༵་ཕྱལ༵་བ༵་གང་ཞེ་ན། དེ་ནི༵་འདི་ལྟར་ཕུང་པོ་ལྔས་བསྡུས་ཀྱི་ཆོས༵་ཐམ༵ས་ཅད༵་རྒྱུ༵་གང་འདྲ་བ་དང༵་འབྲས༵་བུ༵འི་ཁྱད་པར་ཅི་འདྲ་བ་རྣམས་དང༌། འཇིག་རྟེན་ཕ་རོལ་ཡོད༵་པ་དང་མེད༵་པ་གང་དུ༵་འང་དཔྱད་ར་མེད་ཅིང་མ༵་རྟོག༵་ས་ཏེ༵་ཚེ་འདིའི་སྣང་བ་ཏུར་ཏུར་པོ་འདི་ལ་མངོན་པར་དགའ་བས་འཇུག་པ་ཙམ་ཞིག་ལས། སྐྱེས་བུའི་དོན་མཐར་ཐུག་ལ་ཅང་མི་སེམས་པར་ཀུ༵ན་ཏུ༵་རྨོང༵ས་པའོ༵། །དེའང་ཕྱལ་བ་ཞེས་པ་གཞི་གྱེན་དུ་འཛེགས་པ་དང༌། ཐུར་དུ་འབབ་པའི་ངོས་མི་མཉམ་པ་མེད་པའི་ཐང་ཐང་པོ་ལྟ་བུ་སྟེ། སྤང་བླང་གི་འཇུག་ལྡོག་ལ་སེམས་པའི་སྒོ་ནས་འཇིག་རྟེན་ཕྱི་མ་ཕན་ཆད་ཀྱི་དོན་བསྒྲུབ་ཅིང༌། འདི་ནི་དགེ་བའི་ཚོགས་ཏེ་འཛེག་པར་བྱ་བ་དང༌། འདི་ནི་སྡིག་པའི་གཡང་ས་སྟེ་ལྟུང་དུ་དོགས་པ་ཅན་ནོ་ཞེས་འདོར་ལེན་བྱེད་པའི་བག་ཡོད་དང་བྲལ་ཏེ། ཚེ་འདིའི་བདེ་བ་ཙམ་ཞིག་གྲུབ་པས་ཆོག་ཤེས་ཤིང་སྣང་བ་བདེ་ཆམ་མེར་སྡོད་པ་ལ་བྱའོ། །རྒྱང༵་འཕེ༵ན་ནི༵་སྤྱིར་མདོ་སྡེ་ནས་འཇིག་རྟེན་རྒྱང་འཕེན་ཞེས་གསུངས་པ་ཕལ་ཆེར་ཕྱི་རོལ་པ་སྤྱི་མིང་དུ་བླ་མ་འགའ་ཞིག་འཆད་ཅིང༌། ཏརྐ་སྡེ་ལྔའི་ཡ་གྱལ་རྒྱང་འཕེན་ཞེས་པ་ཕུར་བུ་བའི་མིང་ཡིན་ཡང༌། འདི་ནི་གྲུབ་མཐས་བློ་མ་བསྒྱུར་བ་ཞིག་ལ་མཁས་པ་རྣམས་བཞེད་དེ། དེ་ལྟར་བྱས་ན་ཚེ༵་སྔ༵་ཕྱ༵་དང་ལས་རྒྱུ་འབྲས་སོགས་ཡོད༵་པའམ་མེད༵་པ་གང་ཡིན་དུ༵་མ༵་རྟོག༵ས་ཤི༵ང་ཚེ༵་གཅི༵ག་པོ་འདི་ལ༵་བཙ༵ན་ཕྱུག༵་དང༵་མཐུ༵་སྟོབ༵ས་སོགས་མཐོ་རིས་ཀྱི་དགའ་བ་ཁྱད་པར་ཅན་སྒྲུབ༵་པའི་རིག་སྔགས་འཛིན་པ༵་སྟེ༵་འཇི༵ག་རྟེན༵་གྱི༵་དྲང་སྲོང་དང་ལྷས་བསྟན་པའི་གསང༵་ཚིག༵་ལ༵་བརྟེན༵་པའོ༵། །ཞེས་གཞུང་ནས་གསལ་ཁ་འབྱུང་བ་དང་ཡང་མཐུན་ནོ། །དེས་ནི་བསམ་སྦྱོར་གྱི་སྟེང་ནས་འཇིག་རྟེན་ཕ་རོལ་ལ་རྩིས་མི་འཇོག་པར་རྒྱང་རིང་དུ་འཕེན་པ་ལྟར་བྱེད་པས་ན་རྒྱང་འཕེན་ཟེར་བ་ཡིན་ནོ། །གཉིས་པ་མུར་ཐུག་དང་མུ་སྟེགས་ཀྱི་གྲུབ་མཐའ་བཤད་པ་ནི། མུར༵་ཐུག༵་པ༵་ནི༵་། སེམས་ཅན་གྱི་ཕུང་པོས་བསྡུས་པའི་ཆོས༵་ཐམ༵་ས་ཅད༵་ལ་བསགས་པའི་ལས་སྔོན་སོང་གི་རྒྱུ༵་དང༵་ཤེས་པའི་རིགས་འདྲ་སྔ་མ་སྔོན་སོང་ཡོད་པས་སྔོན་གྱི་མཐའ་མ་དམིགས་པར་མ་རྟོགས་ཤིང༌། འདི་ནས་འཆི་འཕོ་བ་ན་འཆི་སེམས་ཐ་མའི་འོག་ཏུ་ཚེ་རབས་གཞན་གྱིས་བསྡུས་པའི་རྣམ་ཤེས་ཀྱི་རྒྱུན་ཕྱི་མ་འབྲས༵་བུ༵་ར་འབྱུང་བ་དང༌། དེར་ཡང་བདེ་སྡུག་ལས་ཀྱི་རྗེས་སུ་བྱེད་པའི་རྒྱུ་མཚན་གྱིས་ཕྱི་མའི་མཐའ་མ་དམིགས་པར་འཁོར་བ་སྲིད་དུ་གནས་པའི་ཚུལ་མི་ཤེས་པར་མ་མཐོང་བའི་རྒྱུ་མཚན་ལྟར་སྣང་ཙམ་ལ་བརྟེན་ནས་སྐྱེ་བ་སྔ་ཕྱི། ལས་དང་དེའི་རྣམ་སྨིན་ཐམས་ཅད་མེད༵་པ༵་སྟེ༵། སྤང་ལ་ཤྭ་མོ་རྡོལ་བ་ལྟར་ཚེ༵་གཅིག༵་ལ༵་སྐྱེས༵་པའི༵་ཚོར་འདུ་རྣམ་ཤེས་སོགས་ཕུང་པོའི་ཆོས༵་ཐམ༵ས་ཅད༵་གློ༵་བུར༵་དུ༵་སྐྱེས༵་ལ༵་མཐའ༵་ཡང་ཆད༵་པར༵་ལྟ༵་བའོ༵། །བདག་རང་བཞིན་གྱིས་གྲུབ་པ་དང་རང་རྐྱ་ཐུབ་པ་ནི་འདིས་ཀྱང་ཁས་ལེན་མོད་ཀྱི། དེ་འདྲའི་བདག་དེ་ཉིད་ཐོག་མཐར་ཆད་པར་སྨྲ་བའོ། །མུ༵་སྟེག༵ས་པ༵་ནི༵་ཆོས༵་སམ་ཕུང་པོ་ལྔ་ལ་དབང་བསྒྱུར་བ་པོའི་བདག་དོན་གཞན་དུ་འཛིན་པའི་རིགས་གནས་ཀྱི་ལྟ་བ་ངན་པ་གང་ཞིག་དམིགས་པ་ང་ལ་དམིགས་ནས་རྣམ་པ་རྟག་གཅིག་རང་དབང་ཅན་དུ་དམ་འཆའ་བ་ལ་ཆོས་ཐམ༵ས་ཅད༵་ལ༵་ཀུན༵་ཏུ༵་བརྟག༵ས་པའི༵་བདག༵་རྟག༵་པ༵་ཞིག༵་ཡོད༵་པར༵་ལྟ༵་བ༵་སྟེ༵་ཞེས་གསུངས་ཤིང༌། དེ༵་ལའང༵་འགྲུབ་བྱེད་ཀྱི་རྒྱུ༵་མེ༵ད་པ་ལ༵་གྲུབ་པའི་འབྲ༵ས་བུ༵་བདག་འགྱུར་མེད་ཅིག་གདོད་ནས་ཡོད༵་པར༵་བལྟ༵་བ༵་དང༵་ཞེས་པ་གཞན་དུ་ན་རྒྱུ་ལ་མ་ལྟོས་པའི་འབྲས་བུའི་འཇོག་ཚུལ་མེད་མོད་ཀྱི། ཡོད་པ་ཉིད་ལ་གྲུབ་པའམ་འབྲས་བུ་ཞེས་བཏགས་པར་རོང་ཟོམ་གསུངས་སོ། །ཡང་སྣོད་བཅུད་ཐམས་ཅད་དབང་ཕྱུག་གིས་བྱས་པར་སྨྲས་ནས་ཕྱུགས་ཀྱི་མཆོད་སྦྱིན་ལ་སོགས་པ་དབང་ཕྱུག་མཉེས་པར་བྱ་བའི་ཆོ་གས་མཐོ་རིས་འདོད་པ་རྒྱུ༵་འབྲས༵་ཀྱི་བླང་དོར་ལོག༵་པར༵་འཛིན་ཅིང་ལྟ༵་བ༵་དང༵་། རྒྱུ༵་སྟེ་ཕུང་པོ་ཡོངས་སུ་འཛིན་པའི་བྱེད་པ་པོ་བདག་བདེན་པར་གྲུབ་ཅིང་རྟག་པའི་ངོ་བོར་ཡོད༵་པ༵་ལ༵་དེའི་འབྲ༵ས་བུ༵་ས་ཕུང་པོ་རྣམས་ནི་མི་རྟག་པའི་ཕྱིར་གཏན་དུ་བའི་རྟེན་གྱི་ངོ་བོར་མེད༵་པར༵་ལྟ༵་བ༵་ཞེས་འགྲེལ་བྱེད་མཁན་འཆད་ལ། ལུགས་དེ་དག་རྟག་ལྟ་བ་རྣམས་ཀྱི་ཞེ་འདོད་ཆེ་ལོང་ཙམ་ཞིག་གསུང་བ་མ་གཏོགས། འདིར་བསྟན་གྱི་འདོད་ཚུལ་གསུམ་ཕྱི་རོལ་པའི་གྲུབ་མཐའ་མི་འདྲ་བ་གསུམ་དང་སྦྱོར་དགོས་པའི་ངེས་པ་མི་སྣང་སྟེ། བདག་རྟག་པ་དང༵་དབང་ཕྱུག་བྱེད་པ་པོ་ཡིན་པ་གཉིས་ཀ་འུག་ཕྲུག་པས་ཁས་ལེན་པ་བཞིན་ནོ། །འདི༵་དག༵་ནི༵་སྲིད་རྩ་གཅོད་པའི་ལམ་ཡང་དག་དང་མ་འབྲེལ་བར་དེ་ཁོ་ན་ཉིད་ལ་རྨོངས་པའི་མ༵་རིག༵་པ་ཕྱི་ཕྱིར་ཞིང་བརྟན་པར་བྱེད་པའི༵་ལྟ༵་བ༵་ངན་པའོ༵། །སྤྱིར་མུ་སྟེགས་ཅན་གྱི་ལུགས་སྙིང་པོ་མེད་པར་སྒྲུབ་པའི་སྒོ་མཐའ་མེད་པ་ཞིག་ཡོད་ཀྱང་གཙོ་བོ་ནི་གཞུང་འདིར་བཤད་པ་ལྟར་ཐར་ལམ་ཡང་དག་དང་བྲལ་བའི་སྒོ་ནས་སྨད་པ་འདི་ཉིད་ཡིན་ཏེ། མགོན་པོ་ནཱ་གརྫུ་ནས། གང་ཟག་ཕུང་པོ་སྨྲ་བ་ཡི། །འཇིག་རྟེན་གྲངས་ཅན་འུག་ཕྲུག་དང༌། །གོས་མེད་བཅས་ལ་གལ་ཏེ་ཞིག །ཡོད་མེད་འདས་པ་སྨྲ་ན་དྲིས། །ཞེས་སོགས་གསུངས་པ་བཞིན་ནོ། །གཉིས་པ་ལ་མདོར་བསྟན་པ་དང༌། རྒྱས་པར་བཤད་པ་གཉིས། དང་པོ་ནི། འདུལ་བ་ལུང་ལས༑ ཆོས་རྣམས་གང་དག་རྒྱུ་ལས་བྱུང་བ་དང༌། །དེ་ཡི་རྒྱུ་དང་དེ་འགོགས་གང་ཡིན་པའང༌། །དེ་བཞིན་གཤེགས་པ་ཉིད་ཀྱིས་བཀའ་སྩལ་ཏེ། །དེ་སྐད་གསུངས་པའི་ཚུལ་ཅན་དགེ་སྦྱོང་ཆེ། །ཞེས་གསུངས་པ་ལྟར་སྡུག་བསྔལ་རྒྱུ་དང་བཅས་པས་འགོག་པའི་ཐབས་ཆེན་པོ་འཇིག༵་རྟེན༵་ལས༵་འདས༵་པའི༵་ཟག་མེད་ཀྱི་ལམ༵་དང་ཉེ་བར་བྱེད་པ་བདེན་བཞིའི་འཇུག་ལྡོག་ལྷུར་ལེན་གྱི་གདམས་ངག་གང་ལ༵་ཡོད་པའི་སྟོན་པ་སངས་རྒྱས་བཅོམ་ལྡན་འདས་ཀྱི་ཚུལ་ལུགས་དེ་བཤད་པར་བྱ་བ་ལའང༵་རྣམ༵་པ༵་གཉིས༵་ཏེ༵་། མཚ༵ན་ཉིད༵་ཀྱི༵་ཐེག༵་པ༵་དང༵་རྡོ༵་རྗེའི༵་ཐེག༵་པ༵འོ༵། །དེ་ལྟར་གསུངས་པ་ནི་བདེན་གཉིས་ཇི་ལྟར་གནས་པའི་མཚན་ཉིད་དམ་ཆོས་ཀྱི་ཡིན་ལུགས་དཔྱིས་ཕྱིན་པ་དེ་ཉིད་དྲི་མ་མེད་པའི་རིགས་པས་གཏན་ལ་འབེབས་པ་ཙམ་ཞིག་མ་གཏོགས། འབྲས་བུའི་རྣམ་པ་སངས་རྒྱས་ཀྱི་སའི་གནས་ལུས་ལོངས་སྤྱོད་ལམ་དུ་མི་བྱེད་པ་དང༌། དེ་ལས་ལྡོག་སྟེ་དེ་བཞིན་གཤེགས་པའི་གཟུགས་ཀྱི་སྐུའི་ང་རྒྱལ་ལ་བརྟེན་ནས་སྒོ་གསུམ་སྐུ་གསུང་ཐུགས་སུ་མི་ཕྱེད་པར་འཆིང་བའི་ཁྱད་པར་གྱིས་ཡིན་ནོ། །མཚན་ཉིད་ཐེག་པ་ལ་གསུམ་དུ་མཛད་པ་ཐོབ་བྱའི་སྒོ་ནས་ཡིན་ཞིང༌། ཐེག་ཆེན་ལ་མཚན་ཉིད་དང་རྡོ་རྗེའི་ཐེག་པ་གཉིས་སུ་བྱེ་བ་ལམ་སྒྲུབ་པའི་ཚུལ་གྱི་ནང་ནས་གཙོ་བོར་ཐབས་ཀྱི་ཕྱོགས་རགས་པ་ལ་མི་འདྲ་བའི་ཁྱད་ཞུགས་པའི་སྒོ་ནས་ཡིན་པ་རྒྱལ་དབང་དྲི་མེད་འོད་ཟེར་གྱི་བཞེད་པའོ༑ ༑གཉིས་པ་ལ། མཚན་ཉིད་ཐེག་པ་དང༌། སྔགས་ཀྱི་ཐེག་པའི་རྣམ་བཞག་གཉིས། དང་པོ་ལའང་གཉིས་ལས། མདོར་བསྟན་པ་ནི། མཚ༵ན་ཉིད༵་ཀྱི༵་ཐེག༵་པ༵་དེ་ལ༵འང་ཐོབ་བྱ་འབྲས་བུའི་མཆོག་དམན་གྱི་སྒོ་ནས་རྣམ༵་པ༵་གསུ༵མ་དུ་བཞག་སྟེ༵་། ཉན༵་ཐོས༵་ཀྱི༵་ཐེག༵་པ༵་དང༵་། རང༵་སང༵ས་རྒྱས༵་ཀྱི༵་ཐེག༵་པ༵་དང༵་། བྱང༵་ཆུབ༵་སེམ༵ས་དཔའི༵་ཐེག༵་པའོ༵། །མཐར་ཐུག་གི་ཐོབ་བྱ་ནི་སངས་རྒྱས་ཀྱི་བྱང་ཆུབ་ཁོ་ནར་ཟད་དེ། དེ་བཞིན་ཏུ་ཐེག་པའི་མཐར་ཐུག་ཀྱང་ཐེག་པ་ཆེན་པོ་ཁོ་ནར་བབས་པའི་ཚུལ་ཤིང་རྟ་ཆེན་པོ་ཀཱ་མ་ལ་ཤི་ལས་དབུ་མ་སྣང་བར་བསྒྲུབ་པ་ལ་སོགས་པ་བཞིན་ནོ། །འོན་ཀྱང་གདུལ་བྱའི་བསམ་པའི་དབང་གིས་རེ་ཞིག་ཐེག་པ་གསུམ་དུ་གསུངས་པ་ནི། དམ་ཆོས་པད་དཀར་ལས། སངས་རྒྱས་ཡེ་ཤེས་རབ་ཏུ་རྟོགས་བྱའི་ཕྱིར། །ཐབས་འདི་རང་བྱུང་གིས་ནི་མཛད་གྱུར་གྱི། །དེ་དག་རྣམས་ལ་ཁྱེད་ཅག་སངས་རྒྱས་སུ། །འགྱུར་རོ་ཞེས་ནི་ནམ་ཡང་མི་གསུང་ངོ༌། །ཅི་ཕྱིར་ཞེ་ན་སྐྱོབ་པ་དུས་ལ་གཟིགས། །ཞེས་སོ། །གཉིས་པ་ལ། ཉན་ཐོས། རང་རྒྱལ། བྱང་སེམས་ཀྱི་ཐེག་པ་བཤད་པ་གསུམ་ལས་དང་པོ་ནི། དེ༵་ལ༵་ཉན༵་ཐོས༵་ཀྱི༵་ཐེག༵་པ༵་ལ༵་ཞུག༵ས་པ༵་རྣམ༵ས་ཀྱི༵་ཞེས་པའི་ཉན་ཐོས་ཀྱི་ཐེག་པར་ཞུགས་མཚམ་ནི་འཁོར་བའི་འཁོར་ལོ་གང་དུ་སྐྱེས་ཀྱང་འཕགས་པ་རྣམས་དང་ཉེ་བར་མི་མཐུན་པའི་མཚན་ཉིད་ནད་དང་འབྲས་དང་ཟུག་རྔུ་ལྟར་མཐོང་བས་ཡིད་བྱུང་སྟེ་རང་གི་དོན་དུ་ཉན་ཐོས་ཀྱི་བྱང་ཆུབ་ཐོབ་པར་འདོད་པའི་འདུན་པ་བཅོས་མིན་ནམ་སྐྱེས་པ་ནས་ཐེག་པ་དེར་ཞུགས་པར་འཇོག་ཅིང༌། དེའང་ཡིད་སྨོན་ཙམ་ཞིག་གིས་རང་འབྲས་འགྲུབ་པ་མི་སྲིད་པར་ཤེས་ནས་དང་པོར་སྲིད་པའི་རྒུད་པ་མཐའ་དག་གི་རྩ་བ་འཇིག་ལྟའི་རྨིང་ལ་ཐུག་པར་མཐོང་ཞིང༌། དེ་ལྡོག་པ་ལ་དེ་དང་འཛིན་སྟངས་དངོས་འགལ་དུ་སོང་བའི་བདག་མེད་རྟོགས་པའི་ཡེ་ཤེས་ཁྱད་པར་ཅན་ཞིག་སྐྱེད་པ་དགོས་སོ། །དེ་ལ་བདག་མེད་ཀྱི་ལྟ༵་བ༵་ཅི་འདྲ་ཞིག་དགོས་པ་ནི༵་། ཕུང་པོ་ལྔའི་ཆོས༵་ཐམ༵ས་ཅད༵་ལ༵་མུ༵་སྟེག༵ས་པ༵་ཞེས་པ་སྨོས་མ་ཐག་པའི་མུར་ཐུག་གི་ཟླས་དྲང་པའི་དེ་ལ་དགོངས་ཤིང༌། ལ༵་སོག༵ས་པའི༵་ཁོངས་ནས་མུར་ཐུག་དང་ཕྱལ་རྒྱང་གཉིས་འཐོན་ཏེ། དེ་རྣམས་ཀྱིས་སྒྲོ༵་དང༵་སྐུར༵་བས༵་ཀུན༵་ཏུ༵་བརྟག༵་པའི༵་ཡེ༵་ནས་མེད༵་པ༵་ཆད༵་པའི༵་ལྟ༵་བ༵་དང༵་། རྟག༵་པར༵་ལྟ་བ་ལ༵་སོག༵ས་པ༵་སྟེ་ཡན་ལག་དང་བཅས་པ་ཞེས་གསུངས་པའི་ཚིག་ཟུར་སྔ་མ་ནི་རྟག་ཆད་དུ་ལྟ་བའི་ལྟ་ངན་ཡེ་ནས་མེད་པར་བསྟན་པ་མིན་གྱི། ལྟ་ངན་དེ་དག་གི་ཞེན་ཡུལ་རྟག་ཆད་ཀྱི་མཐའ་གཉིས་ཡེ་མེད་དུ་བསྟན་པ་ཡིན་ནོ། །དེ་འདྲའི་རྟག་ཆད་ཀྱི་མཐའ་གཉིས་ཡོད༵་པར༵་ལྟ༵་བ༵་ནི༵་ལོག་ཤེས་ཡིན་ཏེ། དེས་ཇི་ལྟར་བཟུང་བ་བཞིན་ཏུ་མ་གྲུབ་པའི་ཕྱིར་ཏེ། ཐག༵་པ༵་ལ༵་སྦྲུལ༵་དུ༵་མཐོང༵་བ༵་བཞིན༵་ནོ་སྙམ་དུ༵་ཚད་མས་དྲང་བའི་ངེས་ཤེས་རྙེད་ནས་ལྟ་བ་ངན་པའི་གཡང་སར་མི་འཇུག་གོ །གོང་གི་ཡེ་མེད་ཅེས་པས་དངོས་པོ་དེའི་བདག་ཉིད་དུ་མ་གྲུབ་པར་བསྟན་ཅིང་སྐབས་འདིར་མེད༵་དེ༵་ཞེས་གསུང་པ་ནི་དེ་ལྟར་མེད་པར་ཚུལ་བཞིན་ཡིད་ལ་བྱེད་པའི་དོན་ཡིན་པས་ཟློས་སྐྱོན་སེལ་ཏེ། འོག་ཏུ་རྣམ་པར་ཤེས་པས་དོན་དམ་དུ་ཡོད་པར་ལྟ་ཞིང༌། ཞེས་པའི་ཐད་ཀྱི་ལྟ་ཞིང་གི་ཚིག་ནི་འདིར་ཡང་ཤིན་ཏུ་གསལ་བའི་ཕྱིར་རོ། །ཚིག་དེ་དག་གིས་རྟག་གཅིག་རང་དབང་བའི་བདག་ཡོད་པ་སེལ་ཏུ་ཟིན་ཀྱང༌། རང་རྐྱ་ཐུབ་པའི་རྫས་ཡོད་ཀྱིས་སྟོང་པའི་གང་ཟག་གི་བདག་མེད་རྟོགས་དགོས་པ་མ་བསྟན་ནོ་ཞེ་ན། དེ་འདྲའི་ལུས་སེམས་གང་ལའང་མ་ལྟོས་པའི་གང་ཟག་ཡོད་འཛིན་ཅིག་འཐད་དོ་སྙམ་པ་ཀུན་བཏགས་ཡིན་པས་དེའང་དེའི་ཁོངས་སུ་འགོག་དགོས་པར་མ་བསྟན་པ་མིན་ནོ། །དེ་ལྟར་གང་ཟག་གི་བདག་མེད་རྟོགས་ཀྱང་ད་དུང་ཆོས་ཀྱི་བདག་མེད་མ་རྟོགས་པར་མ་ཟད། ཆོས་ཀྱི་བདག་ཡོད་འཛིན་འཐད་ལྡན་དུ་བལྟ་བའི་ཚུལ། ཕུང༵་པོ༵་དང་ཁམ༵་ས་དང༵་སྐྱེ༵་མཆེ༵་ད་ལ༵་སོ༵གས་པའི༵་རྩོམ་གཞི། འདོད་ན་དབང་པོ་མེད་པ་དང༌། །སྒྲ་མེད་ཕྲ་རབ་རྡུལ་རྫས་བརྒྱད། །ཅེས་གསུངས་པ་ལྟར་འབྱུ༵་ང་བ༵་ཆེན༵་པོ༵་ས་ཆུ་མེ་རླུང་བཞི༵་འི་རྡུལ༵་ཕྲ༵་རབ༵་སོགས་རྡུལ་ཕྲན་ཆ་མེད་དང༵་། རྣམ༵་པར༵་ཤེས༵་པ༵་སྐད་ཅིག་ཆ་མེད་ནི༵་དོན༵་དམ༵་པར༵་ཡོ༵ད་པར༵་རམ་དོན་དམ་བདེན་པར་ལྟ༵་ཞིང༵་ཞེས་འཆད་དགོས་ཀྱི། དོན་དམ་དཔྱོད་པའི་ངོར་ཡོད་ཅིང་དོན་དམ་བདེན་པ་ཡིན་མི་དགོས་པ་ལྟ་བུ་ལ་བྱར་མི་རུང་སྟེ། རགས་རིམ་ཙམ་མོ། །དེ་ལྟར་གསུངས་པ་ནི་ཉན་ཐོས་ཀྱི་ཐེག་པ་དང་བྱེ་སྨྲའི་གྲུབ་མཐའ། རང་རྒྱལ་གྱི་ཐེག་པ་དང་སེམས་ཙམ་པའི་གྲུབ་མཐའ། བྱང་སེམས་ཀྱི་ཐེག་པ་དང་དབུ་མ་པའི་གྲུབ་མཐའ་རྣམས་རྣམ་བཞག་ཙམ་གྱི་སྒོ་ནས་སྦྱར་ཏེ་འཆད་པར་དགོངས་པའི་ཚུལ་རྗེ་བཙུན་བྱམས་པའི་མན་ངག་དང་མཐུན་ནོ། །ལྟ་བའི་གནད་དེ་འདྲ་ཞིག་ངེས་པར་བྱས་ནས་ཉམས་སུ་ལེན་ཚུལ་ཚང་ལ་མ་ནོར་བ་ནི་ནད་ཤེས་པར་བྱ་བ་སོགས་ཀྱི་དཔེས་བསྟན་པ་ལྟར་སྡུག་བསྔལ་ཤེས་པར་བྱ་བ་དང་ཀུན་འབྱུང་སྤང་བར་བྱ་བ་ལ་སོགས་པ་ཀུན་ཉོན་ཕྱོགས་ཀྱི་སྐྱེ་རིམ་དང་རྣམ་པར་བྱང་བའི་འཇུག་ཚུལ་ངེས་པའི་ཆེད་དུ་བདེ་བར་གཤེགས་པས་འཕ༵གས་པའི༵་བདེ༵ན་པ༵་བཞི༵་གསུངས་པ་དེ་དག་ལ་དེ་ལྟར་སྤང་བླང་མ་ནོར་བའི་ལམ་བསྒོ༵མས་པས༵་རིམ༵་གྱིས༵་འབྲས༵་བུ༵་རྣམ༵་པ༵་བཞི༵་པོ་མངོན་པ་གོང་མ་ལྟར་ན་སྦྱོར་ལམ་སྟན་གཅིག་པ་ནས་འཕགས་པ་རིགས་ནས་རིགས་སྐྱེའི་བར་གྱི་རྒྱུན་ཞུགས་ཞུགས་པ་དང་འབྲས་གནས། དེ་ཉིད་ཀྱི་ཕྱི་མའི་ཕྱོགས་གཅིག་ནས་འདོད་ཉོན་དགུ་པ་མ་སྤང་བའི་བར་ཕྱིར་འོང་ཞུགས་འབྲས། དེ་གཉིས་ལས་ཕྱི་མ་ཁྱད་པར་ཅན་ནས་འདོད་ཁམས་ཀྱི་སྒོམ་སྤངས་དགུ་པ་སྤངས་པའི་བར་གྱི་ཕྱིར་མི་འོང་ཞུགས་འབྲས། དེའི་ནང་ནས་ཕྱི་མ་ཁྱད་པར་བ་དང་སྲིད་རྩེའི་ཉོན་མོངས་དགུ་པ་སྤངས་པའི་བར་དགྲ་བཅོམ་ཞུགས་འབྲས་ཏེ། ད་དུང་ཐོད་རྒལ་དང་ཅིག་ཆར་བའི་རྣམ་བཞག་སོགས་སྤྲོས་པས་ཆོག་གོ །དེ་འདྲའི་རང་གི་འབྲས་བུ་གཙོ་བོར་འཕགས་པའི་ལམ་གྱིས་བསྡུས་པ་རྣམས་འགྲུ༵་བ་པ༵་ཡིན༵་ནོ༵་། །གཉིས་པ་ནི། རང༵་སང༵ས་རྒྱས༵་ཀྱི༵་བྱང་ཆུབ་ལ་འདུན་པའི་སྒོ་ནས་དེའི་ཐེག༵་པ༵་ལ༵་ཞུག༵ས་པ༵འི་གང་ཟག་དམ་པ་རྣམ༵ས་ཀྱི༵་ལྟ༵་བ༵་ནི༵་། ཆོས༵་ཐམ༵ས་ཅད༵་ལ༵་མུ༵་སྟེག༵ས་པ་ལ༵་སོག༵ས་པས༵་མེད་ལ་ཡོད་པར་སྒྲོ༵་འདོགས་དང༵་ཡོད་ལ་མེད་པར་སྐུར༵་བའི༵་བློས་ཀུན༵་ཏུ༵་བརྟག༵་པའི༵་བདག༵་རྟག༵་པ༵་ལ༵་སོག༵ས་པ༵་མེད༵་པར༵་ལྟ༵་བ༵་ཉན༵་ཐོས༵་དང༵་མཐུ༵ན་ཞིང༌། དེ༵་ལས༵་ཁྱད༵་པར༵་དུ༵་བྱེད་པའི་ཆོས་ནི་དབྱིག་གཉེན་ཞབས་ཀྱིས། དྲུག་གི་ཅིག་ཆར་སྦྱར་བ་ན། །ཕྲ་རབ་རྡུལ་ཆ་དྲུག་ཏུ་འགྱུར། །དྲུག་པོ་དག་ནི་གོ་གཅིག་ན། །གོང་བུའང་རྡུལ་ཕྲན་ཙམ་དུ་འགྱུར། །ཞེས་གསུངས་པ་བཞིན་ཕྱི་རོལ་གྱི་དོན་འགོགས་པའི་གཏན་ཚིགས་ལ་བརྟེན་ནས་གཟུག༵ས་ཀྱི༵་ཕུང༵་པོ༵་དང་གཟུགས་ཅན་གྱི་སྐྱེ་མཆེད་བཅུ་དང་ཁམས་བཅུ་དང་ཆོས༵་ཁམས་ཀྱི༵་ཕྱོག༵ས་གཅི༵ག་སྟེ་རྣམ་རིག་མིན་པའི་གཟུགས་རྣམས་ལ༵་བདག༵་མེད༵་པར༵་རྟོག༵ས་ཤིང༵་། ཞེས་པ་ཙམ་ལས་རོང་ཟོམ་འགྲེལ་པར་མི་གསལ་ལ། དོན་ཏུ་དེ་དག་གིས་མཚོན་པའི་ཤེས་བྱ་རྣམ་པར་ཤེས་པ་ལས་དོན་གཞན་དུ་གྱུར་པ་འགའ་ཡང་མེད་པར་རྟོགས་པའི་དོན་ནོ། །དེའང་ཆོས་ཐམས་ཅད་ལ་ཞེས་སོགས་ཀྱིས་གང་ཟག་གི་བདག་མེད་དང༌། དེ་ལས་ཁྱད་པར་དུ་སོགས་ཀྱིས་རྣལ་འབྱོར་སྤྱོད་གཞུང་ནས་བཤད་པའི་ཆོས་ཀྱི་བདག་མེད་བསྟན་པའོ། །དེ་འདྲའི་བདག་མེད་གཉིས་རྟོགས་ཀྱི་ཤེས་རབ་ནི་ཉན་ཐོས་ལས་སྤག་པའི་རང་རྒྱལ་གྱི་ཆེ་བའོ། །ཞེས་བསྟན་པར་འགྱུར་རོ། །ཐེག་པ་འདི་བ་རྣམས་ལ་མིང་དེ་ལྟར་བརྗོད་པ་རྒྱུ་མཚན་ཅི་ལ་བརྟེན་ཞེ་ན། དེ་དག་རྟེན་གང་ལ་རང༵་བྱང༵་ཆུབ༵་ཀྱི༵་འབྲ༵ས་བུ༵་ཐོབ༵་པའི༵་སྲིད་པ་ཐ་མའི་དུས༵་ན༵་འང༵་འང་གི་དོན་ལ་སྤྲོ་རྒྱུ་མང༌། ཉན༵་ཐོས༵་ལྟར༵་དགེ༵་བའི༵་བཤེ༵ས་གཉེན༵་ལ༵་མ༵་ལྟོས༵་པར༵་སྔོན༵་གོམ༵ས་པའི༵་ཤུག༵ས་ཏེ་བསྐལ་པ་བརྒྱར་འདུ་བྱས་པའི་བསོད་ནམས་དང་ཡེ་ཤེས་ཀྱི་མཐུས། རྫོགས་སངས་རྒྱས་རྣམས་མ་བྱུང་ཞིང༌། །ཉན་ཐོས་རྣམས་ཀྱང་ཟད་པ་ན། །རང་སངས་རྒྱས་ཀྱི་ཡེ་ཤེས་ནི། །བརྟེན་པ་མེད་པར་རབ་ཏུ་སྐྱེ། །ཞེས་དཔལ་ན་གརྫུ་ནས་གསུངས་པ་ལྟར་རང་གི་ཤེས་རབ་ཀྱི་སྟོབས་ཀྱིས༵་རྟེན༵་ཅིང༵་འབྲེ༵ལ་བར༵་འབྱུ༵ང་བ༵་ཡན༵་ལག༵་བཅུ༵་གཉིས༵་རྒྱུ་དང་འབྲས་བུ་རྣམ་པར་འགྱུར་བས་རྟག་པར་མ་ཡིན་པ་དང་རྒྱུ་ཉིད་ཀྱི༵་རིགས་འདྲ་འབྲས་བུར་སྨིན་པས་ཆད་པར་མ་ཡིན་པ་ཉིད་དུ་ཤེས་ནས་སྔོན་དང་ཕྱི་མའི་མཐའ་ལ་རྨོངས་པ་བཟློག་སྟེ། དེའི་ཚུལ་གྱི་སྒོ༵་ན༵་ས་ཆོས༵་ཉི༵་ད་ཟབ༵་མོའི༵་ད༵་ན་རྟོག༵ས་ནས༵་རང༵་བྱང༵་ཆུབ༵་ཀྱི༵་འབྲས༵་བུ༵་ལེན་པ་མེད་པར་ཟག་པ་ཟད་ཅིང་ཤེས་སྒྲིབ་ཀྱི་ཕྱོགས་གཅིག་ལས་རྣམ་པར་གྲོལ་བ་འཐོ༵བ་པ༵་ཡིན༵་ནོ༵་། །ཡང་རོང་འགྲེལ་ཏུ་རང་རྒྱལ་གྱི་གྲུབ་མཐའ་མདོ་སྡེ་པ་དང་སྦྱར་བ་འདྲ་བ་ཞིག་ཀྱང་གསུངས་འདུག་ལ། ད་དུང་ཐལ་འགྱུར་བ་རང་ལུགས་ལ་ཉན་རང་རྣམས་ཀྱིས་བདག་མེད་གཉིས་ཡོངས་སུ་རྫོགས་པར་རྟོགས་མ་རྟོགས་ཀྱི་དཔྱད་པ་བྱེད་ན་ས་དགེའི་མཁས་པ་རྣམས་ཀྱི་ཞེ་འདོད་དང༌། ཁོ་བོའི་བླ་མ་བདག་ཉིད་ཆེན་པོ་དག་གི་བཞེད་ཚུལ་ཇི་ལྟར་ཡིན་པ་རྣམས་ཞིབ་ཏུ་བརྟགས་པས་རྟོགས་པར་ནུས་ཤིང༌། གཞན་ཡང་སྤྲོ་བྱ་མང་ངོ༌། །གསུམ་པ་ནི། ཆོས་ཀྱི་རྒྱལ་པོ་རིག་འཛིན་འཇིགས་མེད་གླིང་པའི་ཞལ་སྔ་ནས་བྱང་ཆུབ་སེམས་དཔའ་སྔགས་ཀྱི་ཐེག་པ་ལ། ཞེས་སོགས་གསུངས་པ་ལྟར་ཕར་ཕྱིན་དང་སྔགས་ཀྱི་ཐེག་པ་གཉིས་ཀ་བྱང༵་ཆུབ༵་སེམ༵་ས་དཔའི༵་ཐེག༵་པ༵་ཡིན་མོད་ཀྱི། འདིར་ནི་སྔགས་ཀྱི་ཐེག་པའི་ཟླས་དྲང་བ་ཕར་ཕྱིན་གྱི་ཐེག་པ་སྟེ། དེ་ལ༵་སྙིང་རྗེ་ཆེན་པོས་དྲང་པའི་བྱང་ཆུབ་ཀྱི་སེམས་བཅོས་མིན་རྒྱུད་ལ་སྐྱེས་པའི་སྒོ་ནས་ཞུག༵ས་པ༵་རྣམ༵ས་ཀྱི༵་ཟབ་མོའི་ལྟ༵་བ༵་ཅི་འདྲ་ཞིག་རྙེད་པ་ནི༵་། འཁོར་བ་ཀུན༵་ནས༵་ཉོན༵་མོང༵ས་པ༵་དང༵་མྱང་འདས་རྣམ༵་པར༵་བྱང༵་བའི༵་ཕྱོགས་ཀྱི་ཆོས༵་ཐམ༵ས་ཅད༵་དོན༵་དམ༵་པར༵་ཡོད་མེད་དམ་བཏགས་དོན་འཚོལ་བའི་འཐད་པས་བརྟགས་ན་ནི༵་རྡུལ་ཆ་ཙམ་ཡང་རྙེད་པ་མེད་དེ། གཅིག་དུ་བྲལ་ཡང་རྟེན་འབྲེལ་ཆེན་མོ་ལ་སོགས་པའི་རིག་པའི་འགྲོས་ཀྱིས་དེ་ལྟར་ངེས་པའི་ཕྱིར་རོ། །དེའང་མིང་གིས་བཏགས་པ་ཙམ་གྱི་ཤེས་བྱ་རྟོག་པའི་རང་མདངས་སུ་ཤར་བ་ཞིག་མིན་པར་ཡུལ་དེའི་རང་སྟེང་ནས་ཚུགས་ཐུབས་སུ་གྲུབ་པ་རྡུལ་ཙམ་ཡང་མེད་པས་བདེན་པ་གཉིས་གང་གི་དབང་དུ་བྱས་ཀྱང་རང༵་བཞིན༵་གྱིས་གྲུབ་པའི་ཆོས་ནི་རྣམ་པ་ཐམས་ཅད་དུ་མེད༵་དོ་ཞེས་འཆད་པ་དགོས་ཀྱི་དོན་དམ་པར་རང་བཞིན་མེད་ལ། རང་བཞིན་གྱིས་གྲུབ་པ་ནི་ཐ་སྙད་ཀྱི་འཇོག་ཚུལ་ཙམ་ཞིག་ཡིན་ནོ་ཞེས་ཕྱིན་ཅི་ལོག་སྨྲ་བར་མི་བྱ་བ་ལ་དགོངས་ནས་ཀུན་མཁྱེན་བླ་མས་ཀྱང༌། གྲུབ་མཐའ་མཛོད་ཀྱི་ཐལ་འགྱུར་བའི་རྣམ་བཞག་མཛད་སྐབས། བདེན་པ་གཉིས་ཀྱི་དབྱེ་གཞི་རང་བཞིན་མེད་པ་ལ་གསུངས་པ་དང༌། བདག་ཉིད་ཆེན་པོ་ཤནྟ་སུ་རི་བས་ཀྱང༌། སྣང་ཙམ་དེ་ལས་ཡུལ་གྱི་སྟེང་ཉིད་དུ། །གྲུབ་པའི་ངོ་བོ་རྡུལ་ཙམ་མ་མཆིས་ཀྱང༌། །ཞེས་པ་ལ་སོགས་པ་མང་དུ་གསུངས་སོ། །དེས་ན་ལུགས་འདིར་རང་བཞིན་གྱིས་གྲུབ་པ་ཐ་སྙད་དུ་ཡང་མེད་པ་དང༌། དོན་དམ་པར་གྲུབ་པ་ཐ་སྙད་དུའང་མེད་པ༵་དོན་གཅིག་ཡིན༵་ལ༵ །ཀུན༵་རྫོབ༵་ཏུ༵་ནི༵་སྒྱུ་མ༵་དང་རྨི་ལམ་བཞིན་བློས་བཞག་པ་ཙམ༵་དུ༵་བྱ་བྱེད་རྒྱུ་འབྲས་ཐམས་ཅད་སོ༵་སོའི༵་མཚ༵ན་ཉིད༵་མ༵་འདྲེས༵་པར༵་ཡོད༵་དེ༵་། དེ་མེད་ན་དོན་དམ་བདེན་པ་དང་ཐ་སྙད་བདེན་པ་གཉིས་གཅིག་གྲོགས་སུ་གཅིག་འགྲོ་བ་ཞིག་མི་འོང་ངོ༌། །སྒྱུ་མ་དང་རྨི་ལམ་སོགས་ཀྱི་དཔེ་ནི་དེར་སྣང་དང་དེས་སྟོང་གི་སྒོ་ནས་འཇོག་པར་སྣ་ནམ་ཡེ་ཤེས་སྡེས་ལྷག་མཐོང་ཆེན་མོ་ལ་སོགས་པར་བཤད་པ་འདིར་ཡང་འདྲའོ། །སྐབས་འདི་ལ་ནི་བཤད་བྱ་མཐའ་ཡས་ཀྱང་རེ་ཞིག་མི་སྤྲའོ། །དེ་ལྟ་བུའི་བདག་མེད་ཀྱི་ལྟ་བ་དེ་སྦྱིན་སོགས་ཐབས་ཀྱི་ཆ་དུ་མས་ཁྱད་པར་དུ་བྱས་ཏེ། མདོར་ན་ཚོགས་གཉིས་ཀྱིས་བསྡུས་པའི་ཕ༵་རོལ༵་ཏུ༵་ཕྱིན༵་པ༵་བཅུ༵འི་ཉམས་ལེན་ལ་སྤྱད༵་པའི༵་འབྲ༵ས་བུ༵་འམ་ལག་རྗེས་ནི་རབ་ཏུ་དགའ་བ་ལ་སོགས་པའི་ས༵་བཅུ༵་རིམ༵་གྱིས༵་བགྲོད༵་པའི༵་མཐར༵་བླ༵་ན༵་མེད༵་པའི༵་བྱང༵་ཆུབ༵་འགྲུབ༵་པར༵་འདོ༵ད་པ༵་སྟེ་གསུངས་པ་ཡིན༵་ནོ༵། །དེ་ལྟར་ཡང་འཇུག་པ་ལས། ཀུན་རྫོབ་དེ་ཉིད་གཤོགས་ཡངས་དཀར་པོ་རྒྱས་གྱུར་པ། །ངང་པའི་རྒྱལ་པོ་དེ་ནི་སྐྱེ་བོའི་ངང་པ་ཡིས། །མདུན་ཏུ་བདར་ནས་དགེ་བའི་རླུང་གི་སྟོབས་ཤུགས་ཀྱིས། །རྒྱལ་བའི་ཡོན་ཏན་རྒྱ་མཚོའི་ཕ་རོལ་མཆོག་ཏུ་འགྲོ། །ཞེས་སོ། །ཕར་ཕྱིན་བཅུའི་འབྲས་བུ་ས་བཅུར་གསུངས་པ་ནི་སྐྱེད་བྱ་སྐྱེད་བྱེད་ཀྱི་རྒྱུ་འབྲས་སུ་བསྡུ་དགོས་པར་སྣང་བས་ཕར་ཕྱིན་སྔ་མ་སྔ་མས་ས་ཕྱི་མ་ཕྱི་མའི་ཡོན་ཏན་སྐྱེད་པ་ལ་བྱའོ། །གཉིས་པ་ལ། བསྟན་བཤད་གཉིས་ལས། དང་པོ་ནི། སྤྱིར་གསང་སྔགས་ཐབས་ཀྱིས་ཆེ་བའི་ཐེག་པ་ཞེས་བརྗོད་པའི་རྒྱུ་མཚན་ནི། རྡོ་རྗེ་ཐེག་པ་ཞེས་པའི་མཚན་དོན་ལས་ཤེས་ནུས་ཏེ། རོང་ཟོམ་འགྲེལ་པར་གསང་སྔགས་ཀྱི་གཞུང་རྣམས་ལས་ཕུང་པོ་རྣམས་ལ་རྡོ་རྗེ་ཕུང་པོ་དང༌། སྐྱེ་མཆེད་རྣམས་ལ་རྡོ་རྗེ་སྐྱེ་མཆེད་ཅེས་བྱ་སྟེ། སྐུ་གསུང་ཐུགས་རྡོ་རྗེ་ལྟ་བུའི་ལྷའི་རང་བཞིན་ཡིན་པའི་ཕྱིར་རོ་ཞེས་གསུངས་ཤིང༌། ཐབས་ཀྱིས་ཆེ་ཞེས་པའི་ཐབས་ཀྱང་དེ་ཡིན་པའི་ཕྱིར་ཏེ། རྡོ་རྗེ་གུར་ལས། སྟོན་པ་སུམ་ཅུ་རྩ་གཉིས་མཚན། །མངའ་བདག་དཔེ་བྱད་བརྒྱད་ཅུར་ལྡན། །དེ་ཕྱིར་ཐབས་དེས་བསྒྲུབ་བྱ་སྟེ། །ཐབས་ནི་སྟོན་པའི་གཟུགས་ཅན་ནོ། །ཞེས་གསུངས་པའི་ཕྱིར་རོ། །སྔགས་སུ་གསུང་པའི་ཟབ་མོ་འོད་གསལ་གྱི་ཡེ་ཤེས་དང༌། རྒྱས་པ་ཟུང་འཇུག་གི་སྐུ་སྒྲུབ་ཚུལ་རིམ་པ་གཉིས་ཀའི་སྒོ་ནས་ཕར་ཕྱིན་ལས་འཕགས་ཆོས་ཡོད་མོད་ཀྱི། དེ་ནི་སྔགས་བླ་མེད་ཁོ་ནའི་ཡིན་གྱི་རྡོ་རྗེ་ཐེག་པ་གང་རུང་རུང་གི་མིན་ནོ། །རྡོ༵་རྗེ༵་ཐེག༵་པ༵་ལ༵་སྒྲ་བཤད་ལུགས་མི་འདྲ་བ་ཁུངས་ཐུབ་ཀྱི་གཞུང་ནས་བྱུང་བ་གཞན་ཡང་ཡོད་ལ༑ གང་ལྟར་ཡང༵་དེ་འདྲའི་སྔགས་ཀྱི་ཐེག་པ་ལ་དབྱེ་བ་བྱས་ན་རྣམ༵་པ༵་གསུམ༵་མོ། །འང་སྒྲ་དང་མདོ་སྔགས་གཉིས་ཀྱི་སྔ་མ་ལ་གསུམ་ཡོད་པར་མ་ཟད། ཕྱི་མ་ལའང་གསུམ་དུ་ཕྱེ་སྟེ༵་། ཞེས་བསྟན་ནོ། །གསུམ་པོ་དེ་གང་ཞེ་ན༑ བྱ༵་བའི༵་རྒྱུད༵་ཀྱི༵་ཐེག༵་པ༵་དང༵་། གཉིས༵་ཀ༵འི་རྒྱུད༵་ཀྱི༵་ཐེག༵་པ༵་དང༵་། རྣལ༵་འབྱོར༵་རྒྱུད་ཀྱི༵་ཐེག༵་པ༵འོ༵། །རྣལ་འབྱོར་རྒྱུད་ཀྱི་ཁོངས་སུ་སྔགས་བླ་མེད་ཀྱི་རྒྱུད་རྣམས་ཀྱང་བསྡུས་པ་ཡིན་ཏེ། འོག་ཏུ་འཆད་དོ། །འདིའི་སྐབས་ཀྱི་ཐེག་པ་ཞེས་གསུངས་པ་རྣམས་ནི་རྒྱུད་སྡེ་དེ་དང་དེ་ནས་བཤད་པའི་ལམ་རྣམས་ལ་བྱ་ཡི་གོང་སྨོས་ཀྱི་ཐེག་པའི་དབྱེ་ལུགས་གཉིས་པོ་གང་རུང་ནི་མི་བྱ་སྟེ། དེ་ལྟར་བྱས་ན་མཐའ་ཐོན་པ་མི་འོང་ངོ༌། །གཉིས་པ་ལ། བྱ་རྒྱུད༑ སྤྱོད་རྒྱུད། རྣལ་འབྱོར་རྒྱུད་ཀྱི་རྣམ་པར་བཞག་པ་གསུམ་ལས། དང་པོ་ནི། དེ༵་ལ༵་བྱ༵་བའི༵་རྒྱུད༵་ལ་ཞུགས་ཤིང་ཆུ་ཅོད་པན་གྱི༵་དབང་ཐོབ་པ་ནི་དེའི་ཐེག༵་པ༵་ལའ༵ང་ཞུགས་པ༵་ཞེས་བྱ་ལ། དེ་རྣམ༵ས་ཀྱི༵་ལྟ༵་བ༵་སྟེ་ཤེས་རབ་ཁྱད་པར་ཅན་གྱིས་གང་ཞིག་རྟོགས་ནས་བསྒོམ་པར་བྱ་བ་ནི༵་། ཞེས་པའམ་ཡང་ན་དེ་དག་གིས་ཇི་ལྟར་བལྟ་བ་ནི། ཞེས་རང་ཉིད་ལྷར་བསྒོམ་པ་ལ་རང་ལྷར་བལྟ་ཞེས་པ་བཞིན་ནོ། །སྐབས་འདིར་སྣང་བ་ལྷར་སྒྲུབ་ཀྱི་དཔྱད་པ་འདོད་ན་རོང་ཟོམ་ཞབས་ཀྱིས་སྣང་བ་ལྷར་བསྒྲུབ་པའི་བསྟན་བཅོས་དང༌། ཀུན་མཁྱེན་བླ་མའི་རྒྱུད་འགྲེལ་ཕྱོགས་བཅུ་མུན་སེལ་ལ་སོགས་པ་བལྟ་བར་བྱ་ལ། ལྷར་སྒྲུབ་ལུགས་ལ་རོང་ཀློང་གཉིས་བཞེད་པ་མི་འདྲ་བ་ཅུང་ཟད་སྣང་སྟེ་གཞན་ལས་ཤེས་སོ། །བྱ་བའི་རྒྱུད་ཀྱི་དོན་རྣམས་ནི་རྒྱ་ཆེར་སློབ་དཔོན་སངས་རྒྱས་གསང་བས་བཤད་པ་ལྟར་ལྷ་དྲུག །ཡན་ལག་བཞི། བསམ་གཏན་གསུམ་སོགས་མང་དུ་ཡོད་ཀྱང༌། འདིར་བསྟན་ཚོད་ཀྱི་དོན་ནི། དོན༵་དམ༵་པར༵་ཆོས་ཐམས་ཅད་སྐྱེ༵་འག༵་ག་མེད༵་པ༵་མཐའ་བྲལ་དབུ་མའི་གཤིས་སུ་མཉམ་པར་བཞག་སྟེ། དེའི་ངང་ལས༵་ཀུན༵་རྫོབ༵་ཏུ༵་ལྷའི་གཟུག༵ས་མཚན་དང་དཔེ་བྱད་ཀྱི༵་རང་བཞིན་ཡོངས་སུ་དག་པའི་སྐུར༵་བསྒོམ༵་ཞིང༵་། སྐུའི༵་གཟུག༵ས་བརྙན༵་གསལ་བ་དང༵་། ཐུག༵ས་ཀྱི་མཚ༵་ན་མ་ཕྱག་མཚན་འཛིན་པ་དང༵་། གསུང་བཟླ༵་ས་བརྗོ༵ད་ཀྱི་རྣལ་འབྱོར་བྱ་བ་ད༵ང༌། གྲོགས་སུ་ཁྲུས་དང་གཙང༵་སྦྲ༵་ལ་གནས་པ་དང༵་། བཟླས་བརྗོད་ཀྱི་དུས་དང་དུས༵་མ་ཡིན་པའི་ས་ཚིག༵ས་དང༵་། གཟའ༵་དང༵་རྒྱུ༵་སྐར༵་ཤིས་པ་ལ༵་སོག༵ས་པ༵་གཙོ༵་བོར༵་ཕྱིའི་བྱ་བ་མང་པོ་ལ་བརྟེན་ཅིང་ཡོ༵་བྱད༵་དང༵་ཏིང་ངེ་འཛིན་གྱི་རྒྱུ༵་རྐྱེན༵་ཚོག༵ས་པའི༵་མཐུ༵་ལས༵་ཐུན་མོང་དང་མཆོག་གི་གྲུབ་པ་རྣམས་འགྲུབ༵་པའོ༵། །སྐུའི་གཟུགས་བརྙན་ཞེས་པ་འདི་ཆེ་ལོང་ཙམ་དུ་ན་སྐུ་ལྟར་སྣང་བ་ཞེས་པའི་དོན་ཡིན་པས་ལྷའི་རྣལ་འབྱོར་དང་པོར་བསྒོམ་པའི་དུས་སུ་རིགས་གསུམ་ལ་སོགས་པའི་སྐུའི་གསལ་སྣང་དེ་དོན་སྤྱིའི་གསལ་སྣང་ཙམ་ཞིག་ཡིན་པ་ལ་དགོངས་པ་འདྲ་སྟེ། ཀཱ་མ་ལ་ཤི་ལས་སྒོམ་རིམ་ཐ་མར་རྣམ་པར་མི་རྟོག་པའི་གཟུགས་བརྙན་ལ་སོགས་པས་གསུངས་པ་བཞིན་ནོ། །གཉིས་པ་ནི། གཉིས༵་ཀ༵འི་རྒྱུད༵་ཅེས་པ་ཕྱིའི་བྱ་བ་དང་ནང་གི་རྣལ་འབྱོར་གཉིས་ཀ་ཆ་མཉམ་པར་སྤྱོད་པའི་དོན་ཏེ། དེ་འདྲའི་མིང་འཇུག་པའི་སྤྱོད་རྒྱུད་ཀྱི༵་ཐེག༵་པ༵་ལ༵་ཞུགས༵་པ༵་རྣམ༵ས་ཀྱི༵་ལྟ༵་བ༵་ནི༵་། དོན༵་དམ༵་པར༵་སྐྱེ༵་འག༵ག་གཉིས་སུ་མེད༵་པ༵་ལས༵་ཀུན༵་རྫོབ༵་ཏུ༵་ཟླ་བ་ལ་གནས་པའི་ཨཾ་ལས་རྣམ་པར་སྣང་མཛད་ཀྱི་ལྷའི༵་གཟུག༵ས་ཀྱི༵་སྐུ༵་བསྒོམ༵་པར་རྒྱུད་ནས་གསུངས་པ་ལྟར་བྱ་ཞིང༵་། བདག་གི་དེ་ཉིད་དང༌། ལྷའི་དེ་ཉིད་དང༌། བཟླས་བརྗོད་ཀྱི་དེ་ཉིད་དང༌། བསམ་གཏན་གྱི་དེ༵་ཉིད༵་བཞིའམ། ཡང་ན་བཟླས་བརྗོད་ཡན་ཆད་ཀྱི་ཏིང་ངེ་འཛིན་རྣམས་གོང་གི་ཚིག་དེས་བསྟན་པ། འདིར་སྒྲ་དང༌། སེམས་དང༌། གཞི་ལ་གཞོལ་བ་གཉིས་ཏེ་བཞི༵་དང༵་ལྡན༵་པར༵་སྒོམ༵་པའི༵་ཏིང༵་ངེ༵་འཛི༵ན་ནོ། །གཞན་ཡང་ཡོ༵་བྱད༵་དང༵་སྤྱོད་ལམ་ལའང་མི་ལྟོས་པ་མིན་པས་ཕྱིའི་རྒྱུ་ཚོགས་དང་ནང་གི་རྒྱུ༵་རྐྱེན༵་རྣལ་འབྱོར་དང་ལ༵་སོག༵ས་པ༵འི་ཁོངས་ནས་འདོན་རྒྱུ་ནི་གཙང་སྦྲ་སོགས་ཀྱི་སྤྱོད་ལམ་དུ་བྱས་ན་བདེ་ཞིང་གཉིས༵་ཀ༵་ཞེས་པ་ཕྱི་དང་ནང་ལ༵་སྙེགས་པར་བྱ་སྟེ། དེ་དག་ལ་བརྟེན༵་ནས་ལས༵་འབྲས་བུ་འགྲུབ༵་པ༵་སྟེ། ཟད་པར་སྔོན་པོ་གྲུབ་པ་ན་ཉེར་སྤྱད་དཀར་པོའི་ཁ་དོག་ཅན་ཡང་རྣལ་འབྱོར་གྱི་མཐུས་དངོས་སུ་སྔོན་པོར་བསྒྱུར་ནུས་པ་ལྟར། ལྷའི་རྣལ་འབྱོར་དེ་ཉིད་ཀྱི་གསལ་སྣང་ནུས་ལྡན་དུ་སོང་བ་ལས་དག་པའི་སྐུ་དངོས་གནས་སུ་འགྲུབ་པའོ༵་། །དེའང་ཕྱོགས་ལྷ་སྐུའི་རྣལ་འབྱོར་ལ། བྱ་རྒྱུད་ཀྱི་ཚེ་རང་ལྷར་སྒོམ་པ་ཡོད་མེད་དང༌། སྟོང་ཕྱོགས་དེ་ཁོ་ན་ཉིད་ཀྱི་རྣལ་འབྱོར་ལ། ཕྱི་རྒྱུད་དུ་རྫོགས་རིམ་ཡོད་མེད་ཀྱི་ཟེར་ལུགས་མང་པོ་སྣང་ཡང༌། བྱ་བའི་རྒྱུད་མཚན་ཉིད་པ་ལ་ནི་ངེས་པར་རང་ལྷར་སྒོམ་པ་དགོས་ཏེ། སློབ་དཔོན་སངས་རྒྱས་གསང་བས་རྡོ་རྗེ་གཙུག་ཏོར་དེའི་ཁུངས་སུ་དྲང་པའི་ཕྱིར་རོ། །བྱ་རྒྱུད་ཙམ་པོ་བ་ལ་རང་ལྷར་མི་སྒོམ་པའི་གསལ་ཁ་བུདྡྷ་གུ་ཧྱས་མ་མཛད་ཀྱང༌། ཀུན་མཁྱེན་བླ་མ་ཡབ་སྲས་བཞེད་དོ། །འོན་ཀྱང་ཀུན་མཁྱེན་གྱིས་བྱ་རྒྱུད་ལ་ཙམ་ཁྱད་ཀྱི་དབྱེ་བ་མཛད་པ་ནི་དེ་ལྟར་དངོས་བཏགས་ཀྱི་དོན་ཏུ་གཟུང་གི །བསམ་གཏན་དང་པོ་ལ་ཙམ་ཁྱད་ཕྱེ་བ་བཞིན་ཏུ་མི་གཟུང་ངོ་། །དེའི་བཞེད་པས་མཁན་པོ་ཞི་བ་འཚོའི་སྨན་བླའི་མདོ་ཆོག་ཀྱང་བྱ་རྒྱུད་དུ་ཡོད་བསྡུས་ཀྱི་ཚུལ་དུ་གཏོགས་པར་བྱ་ཞིང༌། དེ་ཉམས་སུ་ལེན་པ་ལ་ཆུ་ཅོད་པན་གྱི་དབང་བསྐུར་སོགས་སྔོན་ཏུ་འགྲོ་མི་དགོས་པས་བྱ་རྒྱུད་དངོས་ཀྱང་མིན་པར་བྱེད་དགོས་ཏེ། གཞན་དུའང༌། གསེར་གྱི་ཁ་དོག་ལྟ་བུའི་སྐུ་ལུས་ཀྱིས། །ཞེས་སོགས་གསུངས་པ་རྣམས་ཀྱང་མདོ་སྔགས་ལ་འཇུག་མཚམ་ཡིན་ཞེས་མང་པོ་ཞིག་སྨྲ་བར་སྣང་བ་དང་འདྲའོ། །རྫོགས་རིམ་ནི་སྲོག་དྷུ་ཏིར་བཅིང་བའི་རྣལ་འབྱོར་གྱི་སྟེང་ནས་འཇོག་པས་དེ་ཕྱི་རྒྱུད་དུ་འོང་དོན་མེད་ལ། འདིར་དེའི་དོད་དུ་མཚན་བཅས་དང་མཚན་མེད་ཀྱི་རྣལ་འབྱོར་གཉིས་འཇོག་པ་རྒྱ་གར་གྱི་ཆེན་པོ་དག་གི་ལུགས་སོ། །གསུམ་པ་ལ་བསྟན་བཤད་གཉིས་ཀྱི། དང་པོ་ནི། རྣལ༵་འབྱོར༵་རྒྱུད༵་ཀྱི༵་རྗེས་སུ་འབྲངས་ཤིང་དེའི་ཐེག༵་པ༵་ལ༵་ཞུག༵ས་པ༵་རྣམ༵ས་ཀྱི༵་ལྟ༵་བ༵་ནི༵་རྣམ༵་པ༵་གཉིས༵་ཏེ༵་། རྣལ༵་འབྱོར༵་ཕྱི༵་པ༵་ཐུབ༵་པ༵་རྒྱུད༵་ཅེས་པ་རོང་འགྲེལ་ཏུ་དཀའ་ཐུབ་གྲུབ་པའི་དྲང་སྲོང་མས་དང་མཚུངས་པར་གཙང་སྦྲ་ལ་སོགས་པ་བྱེད་པའི་དོན་དུ་སྦྱར་ཞིང༌། གཙང་སྟོན་འགྲེལ་བར། ནང་གི་རྣལ་འབྱོར་གཙོ་བོར་བྱེད་པའི་སྒོ་ནས་མ་དག་པའི་སྣང་ཞེན་ཐུབ་པར་བྱེད་པའི་ཕྱིར་ཞེས་ཟེར་རོ། །དེའི་ཟླས་དྲང་བའི་རྣལ༵་འབྱོར༵་ནང༵་པ༵་ཐབ༵ས་ཀྱི༵་རྒྱུད༵་དེ་རིམ་པ་གཉིས་ཀྱིས༵་ཆེ་བའི་ཐེག༵་པ༵འོ༵། །དེ་གཉིས་ལ་ཕྱི་དང་ནང་ཞེས་པའི་ཁྱད་འབྱེད་བྱེད་ནི་གཙང་སྟོན་གྱི་གསུང་ན་མང་པོ་ཞིག་འདུག་ཀྱང་མི་འཐད་པའི་ཆ་འགའ་རེ་སྣང་ལ། གཉིས་སྦྱོར་གྱི་ཆགས་པ་ལམ་དུ་བྱེད་མི་བྱེད་ལས་བཞག་པ་སོགས་འཐད་ལྡན་ཡང་སྣང་ངོ་། །རོང་འགྲེལ་ཏུ། ལྟ་སྤྱོད་གཉིས་ཀྱི་སྒོ་ནས་བཞག་ཀྱང་ཧ་ཅང་ཞལ་གསལ་པོ་མི་འདུག་ལ། དོན་ཏུ་རིམ་པ་གཉིས་ཀྱི་ལམ་ནས་ཆོས་གཟུགས་ཀྱི་སྐུའི་སྒྲུབ་ཚུལ་མཐར་ཐུག་ཡོད་མེད་ཀྱི་ཁྱད་པར་སྟེ། བླ་མེད་རྒྱུད་ནས་བཤད་པའི་ལམ་ཐུན་མོང་ཅིག་ལ་མ་བསྟེན་པར་ལམ་གཞན་ཅི་འདྲ་ཞིག་གིས་ཀྱང་བླ་ན་མེད་པའི་ཡེ་ཤེས་མི་འགྲུབ་པར་གསུངས་ཤིང༌། དེའི་གནད་ཀྱང་གཞི་ལ་ཡོད་པའི་འོད་གསལ་རྣམ་མཁྱེན་དུ་སོང་བ་དང༌། དེའི་ངོ་བོ་ཉིད་ཀྱི་གཟུགས་སྐུ་ཁ་སྦྱོར་ཡན་ལག་བདུན་ལྡན་ལམ་དུ་བྱེད་པ་ནི་བླ་མེད་རྒྱུད་ཁོ་ནའི་དགོངས་པ་དང༌། དེ་འདྲའི་སྐུ་གཉིས་ལས་གཞན་པའི་ཐོབ་བྱ་མཐར་ཐུག་ཀྱང་མེད་པའི་ཕྱིར་རོ། །ལྷའི་རྣལ་འབྱོར་དང་དག་རྫོགས་སྨིན་གསུམ་ཚང་བའི་ལྷའི་རྣལ་འབྱོར་ལ་ཁྱབ་ཆེ་ཆུང་བྱུང་བའང་རྒྱུ་མཚན་དེ་ཉིད་ཀྱི་ཕྱིར་ཡིན་ནོ། །གཉིས་པ་ལ། རྣལ་འབྱོར་ཕྱི་རྒྱུད་དང༌། བླ་མེད་རྒྱུད་ཀྱི་རྣམ་བཞག་གཉིས། དང་པོ་ནི། དེ༵་ལ༵་རྣལ༵་འབྱོར༵་ཕྱི༵་པ༵་ཐུབ༵་པའི༵་རྒྱུད༵་ཀྱི༵་ཐེག༵་པ༵་ལ༵་ཞུག༵ས་པ༵་རྣམ༵ས་ཀྱི༵་ལྟ༵་བ༵་ནི༵་༑ ཕྱི༵་འི་བྱ་བ་མང་པོ་དང་ཡོ༵་བྱད༵་ལ༵་གཙོ༵་བོར༵་མི༵་འཛིན༵་ཅིང་ནང་གི་རྣལ་འབྱོར་ཁྱད་པར༵་ཅན་རྣམ་བྱང་ཆེན་མོའི་ཚུལ་གྱི་རྒྱལ་སྲས་སྲིད་པ་ཐ་མ་བ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བའི་རྣམ་པ་དང་མཐུན་པར་མངོན་བྱང་ལྔ་བསྐྱེད་ལ་སོགས་པ་དེ་ཉིད་བསྡུས་པ་དང་རྡོ་རྗེ་རྩེ་མོ་སོགས་ནས་གསུངས་པ་བཞིན་བྱེད་དོ། །དེའང་འདི་ལྟར་དོན༵་དམ༵་པར༵་སྐྱེ༵་འག༵གས་མེད༵་པའི༵་རང་བཞིན་ཅན་གྱི་ལྷ༵་དང༵་ལྷ༵་མོ༵་ཕན་ཚུན་འཁྱུད་པའི་བདེ་བས་དགྱེས་པའི་རྣམ་ཅན་དུ་བསྒོམ་ཞིང༌། དེ་ལྟར་ལུས་འཇའ་ཚོན་ལྟ་བུའི་ལྷའི་སྐུར་བསྒོམ་པར་མ་ཟད་དེ༵་དང༵་འདྲ༵་བའི༵་བདེན་མེད་ཀྱི་ལྟ་བ་ཕུལ་དུ་བྱུང་བས་རྒྱུད༵་ཡོང༵ས་སུ༵་དག༵་པར་སྒྲུབ་པའི༵་ཏིང༵་ངེ༵་འཛིན༵་གྱིས༵་རབ་ཏུ་བཟུང་བའི་འཕག༵་ས་པའི༵་གཟུག༵ས་ཀྱི༵་སྐུ༵་ཕྱག༵་རྒྱ༵་བཞི༵་དང༵་ལྡན༵་པར༵་བསྒོམ༵་པའི༵་རྣལ༵་འབྱོར༵་གཙོ༵་བོར༵་བྱས༵་པ༵་ལས༵་འགྲུབ༵་པ༵འོ༵། །གཉིས་པ་ལ། བསྟན་བཤད་གཉིས་ལས། དང་པོ་ནི། རྣལ༵་འབྱོར༵་ནང༵་པ༵་ཐབ༵ས་ཀྱི༵་རྒྱུད༵་ཀྱི༵་ཐེག༵་པ༵་ལ༵་ཞུག༵ས་པ༵་རྣམ༵ས་ཀྱི༵་ལྟ༵་བ༵་ནི༵་དབྱེ་ན་རྣམ༵་པ༵་གསུམ༵་སྟེ༵་། བསྐྱེད༵་པའི༵་ཚུལ༵་དང༵་། རྫོག༵ས་པའི༵་ཚུལ༵་དང༵་། རྫོག༵ས་པ༵་ཆེན༵་པོའི༵་ཚུལ༵་ལོ༵། །ཞེས་གསུངས་ཏེ། འདི་ནི་མཉམ་སྦྱོར་དུ་གསུངས་པའི་རྣལ་འབྱོར་དང་རྗེས་སུ་རྣལ་འབྱོར་དང་ཤིན་ཏུ་རྣལ་འབྱོར་གསུམ་དཀྱིལ་འཁོར་འཁོར་ལོའི་རྣལ་འབྱོར་ཆ་ཚང་ཞིག་ལ་རྫོགས་པར་བྱེད་པ་ལྟར་མཆོག་གི་དངོས་གྲུབ་སྒྲུབ་པའི་བསྐྱེད་རིམ་དང༌། ཟབ་ཅིང་རྒྱ་ཆེ་བའི་རྫོགས་རིམ་གཉིས་པོ་གང་ལའང་ཚང་བ་རེ་དགོས་པར་མངོན་ནོ། །གཉིས་པ་ལ། ཚུལ་གསུམ་སྤྱིར་བཤད་པ་དང༌། དེ་ལ་འཇུག་པའི་སྒོ་བྱེ་བྲག་ཏུ་བཤད་པ་སྟེ་གཉིས་ལས། དང་པོ་ལ། བསྐྱེད་པ་དང༌། རྫོགས་པ་དང༌། རྫོགས་པ་ཆེན་པོའི་ཚུལ་གསུམ་ལས། དང་པོ་ནི། དེ༵་ལ༵་བསྐྱེད༵་པའི༵་ཚུལ༵་ནི༵་། ཞེས་གསུངས་ཏེ། སྤྱིར་བསྐྱེད་རྫོགས་ཀྱི་འཇོག་ལུགས་མཆོག་ཏུ་གྱུར་པ་ནི་རླུང་དབང་བསྐུར་ཞུགས་གནས་ཐིམ་གསུམ་བྱས་སྟོབས་ཀྱིས་དྲང་བའི་བརྟན་པའི་ཡེ་ཤེས་ཤིག་དང༌། དེ་ལས་ལྡོག་པ་བཅོམ་པའི་ཡེ་ཤེས་སྔགས་བླ་མེད་ཀྱི་ཐུན་མོང་མ་ཡིན་པའི་རྣལ་འབྱོར་དུ་གྱུར་པ་ཞིག་སྟེ། དེ་གཉིས་ཀྱི་སྒོ་ནས་བྱེད་པ་ལོ་ཆེན་དྷརྨ་ཤྲཱི་སོགས་རང་ལུགས་ཀྱི་མཁས་པ་དག་ཀྱང་བཞེད་ལ། ཀུན་མཁྱེན་རྒྱལ་བའི་དབང་པོ་ཡབ་སྲས་ཀྱི་གཞུང་ན་སྣང་ཕྱོགས་དང་སྟོང་ཕྱོགས་ཀྱི་ལྷའི་རྣལ་འབྱོར་གསང་སྔགས་བླ་མེད་ཀྱི་དབང་དུ་བྱས་པ་གཉིས་ལ་བསྐྱེད་རྫོགས་ཀྱི་མིང་འདོགས་མཛད་པ་ཞིག་ཀྱང་སྣང་ངོ༌། །འདིར་གསུངས་པའི་བསྐྱེད་རྫོགས་ནི་འཇོག་ཚུལ་དེ་གཉིས་གང་རུང་ནས་བཞག་པ་མིན་གྱི། ལྷ་གཅིག་གི་བསྐྱེད་རིམ་ལའང༌། ཏིང་འཛིན་རྣམ་གསུམ་ཡན་ཆད་བསྐྱེད་པའི་ཚུལ་དང༌། དེ་ལས་བསྐྱེད་པའི་ལྷ་སྐུ་བཞིན་ལག་གི་རྣམ་ཅན་སྐུའི་ཆ་ཤས་ཡོངས་སུ་རྫོགས་པར་སྒོམ་པ་རྫོགས་པའི་ཚུལ་དང༌། སྒྱུ་འཕྲུལ་གྱི་ལེའུ་བརྒྱད་པ་ནས་འབྱུང་བ་ལྟར་ལུས་དཀྱིལ་སྒོམ་ཞིང༌། དེ་ཐམས་ཅད་རང་གི་རྡོ་རྗེ་ཕུང་པོའི་ཡན་ལག་ཏུ་ངེས་པར་བྱ་བ་རྫོགས་པ་ཆེན་པོའི་ཚུལ་ཡིན་ནོ་ཞེས་ཁོ་བོ་དེ་ལྟར་སེམས་སོ། །རྒྱ་གར་བའི་ཆེན་པོ་རྣམས་ཀྱི་གཞུང་ན་མིང་འདྲ་ཡང་དོན་མི་མཐུན་པ་དེ་འདྲ་ཤིན་ཏུ་མང་སྟེ། སྤྲོས་པས་ཆོག་གོ །དེ་བཞིན་ཏུ་ཆོས་ཀྱི་རྒྱལ་པོ་ཀློང་ཆེན་པའི་གཞུང་དུ་རྫུས་སྐྱེས་ཀྱི་མི་དང་སྐྱེད་ཚུལ་མཐུན་པའི་ལྷའི་རྣལ་འབྱོར་སྔགས་བླ་མེད་ཐུན་མོང་གི་དབང་དུ་མཛད་པ་ཞིག་གསུངས་པ་ལ་བླ་མེད་རང་ལམ་གྱི་ཚོགས་ལམ་པ་རྟོགས་པ་ན་འཕར་བའི་སྐབས་ཤིག་དང༌། སྦྱོར་ལམ་ཡན་ཆད་ཀྱི་རྒྱུད་ཚོད་དུ་སོང་བའི་སྐབས་ཤིག་སྟེ་གནས་སྐབས་ཀྱི་སྒོ་ནས་གཉིས་སུ་ཡོད་པ་ནི་བཅོས་མ་དང་བརྟན་པའི་རྣལ་འབྱོར་གྱི་ཁྱད་པར་ལས་ཡིན་ནོ། །དེའི་ནང་ནས་ཕྱི་མ་ནི་རྫོགས་རིམ་ཐུན་མོང་མ་ཡིན་པའི་ཡེ་ཤེས་ཀྱི་སྐུ་སྒྲུབ་ཚུལ་ཡིན་ཞིང༌། དེ་ནི་གསང་སྙིང་ལས། གཞི་རྩ་མེད་པའི་སེམས་ཉིད་ནི། །ཕོ་མོ་མ་ཡིན་མ་ནིང་མིན། །མཚན་མེད་མ་ཡིན་རིགས་བརྒྱུད་མིན། །ཁ་དོག་མ་ཡིན་དབྱིབས་སུ་མིན། །དེ་བཞིན་ཉིད་དབྱིངས་ཡེ་ཤེས་ཏེ། །ཐབས་ཀྱི་ཕྱག་རྒྱ་ཀུན་གྱི་རྒྱུ། །ཞེས་པ་ལ་སོགས་པ་ཡེ་ཤེས་སྒྱུ་མའི་སྐུ་གང་ལས་སྒྲུབ་པའི་གཞི་དང༌། ཇི་ལྟར་སྒྲུབ་པའི་ཐབས་ཤིན་ཏུ་གསལ་བར་བཤད་པ་རྣམས་དང་གཅིག་གོ །དེ་འདྲའི་སྐུའི་སྒྲུབ་ཚུལ་རྫུས་སྐྱེས་ལྟར་བྱ་བ་འདི་ནི་མགོན་པོ་ཀླུ་སྒྲུབ་ཡབ་སྲས་ཀྱིས་བཤད་པའི་ཁུངས་དང་བསྟུན་ན་རྒྱ་ཆེར་བཤད་དུ་ཡོད་མོད་ཀྱི་རེ་ཞིག་མི་འཆད་ལ། དོན་ཏུ་དེ་ལྟ་བུའི་སྐུ་དེ་སྐད་ཅིག་དྲན་རྫོགས་ཡིན་ན་གཞུང་འདིར་གསུང་བ་འདི་རྣམས་དེ་དང་སྦྱར་དུ་མི་རུང་སྟེ། དེ་ལ་ནི་རྒྱུའི་ཏིང་ངེ་འཛིན་ཡིག་འབྲུ་ལས་སྐྱེད་པ་དང༌། སྐུ་དེ་ལ་ལུས་དཀྱིལ་འགོད་པ་མེད་པའི་ཕྱིར་རོ་ཞེ་ན། བདེན་མདོག་ཁ་བ་འདྲ་ཤས་ཆེ་ཡང་སྐྱོན་མེད་དེ། རྣམ་མཁྱེན་ཤིང་རྟར་སྙིང་ཀའིརྩ་འཁོར་གྱི་དབུས་སུ་མི་ཤིགས་པའི་ཐིག་ལེ་ཧཱུྃ་གི་རྣམ་པར་གསལ་བ་དང༌། དེ་ཉིད་ཡེ་ཤེས་ཀྱི་སྐུའི་སྒྲུབ་ཚུལ་དུ་གསུངས་པ་བཞིན། ངེས་དོན་གྱི་ཧཱུྃ་ཡིག་སྙིང་པོ་ཡེ་ཤེས་ཆེན་པོའི་བཞོན་པ་དང་མེ་རླུང་གི་ཐིག་ལེ་དེ་ཉིད་ལས་འཇའ་ལུས་རྡོ་རྗེ་སེམས་དཔའི་སྐུར་ལྡང་བ་ཡིན་པས་དེ་རྫོགས་རིམ་ཐུན་མོང་མ་ཡིན་པའི་རྒྱུའི་ཏིང་ངེ་འཛིན་དུ་འགྲོ་བའི་ཕྱིར་དང༌། གཉུག་མའི་ཕུང་ཁམས་ཀྱི་རོལ་རྩལ་ཏུ་དག་པ་རབ་འབྱམས་སྒྱུ་མའི་སྣང་བརྙན་སངས་རྒྱས་ཀྱི་སྤྲིན་ཕུང་ཅིར་ཡང་འཆར་བ་དེ་ལུས་དཀྱིལ་གྱི་དོད་ཡིན་པའི་ཕྱིར་རོ། །འོན་ཀྱང་འདིར་བསྟན་རྫོགས་པ་ཆེན་པོའི་ཚུལ་ལ་ལུས་དཀྱིལ་སྒོམ་ཚུལ་ཁོ་ནའི་ཁྱབ་པ་ནི་མེད་དེ། དཀྱིལ་འཁོར་བའི་ལྷ་གཞན་གཙོ་དང་འཁོར་གྱི་རྣམ་པར་རོལ་པ་ཇི་སྙེད་པ་ཐམས་ཅད་རང་གི་རྟོག་ཚོགས་མ་ལུས་པ་ཡོངས་སུ་དག་པའི་ངོ་བོ་ཉིད་དུ་གསལ་ལ་མ་འདྲེས་པར་སྒོམ་པ་ཡང་དེར་འགྲོ་བའི་ཕྱིར་ཏེ། དེ་ལྟ་མིན་ན་གཞུང་འདི་དག་གིས་དཀྱིལ་འཁོར་འཁོར་ལོའི་ཏིང་ངེ་འཛིན་རྫོགས་པར་མ་བསྟན་པར་འགྱུར་བའི་ཕྱིར་རོ། །དེའི་ཕྱིར་འོད་གསལ་རྫོགས་ཆེན་གྱི་རྒྱུད་སྡེའི་རིང་ལུགས་བླ་མེད་རྒྱུད་གཞན་དང་མི་འདྲ་བའི་ལམ་གྱི་སྲོལ་ཆེན་པོ་ཞིག་ཡོད་པ་ནི་འདིར་བསྟན་གྱི་རྫོགས་པ་ཆེན་པོའི་ཚུལ་དང་ཅུང་ཟད་མི་འདྲ་སྟེ། རྒྱུད་ཀྱི་རྒྱལ་པོ་དེའི་ལུགས་ཀྱི་ཆོས་གཟུགས་ཀྱི་སྐུ་སྒྲུབ་པའི་ཚུལ་རྣམས་རྒྱ་ཆེར་གཞན་དུ་འཆད་པར་འདོད་དོ། །དེ་ལྟར་ཚུལ་གསུམ་གྱི་ནང་ནས་དང་པོ་ནི། སྒྱུ་འཕྲུལ་ལས། དངོས་རྣམས་ཉིད་ནི་དངོས་མེད་པར། །རྟོགས་པའི་དབང་བསྒྱུར་ཏིང་འཛིན་ཡིན། །ཞེས་གསུངས་པ་ལྟར། སྣོད་བཅུད་ཐམས་ཅད་རང་སེམས་ཀྱི་ངོ་བོ་བརྗོད་དུ་མེད་པའི་གཤིས་ལྷུན་མཉམ་ཡངས་པ་ཆེན་པོར་རྩལ་གཞི་ལ་ཐིམ་པའི་ཚུལ་གྱིས་བསྡུས་ཏེ། དམིགས་མེད་ཀྱི་ཏིང་འཛིན་ནམ་མཁའ་ལྟར་ཁྱབ་ཅིང་གདལ་བའི་ངང་ལ་མཉམ་པར་འཇོག་པ་དེ་བཞིན་ཉིད་ཀྱི་ཏིང་ངེ་འཛིན་དང༌། དེ་ལྟར་མ་རྟོགས་པའི་སེམས་ཅན་རྣམས་ལ་སྒྱུ་མ་ལྟ་བུའི་སྙིང་རྗེ་ཕྱོགས་ལྷུང་དང་བྲལ་བའི་རྣམ་ལྡན་དུ་ཤར་བ་ཀུན་སྣང་གི་ཏིང་ངེ་འཛིན་དང༌། དེའི་རྐྱེན་གྱིས་ཆུ་དྭངས་བ་ལས་ཉ་ལྡང་བའི་ཚུལ་ཏུ་རིག་པ་ཡེ་ཤེས་ཀྱི་ངོ་བོ་ཧཱུྃ་ཡིག་ལ་སོགས་པའི་ཡིག་འབྲུར་སྒོམ་པ་རྒྱུའི་ཏིང་ངེ་འཛིན་ནོ། །རྒྱུའི་ཏིང་འཛིན་དང་དེའི་རྒྱུ་ཚོགས་ཀུན་སྣང་གི་ཏིང་འཛིན་གཉིས་བར་སྲིད་སྦྱོང་བྱེད་དང༌། དེའི་སྔ་ལ་སོགས་ཀྱི་དེ་ཏིང་འཆི་སྲིད་སྦྱོང་བྱེད་དང༌། འཆད་འགྱུར་གྱི་རྫོགས་པའི་ཚུལ་མན་ཆད་སྐྱེ་བ་སྦྱོང་བྱེད་དུ་བྱས་ན་སློབ་དཔོན་ཆེན་པོ་འཇིགས་མེད་འབྱུང་གནས་སོགས་ཆེན་པོ་རྣམས་ཀྱི་གཞུང་དང་འགྲིགས་སོ། །དེ་ལྟ་བུའི་ཏིང༵་འཛི༵ན་རྣམ༵་པ༵་གསུམ༵་རིམ༵་གྱིས༵་བསྐྱེད༵་དེ༵་། གསུང་ཡི་གེ་སྒྱུ་འཕྲུལ་གྱི་དཀྱིལ་འཁོར་དང༌། ཐུགས་ཕྱག་མཚན་གྱི་དཀྱིལ༵་འཁོར༵་རྣམས་རིམ༵་གྱིས༵་གདན་ལ་བཀོད༵་ཅིང༵་བསྒོམ༵་པས༵་བསྐྱེད་པའི་ཚུལ་ཞེས་བྱ་བ་དེ་ཡོངས་སུ་རྫོགས་པར་འགྲུབ༵་པའོ༵། །གཉིས་པ་ནི། རྫོག༵ས་པའི༵་ཚུལ༵་ནི༵་། རྒྱུ་རིང་གི་སྐབས་སུ་བསྐྱེད་པའི་ས་བོན་ལ་སོགས་པ་ཡོངས་སུ་གྱུར་པ་ལས་དོན༵་དམ༵་པར༵་སྐྱེ༵་འག༵གས་མེད༵་པའི༵་ལྷ༵་དང༵་ལྷ༵་མོ༵་སྟེ་རྣམ་མེད་སྟོང་པ་ཉིད་ཀྱི་ལྷའི་ཕྱག་རྒྱ་རྣམ་བཅས་ཆུ་ནང་གི་ཟླ་བ་ལྟ་བུའི་ལྷར་བཞེངས་པ་སྟེ། ཀྱཻ་རྡོ་རྗེ་ལས༑ བསྐྱེད་པའི་རིམ་པའི་རྣལ་འབྱོར་གྱིས། །བརྟུལ་ཞུགས་ཅན་གྱིས་སྤྲོས་པ་བསྒོམ། །སྤྲོས་པ་རྨི་ལམ་ལྟར་བྱས་ནས། །སྤྲོས་པ་ཉིད་ཀྱིས་སྤྲོས་མེད་བྱ། །ཞེས་གསུངས་པ་ལྟར། རྣམ༵་པར༵་མི༵་རྟོག༵་པའི༵་ཡེ་ཤེས་ཀྱིས་མ་མཐོང་བའི་ཚུལ་དུ་བལྟ་བར་བྱ་བའི་དོན༵་དབུ༵་མ༵་རྟག་ཆད་དང་བྲལ་བའི་ཆོས༵་ཀྱི༵་དབྱིང༵ས་རང་བཞིན་རྣམ་པར་དག་པའི་ཀློང་ལས༵་ཀྱང༵་ལྟ་སྒོམ་གྱི་རྩི་དང་མ་བྲལ་བས་མ༵་གཡོ༵ས་ལ༵་། ཀུན༵་རྫོབ༵་ཏུ༵་འཕག༵ས་པའི༵་གཟུག༵ས་ཀྱི༵་སྐུ༵་གཙོ་དང་འཁོར་གྱི་རྣམ་རོལ་དུ་བསྟན་པ་ཐམས་ཅད་ཀྱང་མི་འགྱུར་བའི་ང་རྒྱལ་གྱིས་བརྟན་པར་གསལ་ལ། དྭངས་གསལ༵་རྙོག་པ་དང་བྲལ་བར༵་བསྒོམ༵་ཞིང༵་། བརྟན་གཡོ་ཐམས་ཅད་ཡེ་ཤེས་གཅིག་གི་རོལ་མོར་མཉམ༵་ལ༵་སྐུའི་ཆ་ཤས་དང་ཕྱག་ཞབས་ཀྱི་རྣམ་འགྱུར་ཕྲ་མོ་ཚུན་ཆད་མ༵་འདྲེས༵་པར༵་བསྒོམ༵་པས༵་ཏིང་ངེ་འཛིན་གྱི་ཚུལ་དེ་འགྲུབ༵་བོ༵། །དེའང་ཐ་མལ་གྱི་ཞེན་པ་དག་བྱེད་ལྷའི་ང་རྒྱལ་བརྟན་པ་དང༌། སྣང་བ་དག་བྱེད་རྣམ་པ་གསལ་བ་ཞེས་གཉིས་སུ་གྲགས་ཤིང༌། སྔ་མ་ལ་མ་དག་པའི་སྣང་བ་མ་ཞེན་པ་ཙམ་དང༌། དེ་རང་མཚན་གྱིས་གྲུབ་པར་ཞེན་པ་གཉིས་ཀྱི་གཉེན་པོར་ང་རྒྱལ་བརྟན་པའི་ནང་ནས་རང་ཉིད་ལྷ་དེ་དང་དེའོ་སྙམ་པའི་ང་རྒྱལ་དང༌། ལྷ་དེ་བདེན་མེད་དུ་ལྟ་བའི་ང་རྒྱལ་གཉིས་འཇོག་པར་ཀུན་མཁྱེན་རྒྱལ་བ་ཡབ་སྲས་ཀྱི་མན་ངག་ལས་འབྱུང་ངོ༌། །གཙང་སྟོན་འགྲེལ་པར་བསྐྱེད་རྫོགས་ཀྱི་ཚུལ་འདི་གཉིས་ལྟ་བའི་བསྐྱེད་ཚུལ་ཡང་མི་འདྲ་བར་བྱས་ཤིང༌། ཚིག་གི་གོ་དོན་ལ་དཔྱད་ན་དེ་གཉིས་ཀྱི་སྣོད་རུང་དུ་སྒྲུབ་པའི་དབང་ཡང་མི་འདྲ་བ་སྒྱུ་འཕྲུལ་རྩ་རྒྱུད་དང་མདོ་ཚོགས་ཆེན་འདུས་པ་ལྟ་བུ་ལ་མཛད་པར་སྣང་ཡང་སྒྱུ་འཕྲུལ་རྩ་རྒྱུད་ཀྱི་ལེའུ་བཅུ་གསུམ་པའི་མན་ངག་ཏུ་གཞུང་འདི་མཛད་པར་གྲགས་ཤིང༌། བཅུ་གསུམ་པར། འདུལ་བ་དགོངས་པ་གསང་བ་དང༌། །རང་བཞིན་གསང་བའི་དོན་རྣམས་ནི། །ཞེས་རྒྱུད་སྡེ་བཞིའི་དབྱེ་བ་མཛད་པ་ཙམ་ལས་མི་གསལ་ཞིང༌། འཆད་འགྱུར་གྱི་རྫོགས་པ་ཆེན་པོའི་ཚུལ་ཡང་མ་ཧཱ་ཡོ་ག་ནས་བཤད་པ་དང་ཐུན་མོང་དུ་མ་གྱུར་པའི་བཤད་ལུགས་གཞུང་ན་མི་སྣང་བ་དང༌། དེ་ལྟར་མི་སྣང་བཞིན་དུ་རྟོག་བཟོ་བྱས་ནས་སྣོན་ན་ཐུག་པ་མེད་པར་འགྱུར་ལ། སྤྱིར་ཡང་བསྙེན་སྒྲུབ་ཡན་ལག་བཞི་དང་ཏིང་འཛིན་གསུམ་དང་ཡན་ལག་དྲུག་དང་རྣལ་འབྱོར་བཞི་དང་མངོན་བྱང་ལྔ་ལ་སོགས་པའི་རིམ་པ་རྣམས་དཀྱིལ་འཁོར་འཁོར་ལོའི་རྣལ་འབྱོར་ཡོངས་རྫོགས་རེའི་ནང་དུ་སོ་སོའི་གཞུང་ནས་འབྱུང་བ་ལྟར་སྦྱོར་རྒྱུ་ཡིན་པ་འཕགས་ཡུལ་གྱི་གྲུབ་པའི་སློབ་དཔོན་རྣམས་ཀྱི་ལུགས་ཡོངས་སུ་གྲགས་པ་དེ་བཞིན་ཏུ་འདིར་ཡང་སྦྱར་ན་གཞུང་གི་སྒྲས་ཟིན་དང་འགྲིགས་པར་འགྱུར་བའི་ཕྱིར་ལམ་གྱི་འགྲོ་ལུགས་མི་མཐུན་པ་གསུམ་དུ་མི་བྱ་ཞེས་རེ་ཞིག་དེ་ལྟར་བྱའོ། །གསུམ་པ་ནི། རྫོག༵ས་པ༵་ཆེན༵་པོ༵་ཤིན་ཏུ་རྒྱས་པའི༵་ཚུལ༵་ནི༵། འཇིག༵་རྟེན༵་ཟག་བཅས་དང༵་། འཇིག༵་རྟེན༵་ལས༵་འད༵ས་པ༵་ཟག་མེད་ཀྱི་ཆོས༵་ཐམ༵ས་ཅད༵་དབྱེར༵་མེད༵་པར༵་བསྒོམ་མོ། །དེ་ལ་འཇིག་རྟེན་ནི་རང་གི་ཕུང་པོ་ལྔའོ། །འཇིག་རྟེན་ལས་འདས་པ་ནི་དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱི་རྡོ་རྗེ་གསུམ་པོ་སྐུ༵་གསུ༵་ང་ཐུག༵ས་མི་ཟད་པ་སྟེ། དེ་གཉིས་དབྱེར་མེད་པ་ནི། གུར་ལས། དེ་ཕྱིར་དཀྱིལ་འཁོར་འཁོར་ལོ་ཞེས། །བདེ་བའི་ཐབས་ཀྱི༵་སྡོམ་པ་སྟེ། །སངས་རྒྱས་ང་རྒྱལ་རྣལ་འབྱོར་གྱིས། །སངས་རྒྱས་ཉིད་ཡུན་རིང་མི་འགྱུར། །ཞེས་གསུངས་པ་ལྟར་དག་པ་རབ་འབྱམས་ཡེ་ཤེས་ཀྱི་རོལ་མོ་གཅིག་ཏུ་ཁམས་གསུམ་མ་ལུས་པ་ཡོངས་སུ་སད་པའི་རྣལ་འབྱོར་གྱི་དེ་ཁོ་ན་ཉིད་ལ་བརྟེན་ཏེ། ཐ་མལ་གྱི་སྣང་བའི་སྟེང་དུ་དག་པའི་ལྷ་སྣང་ལྷན་པོ་བཏབ་པ་ལྟ་བུའི་ཚུལ་མ་ཡིན་པར། སྣང་གྲགས་རིགས་གསུམ་རྒྱལ་བའི་དཀྱིལ་འཁོར་དུ་དངོས་སུ་འཕོས་པའི་ང་རྒྱལ་བྱ་བའོ། །དེའང་རྡོ་རྗེའི་སྔགས་རྒྱུད་དུ་མ་ཞིག་ཏུ་ངེས་དོན་གྱི་དཀྱིལ་འཁོར་སྡུད་རིམ་རྒྱས་པར་གསུངས་པ་དང༌། གཞན་ཡང་གྲུབ་པའི་སློབ་དཔོན་རྣམས་ཀྱི་གཞུང་ན་སྦྱང་གཞིའི་མིང་སྦྱོང་བྱེད་ལ་བཏགས་པ་དང་། སྦྱོང་བྱེད་ཀྱི་མིང་སྦྱང་གཞི་ལ་བཏགས་པ་མང་དུ་སྣང་བ་ནི་སྐྱེ་འཆི་བར་དོ་གསུམ་གྱི་དྲི་མ་རྣམས་དེ་དང་རྣམ་པ་མཐུན་པའི་ལམ་གྱིས་སྦྱངས་ཏེ། གཞི་དུས་ཀྱི་སྐྱེ་འཆི་བར་དོ་གསུམ་ཇི་འདྲ་བའི་འབྲས་བུ་ཡོངས་སུ་དག་པའི་སྐུ་གསུམ་བསྒྲུབ་པར་བྱ་བའི་ཆེད་དུ། གཞི་འབྲས་གཉིས་དང་རྗེས་མཐུན་གྱི་བསྐྱེད་པ་དང་རྫོགས་པའི་རིམ་པ་སྒོམ་དགོས་པར་གོ་བའི་སླད་དུ་གསུང་བའོ། །དེའི་ཕྱིར་རང་གི་སྒོ་གསུམ་དཀྱི༵་ལ་འཁོ༵་ར་གསུམ་གྱི༵་བསྒྲུབ་གཞིའི་རང༵་བཞི༵ན་ཡེ༵་ནས༵་ཡིན༵་པར༵་རྟོག༵ས་ནས༵་སྒྱུ་འཕྲུལ་དྲྭ་བའི་འཁོར་ལོར་བསྒོམ༵་པ༵་སྟེ༵་། དེའི་ཚུལ་ལ། རྡོ་རྗེ་ལུས་ཀྱི་དཀྱིལ་འཁོར་བསྒོམ་པས་ནི་རྩ་རླུང་ཐིག་ལེ་བྱིན་གྱིས་རློབ་པའི་ཁྱད་པར་གྱིས་བདེ་བ་ཆེན་པོའི་ཏིང་ངེ་འཛིན་མྱུར་དུ་འགྲུབ་པའི་དགོས་ཆེད་ཡོད་ལ། དཀྱིལ་འཁོར་གྱི་རྒྱལ་པོ་མཆོག་གི་ཏིང་འཛིན་གཙོ་བོ་དང་འཁོར་གྱི་རྣམ་པ་མང་པོར་སྤྲོ་བས་ནི། སངས་རྒྱས་ཀྱི་སའི་འཁོར་ངེས་པ་ཕུན་སུམ་ཚོགས་པ་ལམ་དུ་བྱེད་ཅིང༌། ཁ་ཅིག་དེར་མ་ཟད་སྲིད་པའི་གནས་ཀྱི་སྡེ་རིས་དང་བྲན་གཡོགས་རྒྱས་པ་སོགས་འདོད་ཡོན་ལ་ཞེན་པའི་བག་ཆགས་ཀྱང་སྦྱོང་བར་འཆད་དོ། །འོན་ཀྱང་འདིའི་བཀའ་སྲོལ་ཏུ་རྒྱལ་མཆོག་གསུམ་གྱི་ཐ་སྙད་མི་འབྱུང་ངོ༌། །སྤྱིར་རྣམ་བྱང་ཕྱོགས་ཀྱི་མཐར་ཐུག་མི་སློབ་པའི་སས་བསྡུས་ཀྱི་སྟོན་པ་དང་འཁོར་གྱི་རྣམ་པ་ལམ་དུ་བྱེད་པ་ནི་སྔགས་སྤྱིའི་འགྲོ་ལུགས་སུ་བདག་ཉིད་ཆེན་པོ་གཏེར་ཆེན་པྲ་ཛྙཱའི་ཞབས་ཀྱིས། སྐུ་དང་ཞིང་ཁམས་འཁོར་དང་མཛད་པ་ནི། །ཡོངས་སུ་དག་པའི་ཡེ་ཤེས་ལམ་ཉིད་དུ། །བྱེད་དང་མི་བྱེད་སྔགས་དང་ཕ་རོལ་ཕྱིན། །ཞེས་གསུངས་ཤིང་སྔར་བཤད་པས་ཀྱང་རྟོགས་པར་ནུས་སོ། །སྔགས་བླ་མེད་ཀྱི་ཁྱད་ཆོས་གཞིའི་སྲིད་པ་སྐྱེ་འཆི་བར་དོ་གསུམ་དང་རྣམ་མཐུན་དུ་སྒོམ་པ་ཡང་རྣལ་འབྱོར་དེ་དང་དེའི་ཁོག་རགས་པ་ཞིག་བསྟུན་པས་ཆོག་གི །དམིགས་པའི་ཕྲ་ཞིབ་རེ་རེ་ནས་བསྟུན་པ་མི་དགོས་ཏེ། དེ་དགོས་ན་ཐུག་པ་མེད་པར་ཐལ་བའི་ཕྱིར་རོ། །ཞེས་སྨྲ་བ་རྒྱ་གར་གྲུབ་ཆེན་གྱི་ལུགས་སུ་མ་ཟད། སངས་རྒྱས་གཉིས་པའི་ཟབ་གཏེར་གྱི་བདག་པོ་ཆེན་པོ་རྣམས་ཀྱི་བཀའ་སྲོལ་རྨ་མེད་པའང་ཡིན་ཏེ་ཁོ་བོའི་མགྲིན་པ་གཟེངས་མཐོན་པོར་བསྟོད་དེ་བཤད་ཀྱང་མི་ནུས་པ་མིན་ནོ། །མདོར་ན་དེའི་ཕྱིར་དཀྱིལ་འཁོར་སྤྲོ་བ་དེ་འཁོར་བའི་གནས་སྐབས་ན་བུ་དང་བུ་མོ་མང་དུ་འཕེལ་བ་དང་སྡེ་རིས་ཀྱི་རྣམ་གྲངས་མང་བ་ལྟ་བུའི་ཆོས་མཐུན་པའི་ཡུལ་ཞིག་དང་མ་སྦྱར་ཡང་ཆོག་པར་སེམས་སོ། །དེ་ལྟར་ཞར་ལ་འོང་བ་ཉུང་ངུ་ཞིག་བརྗོད་ནས་དཀྱུས་དོན་ནི། སྒྱུ་འཕྲུལ་གྱི་དཀྱིལ་འཁོར་དེ༵་འང༵་དག་པ་སྦྱོར་ཚུལ་གྱི་སྒོ་ནས་བཤད་ན་རྩ་བརྒྱུད༵་གསང་སྙིང་ལས༵། རྡོ་རྗེའི་ཕུང་པོའི་ཡན་ལག་ནི། །རྫོགས་པའི་སངས་རྒྱས་ལྔ་རུ་གྲགས། །སྐྱེ་མཆེད་ཁམས་རྣམས་མང་པོ་ཀུན། །བྱང་ཆུབ་སེམས་དཔའི་དཀྱིལ་འཁོར་ཉིད། །ས་ཆུ་སྤྱན་དང་མཱ་མ་ཀཱི། །མེ་རླུང་གོས་དཀར་སྒྲོལ་མ་སྟེ། །ནམ་མཁའ་དབྱིངས་ཀྱི་དབང་ཕྱུག་མ། །སྲིད་གསུམ་ཡེ་ནས་རྣམ་པར་དག །ཅེས༵་འབྱུང༵་བ་ལྟར་ཤེས་པར་བྱ་སྟེ༵་། དོན་དམ་པར་འཁོར༵་བ༵་དང༵་མྱ༵་ངན༵་ལས༵་འད༵ས་པའི༵་ཆོས༵་ཐམ༵ས་ཅད༵་ཡེ༵་ནས༵་མ༵་སྐྱེས༵་མ་བྱུང་བར་གསུངས་ཏེ། རིགས་པ་དྲུག་ཅུ་པར། འཁོར་བ་དང་ནི་མྱ་ངན་འདས། །དེ་གཉིས་ཡོད་པ་མ་ཡིན་ནོ། །ཞེས་སོགས་གསུངས་ལ༵་། ཐ་སྙད་དུ་ནི་དཀོན་མཆོག་བརྩེགས་པ་ལས། རྟོག་པའི་དབང་གིས་འཇིག་རྟེན་རྣམ་བཏགས་ཏེ། །འདུ་ཤེས་འཛིན་པས་བྱིས་པ་རྣམ་པར་ཕྱེ། །ཞེས་རྟོག་པའི་རང་རྩལ་འབའ་ཞིག་ཏུ་དངོས་པོ་རྣམས་བསྙོན་མེད་དུ་ཡོད་པར་གསུངས་ཤིང༌། དེ་ཡོད་པར་འཇོག་བྱེད་ཀྱང༌། རང་རང་གི་བྱ༵་བ༵་བསླུ་མེད་བྱེད༵་ནུས༵་པའི༵་སྒོ་ནས་སྒྱུ༵་མ༵་བཞིན་ཏུ་སྣང་བ་སུས་ཀྱང་འགོག་མི་ནུས་པའོ། །ཐེག་པ་འདིར་དོན་དམ་པར་རང་བཞིན་མེད་པའི་ལྟ་བ་ཙམ་དུ་མ་ཟད། ཀུན་རྫོབ་པའི་ཆོས་ཕུང་པོ་ལྔའམ་ལྷར་ལྟ་བར་བྱེད་དེ། ལྟ་ཚུལ་ནི། བདེ༵་བར༵་གཤེག༵ས་པ༵་འི་མཚན་ཅན་ཡབ༵་ཡུམ༵་བཅུ༵་ལ༵་སོག༵ས་པ༵འི་བསྒྲུབས་གཞིའི་རང༵་བཞི༵ན་མེའི་བསྒྲུབ་གཞིར་གཙུབ་ཤིང་འགྲོ་བ་བཞིན་ཏུ་ཡེ༵་ནས༵་ཡིན༵་པའི༵་ཚུལ་དུ་ལྟའོ། །འདི་འདྲ་ཞིག་སྔོན་དུའང་སྨོས་མ་ཐག་པ་ཡིན་མོད་ཀྱི། སྐབས་འདིར་གསུངས་པ་ནི་ཚིག་སྔ་མ་དེའི་གསལ་བྱེད་ཡིན་པས་ཟློས་སྐྱོན་མེད་དོ། །འདི་དག་གི་སྐབས་སུ་ཁོ་བོ་ཅག་གི་བླ་མ་རྣམས་ཆོས་ཐམས་ཅད་ཡེ་ཤེས་རང་སྣང་དུ་སྒོ་དུ་མ་ནས་སྒྲུབ་པར་མཛད་དེ། དེའི་ཚུལ་གཏན་ཚིགས་ཀྱི་སྒོ་ནས་སྤྲོས་ན་མཐའ་མེད་ཀྱང་བསྒྲུབ་བྱའི་སྙིང་པོ་ཚིག་གཅིག་ལ་དྲིལ་ན། ཡེ་ཤེས་ཞེས་པ་ཡེ་ཤེས་གང་རུང་རུང་ལ་མི་བྱའི་གཉུག་མའི་སེམས་ལ་བྱའོ། །རང་སྣང་ཞེས་པ་རང་ལ་སྣང་ལུགས་གང་བྱུང་བྱུང་ལ་མི་བྱའི་གཉུག་སེམས་ཀ་ནས་དག་པའི་རང་རིག་ནང་གསལ་ཤིན་ཏུ་ཕྲ་བའི་ཡེ་ཤེས་ཀྱི་རང་སྣང་ལ་བྱའོ། །དེའི་ཕྱིར་ཀ་བ་དང་བུམ་པ་ལ་སོགས་པའི་དངོས་པོ་རྣམས་དཀར་སེར་དམར་ལྗང་གི་ལྷ་དང་ལྷ་མོ་བཞིན་ལག་ཅན་ཤ་སྟག་པར་སྒྲུབ་མི་དགོས་ཀྱི། འོད་གསལ་གྱི་སེམས་འབའ་ཞིག་མེད་ན་འཁོར་འདས་ཐམས་ཅད་སྣོད་དུ་མི་རུང་བ་དང༌། དེ་ཡོད་པས་ཆོས་ཐམས་ཅད་དེའི་རོལ་རྩལ་དུ་ཤར་བར་གྲུབ་ན་དེ་ཙམ་ཞིག་གིས་ཡེ་ཤེས་རང་སྣང་གི་གནད་དཔྱིས་ཕྱིན་པར་ཟིན་པ་ཡིན་ཏེ། མདའ་བསྣུན་ཞབས་ཀྱིས། སེམས་ཉིད་གཅིག་པུ་ཀུན་གྱི་ས་བོན་ཏེ། །གང་ལ་སྲིད་དང་མྱ་ངན་འདས་འཕྲོ་བ། །འདོད་པའི་འབྲས་བུ་སྟེར་བར་བྱེད་པ་ཡི། །ཡིད་བཞིན་ནོར་འདྲའི་སེམས་ལ་ཕྱག་འཚལ་ལོ། །ཞེས་གསུངས་པ་འདི་ཁུངས་སུ་དྲངས་ཏེ་ཆོས་དབྱིངས་མཛོད་ཀྱི་འགྲེལ་པར་ཐམས་ཅད་མཁྱེན་པ་དྲི་མེད་འོད་ཟེར་གྱིས་རྒྱ་ཆེར་ཇི་སྐད་གསུངས་པ་བཞིན་ནོ། །དེ་ལྟར་འཆད་པ་དེ་སྐབས་འདིར་འབྲེལ་བ་གྲུབ་ཚུལ་སོགས་མང་བས་མ་བཀོད་དོ། །དེ་ལྟར་ཆོས༵་ཐམ༵ས་ཅད༵་རང༵་བཞིན༵་གྱིས༵་མྱ༵་ངན༵་ལས༵་འད༵ས་པ༵་སྟེ༵་། ཆེན༵་པོ༵་ལྔ༵་ནི༵་ཡུམ༵་ལྔའི༵་རང༵་བཞི༵ན་སྣང་བའི་དབྱིངས་དང༌། སྲ་བའི༴ མཉེན་པའི༴ དྲོ་བའི༴ བསྐྱོད་པའི་དབྱིངས་ཞེས་རྩ་རྒྱུད་ལེའུ་དང་པོར་གསུངས་པ་བཞིན་ནོ། །དེ་བཞིན་དུ་ཕུང༵་པོ༵་ལྔ༵་ནི༵་རིག༵ས་ལྔའི༵་སང༵ས་རྒྱས༵་ཏེ། རྣམ་པར་ཤེས་པའི་རྒྱལ་པོ་མི་བསྐྱོད་པ་ལ་སོགས་པའོ། །མཐོང་བ་དང་ཐོས་པ་སོགས་རྣམ༵་པར༵་ཤེས༵་པ༵་བཞི༵་ནི༵་ས་སྙིང་ལ་སོགས་པ་ཐུགས་ཀྱི་སེམ༵ས་དཔའ༵་བཞི༵འི་རང༵་བཞི༵ན། གཟུགས་ཀྱི་སྐྱེ་མཆེད་ནས་རོའི་ཁམས་ཀྱི་བར་ཡུལ༵་བཞི༵་ནི༵་སྒེག་མོ་ལ་སོགས་པ་མཛེ༵ས་པའི༵་ལྷ༵་མོ༵་བཞི༵འི་རང༵་བཞི༵ན། མིག་ཤེས་སོགས་ཀྱི་ཐུན་མོང་མ་ཡིན་པའི་བདག་རྐྱེན་དབ༵ང་པོ༵་གཟུགས་ཅན་པ་བཞི༵་ནི༵་བྱམ་པ་ལ་སོགས་པ་སྐུའི་བྱང༵་ཆུབ༵་སེམ༵ས་དཔ༵འ་བཞི༵འི་རང༵་བཞི༵ན། འདས་པ་ནས་མ་ངེས་པའི་བར་གྱི་དུས༵་བཞི༵་ནི༵་སྤོས་མ་ལ་སོགས་པ་མཆོད༵་པའི༵་ལྷ༵་མོ༵་བཞི༵འི་རང༵་བཞི༵ན། ལུས༵་ཀྱི༵་དབ༵ང་པོ༵་དང༵་ལུས་ཀྱི་རྣམ༵་པར༵་ཤེས༵་པ༵་དང༵་ཡུལ༵་རེག་བྱའི་སྐྱེ་མཆེད་དང༵་དེ༵་ལས༵་བྱུང༵་བའི༵་བྱང༵་ཆུབ༵་ཀྱི༵་སེམ༵ས་སམ་དེ་མ་ཐག་རྐྱེན་ཡིད་ཤེས་རྣམས་ནི༵་རྣམ་རྒྱལ་སོགས་ཁྲོ༵་བོ༵་བཞི༵འི་རང༵་བཞི༵ན། རྟག༵་ཆད༵་བདག་མཚན་གྱི་མུའམ༵་མཐའ་བཞི༵་ནི་ལྕགས་ཀྱུ་མ་ལ་སོགས་པ་ཁྲོ༵་མོ༵་བཞིའི༵་རང༵་བཞི༵ན་དུ་གསུངས་སོ༑ ༑ཡིད༵་ཀྱི༵་རྣམ༵་པར༵་ཤེས༵་པ༵་ནི༵་བྱང༵་ཆུབ༵་ཀྱི༵་སེམ༵ས་རྡོ༵་རྗེ༵་སྟེ་རིགས་དྲུག་པ་ཀུན༵་ཏུ༵་བཟང༵པོ༵འི་རང༵་བཞི༵ན་ཞེས་གསུངས་པའི་སྐབས་འདིའི་ཡིད་ཀྱི་རྣམ་ཤེས་དང་སྨོས་མ་ཐག་པའི་ཡིད་ཤེས་གཉིས་ཀྱི་ཁྱད་པར་ནི༑ རྣམ་པར་ཤེས་པའི་ཁམས་དྲུག་པ་ནི་འདིར་གསུངས་པ་ཡིན་ཞིང༌། དབང་པོའི་ཁམས་དྲུག་པ་ནི་སྨོས་མ་ཐག་པ་ཡིན་ཏེ། འོན་ཀྱང་ལྷ་དང་དག་སྦྱོར་གྱི་ཚེ་སྦྱང་གཞི་རྣམས་རེ་རེ་ནས་སྣེ་མོ་བ་བསྐོ་བ་ལྟར་མི་འདྲ་བ་ཤ་སྟག་མི་དགོས་ཏེ། གཟུགས་ཕུང་རྣམ་སྣང་དང༌། དེའི་ཡ་གྱལ་འབྱུང་བཞི་ཡུམ་བཞིར་གསུངས་པའི་གསར་རྙིང་གི་རྒྱུད་སྡེ་ཆེས་མང་བ་བཞིན་ནོ། །ཡིད་ཀྱི་རྣམ་ཤེས་དེའི་ཡུལ༵་ཆོས༵་ཁམས་འདུས༵་བྱས༵་པ་ཚོར་བ་འདུ་ཤེས་འདུ་བྱེད་རྣམས་རིག་མིན་གཟུགས་བཞི་དང༵་འདུས༵་མ༵་བྱས༵་འགོགས་པ་གཉིས་དང་ནམ་མཁའ་སྟེ་བདུན་ནི༵་ཆོས༵་བྱ༵་བ༵་མོ༵་ཀུན༵་ཏུ༵་བཟང༵་མོའི༵་རང༵་བཞི༵ན་ཏེ༵་། བཀའ་སྲོལ་ཏུའང་འདི་སྐོར་ལ་དཔྱད་པ་མཛད་པ་ཆེར་མི་སྣང་ངོ༌། །དེ་ལྟ་བུའི་ཕུང་ཁམས་སྐྱེ་མཆེད་དེ༵་དག༵་ཀྱང༵་ཡེ༵་ནས༵་མངོན༵་པར༵་རྫོག༵ས་པར༵་སང༵ས་རྒྱས༵་པའི༵་རང༵་བཞི༵ན་ཡིན༵་གྱི༵་། ལམ༵་གྱིས༵་གསར་དུ་བསྒྲུབ༵་པ༵་ལ་ལྟོས་པ་མ༵་ཡིན༵་ཏེ། རྫོགས་པ་ཆེན་པོ་འདུས་མ་བྱས་སུ་གསུངས་པའི་དོན་ཡང་གཉུག་མར་གནས་ཤིང་འཆི་འཕོ་དང་བྲལ་བའི་དོན་ཡིན་གྱི། ནམ་མཁའ་ལྟ་བུའི་མེད་དགག་ཅིག་ལ་མི་བྱ་ཞེས་སྨྲའོ། །སངས་རྒྱས་ལུགས་ནི། ས་ར་ཧས། མཁས་པ་དག་ནི་བསྟན་བཅོས་འཆད་པ་ན། །ལུས་ལ་ཡེ་ཤེས་ཡོད་པ་མ་རྟོགས་སོ། །ཞེས་ཤིན་ཏུ་རྟོགས་པར་དཀའ་བའི་འོད་གསལ་གྱི་སེམས་ངོ་བོ་དྲི་མའི་རྣམ་ལྡན་དུ་སྐྱེ་མ་མྱོང་བའི་ངང་ཚུལ་ཅན་དེ་སེམས་ཅན་ཐམས་ཅད་ཀྱི་རྒྱུད་ལ་ཏིལ་འབྲུ་སྣུམ་གྱིས་ཁྱབ་པ་བཞིན་འདུ་འབྲལ་སྐད་ཅིག་ཀྱང་ཡོད་མི་སྲིད་དེ། ཉི་ཟླ་ཁ་སྦྱོར་ལས། ཙིཏྟ་རིན་ཆེན་གཞལ་ཡས་ན། །སྟོང་གསལ་འདུས་པ་སྐུ་ཡི་གདངས། །ཞེས་སོགས་གསུངས་པ་དང་། གྲོལ་བའི་ཐིག་ལེ་ལས། དེར་གསལ་མི་ཤིགས་ཐིག་ལེའི་གཟུགས། །ཡུངས་ཀར་གྱི་ནི་འབྲུ་ཚོད་ཙམ། །འོད་ཟེར་སྣ་ལྔ་འབར་འཛགས་པའི། །ཚུལ་ཅན་སྙིང་དབུས་རྟག་ཏུ་གནས། །ཞེས་བཤད་དོ། །དེ་འདྲའི་སེམས་ཀྱི་གཏུགས་པ་མཐར་ཐུག་དེའི་ཆོ་འཕྲུལ་འབའ་ཞིག་ཏུ་སེམས་ཅན་རྣམས་ཀྱི་ནང་གི་ཁམས་དྲུག་དང༌། ཕྱི་རོལ་གྱི་གནས་རིས་དང་ལོངས་སྤྱོད་ལ་སོགས་པ་ཐམས་ཅད་ཡོངས་སུ་བཀོད་ཅིང་ཚེ་རབས་སྔ་མའི་ཕུང་པོ་རྣམས་འོད་གསལ་ཏུ་ཐིམ་ནས་སླར་ཡང་ལས་ཀྱི་དབང་གིས་འཇིག་རྟེན་ཕྱི་མའི་བར་སྲིད་ཀྱི་ཕུང་པོ་རྣམས་སེམས་དེའི་རང་གདངས་སུ་ཤར་བའི་ཚུལ་གྱིས། ཐོག་མ་མེད་པའི་འཁོར་བའི་འཁོར་ལོ་རྣམས་སུ་སྐྱེ་འཆི་འགལ་མེའི་རྩེད་མོ་ལྟར་སྣང་བར་ཟད་པས། སྣང་གཞིའི་སེམས་མཐར་ཐུག་གི་ངོས་ན་གོས་པ་རྡུལ་ཆ་ཙམ་ཡང་མེད་ལ། ནམ་ཞིག་ལས་རླུང་གི་གཡོ་བ་བག་ཆགས་དང་བཅས་པ་ཐམས་ཅད་སླར་མི་ལྡང་བའི་ཚུལ་ཏུ་ཉེ་བར་ཞི་བ་དེའི་ཚེ། སེམས་གཉུག་མ་མི་ཤིགས་པའི་ངོ་བོ་ཉིད་དེ་ཀ་རང་རྣམ་པ་ཐམས་ཅད་མཁྱེན་པའི་ཡེ་ཤེས་སུ་འགྲོ་བ་མ་གཏོགས་ཐོབ་བྱའི་ཡོན་ཏན་གཞན་ཚོལ་ལ་རག་མི་ལས་པའི་རྒྱུ་མཚན་གྱིས་གཞིའི་ཆོས་སྐུ་དང༌། གཞིའི་སངས་རྒྱས་ལ་སོགས་པར་མིང་བཏགས་པ་སྟེ། མཐའ་ཆོད་པར་ན་སེམས་དེའི་རྩལ་ལས་སྣང་བ་རྣམ་པ་གསུམ་སྐྱེ་ཚུལ་དང༌། དེ་ལས་ཀུན་རྟོག་གི་རྒྱུ་བ་འཕེལ་ཚུལ་དང༌། དེ་ལས་ཀྱང་སྒོ་ལྔའི་ཤེས་པ་རིམ་གྱིས་རྒྱས་ཚུལ་དང༌། ཀུན་རྟོག་ཉོན་མོངས་པ་རྣམས་ལ་བརྟེན་ནས་ལས་གསོག་ཚུལ་དང༌། ལས་དེའི་དབང་གིས་འགྲོ་བ་སོ་སོའི་སྐྱེ་གནས་ཐམས་ཅད་དང་གནས་མཐོ་དམན་ཐམས་ཅད་ཡོངས་སུ་འགྲུབ་པའི་ཚུལ་དང༌། དེ་དག་གི་གནས་སྐབས་ཐམས་ཅད་དུ་གཉུག་མའི་སེམས་འཕོ་འགྱུར་མེད་ཅིང་རྨི་ལམ་གཉིད་ཀྱི་ཆོ་འཕྲུལ་དུ་ཤར་བ་ལྟར་སྣོད་བཅུད་དེའི་སྒྱུ་འཕྲུལ་དུ་སྣང་ཚུལ་དང༌། དེ་ཉིད་ཀྱི་རྒྱུ་མཚན་ལས་སྒྱུ་མ་ལྟ་བུའི་ཏིང་ངེ་འཛིན་གྱིས་རྣམ་པར་སྦྱངས་བ་ན་སྣོད་བཅུད་གང་ནའང་མ་དག་པ་གཤིས་ལ་ཞུགས་པ་ཞིག་མེད་ཚུལ་སོགས་རྒྱུད་འགྲེལ་ཆེན་མོ་རྣམས་སུ་བལྟ་བར་བྱ་ལ། གསང་བདག་ཞལ་ལུང་དང་རོང་ཟོམ་ཞབས་ཀྱིས་སྣང་བ་ལྷར་སྒྲུབ་ན་མཐུན་སྣང་གི་བལྟ་བྱ་ལ་སྦྱར་བའི་རིགས་པ་གཞན་ཡང་གསུངས་མོད་ཀྱི། རྡོ་རྗེའི་ལམ་ལ་ངེས་པ་སྟེར་བ་ལ་འདི་རང་འགགས་ཆེ་བར་གོའོ། །དེ༵་ལྟར༵་ན་ཕྱོག༵ས་བཅུ༵་དུས༵་གསུམ༵་དང༵་ཁམ༵་ས་དུ་མ་ལ༵་སོག༵ས་པ༵་འདུ༵ས་བྱས༵་དང༵་འདུས༵་མ༵་བྱས༵་པའི༵་ཆོས༵་ཐམ༵ས་ཅད༵་རང༵་གི༵་སེམ༵ས་འོད་གསལ་བའི་ཡེ་ཤེས་ཀྱི་རྣམ་རོལ་ལས༵་གུད༵་ན༵་མེད༵་དེ༵་ཞེས་མ་བཤད་ན། ཁ་ཅིག་འདི་ལ་བརྟེན་ནས་ཚིག་འདིའི་བསྟན་དོན་སྣང་བ་སེམས་སུ་ངོ་སྤྲད་པ་འདྲ་བ་ཅིག་ལ་འཁྲུལ་བའི་དོགས་པ་ཡང་འབྱུང་ཆོག་པའི་ཚོད་དུ་སྣང་མོད། འོན་ཀྱང་འཇིག་རྟེན་ན་ལྷ་སྦྱིན་གྱི་བུམ་པ་ཙེ་རྟས་བརྐུས་ན། ལྷ་སྦྱིན་གྱིས་དེ་ལ་འདི་སྐད་དུ། ངའི་བུམ་པ་དེ་ནི་ཁྱོད་ལས་གཞན་ན་མེད་དོ་ཞེས་ཟེར་མོད་ཀྱི། ཁྱོད་ལྟོ་ལྡིར་ཡིན་ཟེར་བ་མིན་པར། བུམ་པ་རྐུ་མཁན་དེ་ཁྱོད་ཡིན་ཞེས་བསམ་ནས་ཡིན་ཅིང༌། ཚིག་དེ་སྐྱོན་ཅན་ཡང་ཡིན་མི་དགོས་ཏེ། བརྡའི་ཐ་སྙད་དེ་ཙམ་གྱིས་ཕ་རོལ་པོ་ལ་དོན་དེ་སྟོན་ནུས་པའི་ཕྱིར་རོ༑ ༑གཞན་དུ་ན་མགོན་པོ་ཀླུས། གཟུགས་ཀྱི་དངོས་པོ་མིང་ཙམ་ཕྱིར། །ཞེས་གསུངས་པའང་མི་རིགས་པར་ཐལ་བར་འགྱུར་ཏེ། གཟུགས་མིང་གིས་བཞག་པ་ཙམ་ལ་དགོངས་པ་མིན་ན་གཟུགས་མིང་ཡིན་ཚུལ་མེད་པའི་ཕྱིར་ཏེ། མིང་ལྡན་མིན་འདུ་བྱེད་ཡིན་པའི་ཕྱིར་རོ། །ཆོས་ཐམས་ཅད་སེམས་ཉིད་ཀྱི་རོལ་པར་ཤར་བའི་ཁུངས་སུ། ཇི༵་སྐད༵་དུ༵་ཞེས་འཇིག་རྟེན་གསུམ་རྒྱལ་ལས། རང་སེམས་སོ་སོར་རྟོག་པ་ནི། །སངས་རྒྱས་བྱུང་བ་དེ་ཉིད་དོ། །འཇིག་རྟེན་གསུམ་པོ་དེ་ཉིད་དོ། །འབྱུང་བ་ཆེ་རྣམས་དེ་ཉིད་དོ། །ཞེས་འབྱུང་ངོ་། །གཞན་ཡང་ཇི་སྐད་དུ། ཆོས་ཐམས་ཅད་ནི་སེམས་ལ་གནས་སོ། །སེམས་ནམ་མཁའ་འམ་ལྟ་བུའི་སྟོང་ཉིད་ཀྱི་གཞི་ལ་གནས་སོ༑ ༑ནམ་མཁའ་འམ་ལྟ་བུའི་གནས་ལུགས་དེ་ནི་མཚན་མ་ཅི་ལའང་མི་གནས་སོ། །ཞེས་འབྱུང་བ་དང༌། ཆོས་ཐམས་ཅད་ནི་ངོ་བོ་ཉིད་ཀྱིས་སྟོང་པའོ། །ཆོས་ཐམས་ཅད་ནི་གདོད་མ་ནས་རང་བཞིན་རྣམ་པར་དག་པའོ། །ཆོས་ཐམས་ཅད་ནི་ཡོངས་ཀྱིས་འོད་གསལ་བའོ། །ཆོས་ཐམས་ཅད་ནི་ཏིང་འཛིན་གྱིས་མྱ་ངན་ལས་འདས་པའོ། །ཆོས་ཐམས་ཅད་ནི་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་པའོ། །ཞེས་གསུངས་སོ། །འདིར་ལུང་གི་དྲང་མཚམས་ལ་བཤད་དགོས་པ་མང་ཡང་ཕལ་ཆེར་སྤྱོད་བསྡུས་སྒྲོན་མེའི་རིགས་པ་ཞིག་དང་སྦྱར་ན་འཆར་ཐུབ་པར་སྣང་སྟེ་སྤྲོས་པ་བསྐྱུངས་སོ། །འདི༵་ནི༵་རྫོ༵གས་པ༵་ཆེན༵་པོའོ༵། །ཞེས་པ་ལ་བཅུ་བདུན་རྒྱུད་སོགས་ནས་ཡོངས་སུ་གྲགས་པའི་རྫོགས་པ་ཆེན་པོའི་སྒྲ་བཤད་ནི་རེ་ཞིག་མི་བརྗོད་ལ། འདིར་བསྟན་གྱི་རྫོག༵ས་པ༵་ཆེན༵་པོའི༵་དོན་ནི༵་ཏིང་ངེ་འཛིན་གྱི་སྤྲོས་པ་ཆེ་ཞིང་རྒྱས་པའི་དོན་ནོ་ཞེས་བཤད་ཟིན་མོད་དེ། འདི་ཐད་ཀྱི་མཆན་དཀྱུས་ཤོར་ཞིག་ན་བསོད་ནམས་དང་ཡེ་ཤེས་ཀྱི་ཚོགས་རྫོགས་པ་འབྲས་བུའི་ཆོས་ལྷུན་གྱིས་གྲུབ་པའི་དོན་ཏོ་ཞེས་བཤད་པ་ཡང་ཡིད་ངོར་མི་འགྲིགས་པ་མིན་ནོ། །གཉིས་པ་ལའང༌། རྟོགས་པ་བཞི་དང༌། མཚན་ཉིད་གསུམ་དང༌། ཡན་ལག་བཞི་དང༌། དཀྱིལ་འཁོར་དུ་འཇུག་པའི་རིམ་པ་སྟེ་ཚུལ་བཞིས་བཤད་པ་ལས། དང་པོ་ནི། རྟོགས་པ་རྣམ་པ་བཞི་ནི། རྩ་རྒྱུད་ལེའུ་བཅུ་གཅིག་པ་ལས། རྒྱུ་གཅིག་པ་དང་ཡིག་འབྲུའི་ཚུལ། །བྱིན་གྱིས་རླབས་དང་མངོན་སུམ་པ། །རབ་ཏུ་རྟོགས་པ་རྣམ་བཞི་ཡིས། །ཐམས་ཅད་མངོན་རྫོགས་རྒྱལ་པོ་ཆེ། །ཞེས་པ་ལྟར་རྒྱུ་གཅིག་པར་རྟོགས་པ་དང༌། ཡིག་འབྲུའི་ཚུལ་གྱིས་རྟོགས་པ་དང༌། བྱིན་གྱིས་བརླབས་ཀྱིས་རྟོགས་པ་དང༌། མངོན་སུམ་པར་རྟོགས་པའོ། །འདི་རྣམས་སྒྱུ་འཕྲུལ་དྲྭ་བའི་རྒྱལ་པོ་བསྒྱུར་བའི་ཏིང་ངེ་འཛིན་གྱི་སྐབས་དང་སྦྲེལ་ཏེ་གསུངས་ཤིང༌། དེའི་ནང་ནས་ཀྱང་ལེའུ་དེའི་བསྟན་དོན་ཚོགས་ཀྱི་དཀྱིལ་འཁོར་ལ་བརྟེན་ནས་གང་ཞིག་བསྒྲུབ་བྱའི་གཙོ་བོ་བསྟན་པར་སྣང་ཞིང༌། མངོན་རྫོགས་རྒྱལ་པོའི་དོན་ཡང་ཟུང་འཇུག་སྒྱུ་འཕྲུལ་དྲྭ་བའི་གོ་འཕང་ལ་བྱེད་དགོས་པར་མངོན་ཏེ། སྤྱིར་སྒྱུ་འཕྲུལ་དྲྭ་བའི་དོན་ནི། འཕགས་པ་ཀླུས། དེ་བཞིན་ཐམས་ཅད་སྟོང་གསལ་ལས། །སྒྱུ་འཕྲུལ་དྲྭ་བ་འབྱུང་བར་བཤད། །ཅེས་རྡོ་རྗེ་འཛིན་པའི་ཡེ་ཤེས་ཀྱི་སྐུ་ལ་གསུངས་སོ། །བླ་མ་སྔ་མ་རྣམས་འདི་བཞི་ལ་གཏན་ཚིགས་བཞིར་མཛད་མོད། ཤེས་པར་འདོད་ན་གཞན་དུ་བལྟའོ། །དེ་ལ་རྒྱུ་གཅིག་པར་རྟོགས་པ་ནི། ཆོས་ཐམས་ཅད་དོན་དམ་པར་མ་སྐྱེས་ཞེས་པ་འདི་ལ་རྒྱལ་དབང་དྲི་ཐོད་འོད་ཟེར་དང་རྡོ་རྗེ་སེམས་དཔའ་འོད་གསལ་སྣ་ཚོགས་རང་གྲོལ་གྱི་གཞུང་ན་དོན་དམ་བདེན་པའི་འཇོག་ཚུལ་ཐུན་མོང་མ་ཡིན་པ་འོད་གསལ་གྱི་སེམས་དེ་ཁོ་ན་ཉིད་དུ་ཞུགས་པ་ལ་གསལ་བར་བཤད་པས་འདིར་ཡང་དེའི་ངོར་ཤེས་བྱ་གང་ཡང་མི་རྙེད་པའི་དོན་ནོ། །དེའི་ཕྱིར་རིག་པའི་ཡེ་ཤེས་ཆོས་ཐམས་ཅད་དང་སོ་སོ་མ་ཡིན་པ་ཞེས་གསུངས་ཏེ། འཇིགས་མེད་འབྱུང་གནས་ཞབས་ཀྱིས་ཀྱང་ལྷན་སྐྱེས་ཀྱི་བདེ་ཆེན་ཤེས་བྱའི་དེ་ཉིད་དང་རོ་གཅིག་ཏུ་ཞུགས་པའི་སྒོ་ནས་ཤེས་བྱ་ཐམས་ཅད་ཡེ་ཤེས་ཀྱིས་ཁྱབ་པའི་ཚིག་བསྟན་པར་གསུངས་པ་བཞིན་ནོ༑ ༑ཀུན་རྫོབ་ལའང་གང་གི་ཟླས་འདྲེན་པའི་དོན་དམ་འཇོག་ཚུལ་གྱི་དབྱེ་བས་མི་མཚུངས་པ་འགའ་ཞིག་ཡོད་པ་ལས། འདིར་ནི་འོད་གསལ་བའི་ཡེ་ཤེས་ལ་དོན་དམ་གྱི་སྒྲས་བསྙད་པ་དེའི་ཟླས་དྲང་བ་ཡིན་པས། ཡེ་ཤེས་དེའི་སྣང་ཆའམ་དེའི་རང་མདངས་སམ་དེའི་སྙིང་པོ་མི་ཤིགས་པའི་རླུང་ཕྲ་བ་ལས་བྱུང་བའི་སྐུའི་མཐར་ཐུག་ཅིག་ལ་བྱ་སྟེ། དེ་འདྲའི་ཀུན་རྫོབ་ཏུ་ནི་རིག་པའི་སྣང་བརྙན་ཆུ་ཟླ་དང་སྒྱུ་མའི་མཚན་ཉིད་བཏགས་པ་ཙམ་དང་བརྟེན་ནས་འབྱུང་བ་ཙམ་སྟེ་སྣང་ལ་རང་བཞིན་མེད་པའི་ངང་ཚུལ་ཅན་དུ་གྲུབ་པ་དང༌། ཤེས་བྱ་ཐམས་ཅད་ཀྱང་དེ་དང་འདྲ་བར་སེམས་ཀྱི་གདངས་ཕྱིར་གསལ་ཐུགས་རྗེའི་སྣང་ཆར་ཤར་བ་ཙམ་གྱི་ཆོས་སུ་ཟད་པས་ན་ཚུལ་གཅིག་པའི་སྒོ་ནས་སོ་སོ་མ་ཡིན་པ་དང༌། ཞེས་པ་སྟེ། འདི་ལ་ཆོས་ཐམས་ཅད་རང་བཞིན་མེད་པའི་རྟེན་འབྲེལ་ཙམ་དུ་འདྲ་བ་ལས་ལྷག་པའི་སྔགས་བླ་མེད་ཀྱི་ལུགས་ཐུན་མོང་མ་ཡིན་པས། རྒྱུད་གཞན་ལས་ཀྱང༌། སྒྱུ་མ་རླུང་གིས་ཁམས་གསུམ་པ། །རྨི་ལམ་རྨི་བ་བཞིན་དུ་མཐོང༌། །ཞེས་པ་ལྟ་བུའི་གོ་ལུགས་ཁྱད་པར་ཅན་ཞིག་ཡོད་པ་དེ་ཤེས་དགོས་སོ། །དེ་ལྟར་འོད་གསལ་བའི་རིག་པ་ཤེས་བྱའི་ཡིན་ལུགས་དང་མཉམ་པར་སྦྱོར་བའི་ཟུང་འཇུག་ཅིག་དང༌། དེའི་རང་མདངས་ནི་འོད་གསལ་ལྷུན་གྲུབ་ཀྱི་དཀྱིལ་འཁོར་ཐམས་ཅད་ཤེས་བྱ་མཐའ་དག་དང་ཚུལ་གཅིག་ཏུ་གྱུར་པའི་རོ་གཅིག་གཉིས་བསྟན་ནས། ད་ནི་ཐུན་མོང་མ་ཡིན་པའི་ཟུང་འཇུག་དཔྱིས་ཕྱིན་པ། ཀླུ་སྒྲུབ་ཞབས་ཀྱིས། སྣང་བ་དང་ནི་སྟོང་པ་དག །སོ་སོའི་ཆར་ནི་ཤེས་གྱུར་ནས། །གང་དུ་ཡང་དག་འདྲེས་གྱུར་པ། །ཟུང་དུ་འཇུག་པ་དེ་བཤད་དོ། །ཞེས་དང་། གཏེར་ཆེན་པྲ་ཛྙཱའི་ཞལ་སྔ་ནས་ཀྱང༌། རིམ་པ་བཞི་པའི་དོན་གྱི་འོད་གསལ་དང༌། །དབྱེར་མེད་ལྷན་ཅིག་སྐྱེས་པ་བདེ་བའི་དངོས། །བདེན་གཉིས་ཟུང་དུ་འཇུག་པའི་ཧེ་རུ་ཀ །ཞེས་ཇི་སྐད་དུ་གསུངས་པ་བཞིན་གྱི་ཚུལ་དེ་བསྟན་པར་བཞེད་ནས། ཆོས་ཐམས་ཅད་དང་རོ་གཅིག་པའི་གཅིག་གཞི་སྒྱུ་མ་ལྟ་བུའི་དཀྱིལ་འཁོར་དང༌། མ་སྐྱེས་པའི་ཡུལ་ཅན་འོད་གསལ་དེ་གཉིས་ཟུང་འཇུག་མཉམ་པ༵་ཉིད༵་དུ་གྱུར་ཚུལ་ནི། བདེ་བ་ཆེན་པོའི་ཡེ་ཤེས་གཅིག་ཉིད་དག་པའི་བརྟན་གཡོ་སྣ་ཚོགས་པའི་གར་གྱིས་ཆུ་ཟླ་ལྟར་རོལ་པ་ལ་སྒྱུ་མ་སྣ་ཚོགས་སུ་སྣང་ཞིང་ཞེས་གསུངས་སོ། །དེ་ལྟར་སྣང་བ་དེའང༌། རྩ་རྒྱུད་དུ། བྱ་བྱེད་ནུས་པའི་གཟི་བྱིན་གྱིས། །སྒྱུ་མ་རྡོ་རྗེ་བཙན་པོ་མཆོག །ཅེས་པ་ལྟར་དོན་གཉིས་སྒྲུབ་པའི་བྱ་བ་རླབ༵་པོ་ཆེ་བྱེད༵་ནུས༵་པ༵་དང༵་། སྣང་དུས་ཉིད་ནས་སྒྱུ༵་མ༵་ལྟ་བུའི་སྣང་བ་དེ་ཉིད༵་ལ་རྣམ་སྨིན་གྱི་ཕུང་པོའི་རྡུལ་དང་གཞན་ཡང་རུང་སྟེ་མདོར་ན་རྡུལ་ཕྲ་རབ་ཀྱི་ངོ༵་བོ༵་ཉིད་དམ་ཡེ་ཤེས་ལས་ངོ་བོ་གཞན་དུ་གྱུར་པའི་རང་བཞིན་གྱི་ཅི་ཡང་མེད་དེ་དོན་གཞན་དུ་མ་སྐྱེས་པས་ཀུན་རྫོབ་པའི་སྐུ་དང་དོན་དམ་པའི་བདེ་བ་ཐ་དད་མེད་པར༵་གནས་པའམ་དབྱེར་མེད་པས་ན། དེ་འདྲའི་བདེན་གཉིས་ཟུང་འཇུག་ཐོབ་པ་ན་རྒྱུ་གཅིག་པར་རྟོགས་པའོ། །སྤྱིར་ཆོས་དེ་རྟོགས་ན་ཆོས་དེ་ཐོབ་མི་དགོས་པ་ཡོད་མོད་ཀྱང༌། འདི་ནི་བཤད་མ་ཐག་པར་མངོན་ནོ། །ཡིག་འབྲུའི་ཚུལ་གྱིས་རྟོགས་པ་ནི། ཆོས་ཐམས་ཅད་མ་སྐྱེས་པའི་རང་བཞིན་དང་གཉིས་སུ་མེད་པར་ཞུགས་པའི་བདེ་ཆེན་ནི་ངེས་དོན་གྱི་ཨ་སྟེ་གསུང་གི་རང་བཞིན། མ་སྐྱེས་པ་ཉིད་དམ་ཡེ་ཤེས་རྣམ་པར་མི་རྟོག་པ་ཡུལ་མིང་ཡུལ་ཅན་ལ་བཏགས་པ་དེ་ཉིད་ཀུན་རྫོབ་ཕྱག་རྒྱ་སྒྱུ་མར་སྣང་ཞིང་བྱ་བ་བྱེད་ནུས་པ་ནི་ཨོ་སྟེ་སྐུའི༵་རང༵་བཞི༵ན། དེ་ལྟར་རྟོགས་པའི་རིག་པ་ལྷན་སྐྱེས་བདེ་མཆོག་སྒྱུ་མའི་ཡེ་ཤེས་མཐའ་དང་དབུས་མེད་པ་ཞེས་ལུས་ལྟ་བུའི་སེམས་དང༌། སེམས་ལྟ་བུའི་ལུས་ཀྱི་རྣམ་བཞག་དཔལ་ལྡན་ཟླ་བ་གྲགས་པས་བཀྲལ་ནས་གསུངས་པ་ལྟ་བུ་ནི་ༀ་གྱི་དོན་ཏེ་ཐུགས་ཀྱི་རང་བཞིན་དུ་རྟོགས་པའོ། །དེའང་ཡེ་ཤེས་ལ་གསུང་དང༌། གཉུག་མའི་ལུས་ལ་སྐུ་དང༌། དེ་གཉིས་དབྱེར་མེད་ལ་ཐུགས་སུ་གསུངས་པ་ནི་སེམས་དཔའ་སུམ་བརྩེགས་ཀྱི་གོ་བའི་སྟེང་ནས་ཤེས་པར་བྱ་བ་ཁོ་བོའི་ཞེ་འདོད་ཡིན་ཏེ། འདི་ལྟར་འོད་གསལ་ནམ་མཁའི་གཟུགས་ཅན་གྱི་སེམས་ཆེན་པོ་དེའི་བཞོན་པ་གཞོམ་དུ་མེད་པའི་ན་དཱ་ནི་རྟོགས་པ་གསུམ་པའི་སྐབས་སུ་འཆད་འགྱུར་ལྟར་ཡེ་ཤེས་སེམས་དཔའ་ལོངས་སྤྱོད་རྫོགས་པའི་སྐུ་མཚན་དཔེ་འབར་བའི་རྣམ་པར་བཞེངས་པ་ཡིན་ལ། དེ་དང་ངོ་བོ་དབྱེར་མི་ཕྱེད་པའི་བདེ་བ་ཆེན་པོ་གཉུག་མའི་ཡིད་ནི་ཏིང་ངེ་འཛིན་སེམས་དཔའ་ངེས་པ་དོན་གྱི་ཡི་གེ་ཡིན་ཏེ། བཤད་རྒྱུད་རྒྱ་མཚོ་ལས༑ མི་ཤིག་ཐིག་ལེ་ཝཾ་དབྱིབས་ཅན། །རང་ལ་བདེ་བ་གསལ་བྱེད་དག །ཅེས་སོ། །དེ་གཉིས་ཀའང་དམ་ཚིག་སེམས་དཔའི་ཐུགས་ཀྱི་གསང་བར་ངེས་པས་ན་ཐུགས་སུའང་གསུངས་པ་སྐུ་ཐུགས་དུ་མ་རོ་གཅིག་ལ་གོ་བ་སྐྱེད་པའི་ཆེད་དུ་ཡིན་ཞེས་སྨྲའོ། །རྟོགས་ཚུལ་འདིས་ཀྱང་རྒྱུ་གཅིག་པའི་སྐབས་ཀྱི་བརྗོད་བྱ་དེ་སླར་ཡང་དཔེས་བསྟན་པ་ཙམ་ཞིག་མཛད་པའོ། །བྱིན་གྱིས་བརླབས་ཀྱིས་རྟོགས་པ་ནི། དཔེར་ན་རས་དཀར་པོ་ལ་དམར་པོར་བྱིན་གྱིས་རློབ་པའི་མཐུ་བཙོད་ལ་ཡོད་པ་བཞིན་དུ། གཉུག་མའི་ཁམས་དང་སྐྱེ་མཆེད་ཀྱི་ཆོས་ཐམས་ཅད་གདོད་ནས་རང་བཞིན་དག་པའི་སངས་རྒྱས་ཡིན་པ་ལ་རྣལ་འབྱོར་པས་ཐབས་ཁྱད་པར་ཅན་གྱིས་དག་པ་མཐར་ཕྱིན་པའི་སངས་རྒྱས་གསང་ཆེན་རིགས་གཅིག་ཏུ་ངེས་པར་བྱིན་གྱིས་རློབ་པའི་མཐུ་ཡང་བདེ་བ་ཆེན་པོའི་རྫོགས་རིམ་ལ་ཡོད་དེ། ཡོད་ཚུལ་ནི་རྒྱུ་གཅིག་པ་དང་ཡིག་འབྲུའི་ཚུལ་གྱིས་སྔར་བསྟན་པ་བཞིན་རིག་པ་བྱང་ཆུབ་ཀྱི་སེམས་ཉིད་ཀྱིས་རང་གི་སྣང་ཆ་ཤིན་ཏུ་ཕྲ་བའི་རླུང་དེ་ཉིད་རྡོ་རྗེ་འཛིན་པའི་གཟུགས་སྐུར་སྦྱོང་ཞིང་བྱིན་གྱིས་རློབ་པ་སྟེ། ཟབ་པའི་བདེ་ཆེན་མ་རྟོགས་པ་དེ་སྲིད་དུ་རྒྱ་ཆེ་བའི་སྐུ་དེའང་རྟོགས་མི་སྲིད་ལ༑ ཟབ་པའི་ཡེ་ཤེས་ལ་ནུས་པ་ཅི་ཙམ་ཐོབ་པ་དེ་ཙམ་དུ་ཟབ་གསལ་གཉིས་མེད་ཀྱི་སྐུའི་ཁྱད་པར་ཡང་གོང་ནས་གོང་དུ་འཇུག་པའི་གནད་མཐའ་ཆོད་པ་ཞིག་གི་སྟེང་ནས་ཡུན་རིང་པོར་ཚོལ་དགོས་པ་ཡིན་ནོ། །དེ་ནི་བྱིན་གྱིས་བརླབས་ཀྱིས་རྟོགས་པའོ། །མངོན་སུམ་པར་རྟོགས་པ་ནི། ཆོས་ཐམས་ཅད་ཡེ་ནས་སངས་རྒྱས་པར་གནས་པ་དེའང༌། གསང་སྔགས་བླ་མེད་ཀྱི་རྒྱུད་སྡེའི་ལུང་དང་མི་འགལ་ཞིང༌། བླ་མ་རྩ་བཤད་ཀྱི་རྒྱུད་དོན་རྙེད་པའི་མན་ངག་དང་འགལ་བའང་མ་ཡིན་ལ། ཐོས་བསམ་གྱི་སྒོ་ནས་ལུང་དང་མན་ངག་གི་ཚིག་ལ་ཐོས་རྒྱ་མང་བ་དང༌། དོན་ལ་དཔྱད་སྟོབས་རྫོགས་པ་ཙམ་ཞིག་ལ་གདེང་འཆའ་ཞིང་བརྟེན༵་བཅས་པ༵་མ༵་ཡིན༵་པར༵་། བསྒོམ་བྱུང་གི་རྩལ་ཡོངས་སུ་རྫོགས་པའི་སྒོ་ནས་རང་གི་རིག་པའི་བློའི་གཏིང་དུ་ཡིད་ཆེས་ཤིང་བརྟན་པ་ཐོབ་པས་སྒྱུ་མ་ལྟ་བུའི་ཡེ་ཤེས་ཀྱི་སྐུ་དྭངས་བའི་མཁའ་ལ་འཇའ་ཚོན་ལྟར་ཤར་བའི་ཏིང་ངེ་འཛིན་མངོན་སུམ་དུ་རྟོགས་པའོ༑ ༑མངོན་སུམ་དུ་རྟོགས་ཚུལ་འདི་ལ་ནི་དོན་གྱི་འོད་གསལ་མཐའ་གཅིག་ཏུ་མི་དགོས་སོ། །དེ་ལྟར་ན་རྒྱུ་གཅིག་པ་དང་ཡིག་འབྲུའི་ཚུལ་གཉིས་ཀྱིས་ཟུང་འཇུག་གི་གོ་འཕང་དངོས་བསྟན། བྱིན་གྱིས་བརླབ་པ་དང༌། མངོན་སུམ་པའི་ཚུལ་གཉིས་ཀྱིས་གོ་འཕང་དེ་འགྲུབ་ཚུལ་བསྟན་པ་ཡིན་པས། འོད་གསལ་རྫོགས་པ་ཆེན་པོའི་ལུགས་ལ་རིག་སྟོང་གཞོན་ནུ་བུམ་སྐུའི་གོ་འཕང་སྒྲུབ་ཚུལ་སྣང་བ་རྣམ་པ་བཞིའི་ལམ་ནས་བསྟན་པའི་ཤིང་རྟའི་སྲོལ་ཆེན་པོ་དེ་མ་གཏོགས་པ། ཧེ་རུ་ཀ་གལྤོ་ལ་སོགས་པའི་ཕ་རྒྱུད་དང་མ་རྒྱུད་ཀྱི་ལམ་མཐའ་དག་གི་ཆིངས་ཡང་རྫོགས་རིམ་པའི་ཟུང་འཇུག་གི་གོ་བ་འདིས་བྱེད་པས་ལམ་དུམ་བུ་བ་ཞིག་ཏུ་གཟུང་བར་བྱས་ན་མི་རིགས་པ་ཤིན་ཏུ་ཆེན་པོ་སྟེ། གཏན་ལ་ཕབ་སྟེ་སྤྲོས་ན་མཐའ་ཡས་སོ། །ཚུལ་འདི་ལྟར་ཡང་དག་པའི་ལམ་གྱིས་ཡིད་ཆེས་པ་ནི་སྔགས་བླ་མེད་སྤྱིའི་ལམ་ལ་ཡིད་ཆེས་པ་ཡིན་ཏེ། རྟོགས་པ་རྣམ་པ་བཞིའི་དོན་རྡོ་རྗེ་སྐུའི་རྣལ་འབྱོར་དང༌། བདེ་ཆེན་ཟབ་མོའི་རྣལ་འབྱོར་གཉིས་ཇི་ལྟ་བ་བཞིན་དུ་རིག༵་པ་ཉིད་ནི་རྣལ་འབྱོར་པ་བླ་མེད་ཀྱི་སྔགས་ལ་ཞུགས་པའི་ལམ་གྱི་གཙོ་བོ་སྟེ། གལྤོ་ལས། བདེན་པའི་དགོངས་པ་རྣམ་པ་གཉིས། །ཀུན་རྫོབ་དོན་དམ་བཤད་པ་གྱིས། །ཞེས་རྒྱུད་དོན་བདེན་པ་གཉིས་ཀྱི་རྫོགས་རིམ་དུ་བསྡུས་པ་དེ་ལ་གོ་བའི་རླབས་ཆེན་པོ་ཡོད་པའི་ཕྱིར་ར༵་། །གལྤོའི་ལུང་དོན་དེ་ནི། བདེན་པ་གཉིས་ཀྱི་རྫོགས་རིམ་འབའ་ཞིག་ལ་འདོད་མི་དགོས་ཀྱི། རྫོགས་རིམ་ཁོ་ན་ནི་ལམ་ཉི་ཚེ་བ་ཡིན་པས་ལམ་དེ་ལ་སྣོན་པ་གཞན་ཞིག་དགོས་སོ་སྙམ་ན་ནི་རྡོ་རྗེ་འཆང་གི་རྒྱུད་ལུང་དཔང་པོར་བཞག་སྟེ་རྒྱུ་མཚན་གྱི་རིགས་པ་འདོད་དགུར་བསྙད་པས་ཆོག་པ་བློ་གྲོས་ཆེན་པོ་རྣམས་ཀྱིས་མཁྱེན་པ་ཉིད་ནའང་མ་བརྗོད་དོ། །དེའང་རྩ་རྒྱུད་ལས། ངོ་མཚར་ཆོ་འཕྲུལ་རྨད་ཀྱི་ཆོས། །གཞན་ནས་ཡོད་འོང་མ་ཡིན་ཏེ། །ཐབས་ལ་བརྟན་པའི་ཤེས་རབ་ཉིད། །དེ་ལྟ་བུ་ཡི་ངང་དུ་འབྱུང༌། །ཞེས་རྒྱུ་ལམ་བསྒྲུབ་པའི་འབྲས་བུ་ཕྱི་རོལ་ནས་འབྱུང་བའི་དུས་ལ་ལྟོས་དགོས་པ་ལྟ་བུ་མ་ཡིན་གྱི། རང་གིས་མངོན་དུ་རྟོགས་ཤིང་ཡིད་ཆེས་ཏེ་བསྒོམ་པའི་རྣལ་འབྱོར་གྱི་མཐུས་རང་གི་ཤིན་ཏུ་ཕྲ་བའི་སྒོ་གསུམ་ཉིད་སངས་རྒྱས་སུ་འགྱུར་བར་གསུངས་པ་ལྟར་ཤེས་པར་བྱའོ༵་། །གཉིས་པ་ནི། སླར་ཡང་མཚན་ཉིད་རྣམ་པ་གསུམ་གྱི་སྒོ་ནས་བསྟན་པར་བཞེད་དེ། རྟོགས་པ་བཞིའི་དོན་དེ་གང་གིས་འགྲུབ་ན། འབྲས་བུ་རྣམ་པར་དག་པ་ཚུལ་བཞིའི་སྒོ་ནས་འཇུག་པར་བྱ་བ་འདི་ཡོངས་སུ་སྒྲུབ་པ་ལ་ནི་མཚན་ཉིད་གསུམ་གྱིས་སྔགས་པའི་འདོད་དོན་མཐར་ཕྱིན་པར་འགྱུར་ཏེ། བཤད་མ་ཐག་པའི་རྟོགས་པ་རྣམ་པ་བཞིའི་ཚུལ་དེ་ཤེས་རབ་ཀྱིས་མ་ནོར་བར་རིག་པ་ནི་རྒྱུ་ཤེས་པའི་མཚན་ཉིད་ལྟ་བ་ཞེས་བཤད་དོ། །དེའི་རྗེས་སུ་སྟོང་པ་ཆེན་མོ་ལ་སོགས་པའི་རྣལ་འབྱོར་ལྔས་ཡང་ནས་ཡང་དུ་གོམས་པར་བྱེད་པ་ནི་རྐྱེན་འཇུག་པའི་མཚན་ཉིད་ཏིང་ངེ་འཛིན་ཏོ། །རྣལ་འབྱོར་ལྔ་པོ་རེ་རེ་ཞིང་གཡོ་ཐོབ་གོམས་བརྟན་མཐར་ཕྱིན་གྱི་ཉམས་ལྔ་རྫོགས་སུ་སློབ་ཚུལ་ནི་སྒྱུ་འཕྲུལ་ལམ་རིམ་དུ་བལྟ་ལ། ཁ་ཅིག་ཚོགས་ལམ་གྱི་རྣལ་འབྱོར་བཞི་དང༌། སྦྱོར་ལམ་གྱི་རྣལ་འབྱོར་གཅིག་ཅེས་འཆད་བཞིན་ཏུ། རྫོགས་རིམ་དང་སྦྱར་བའི་རྣལ་འབྱོར་ལྔ་ཞལ་གྱིས་འཆེས་པ་དེ་དག་ལ་ནི་སྔགས་བླ་མེད་ཀྱི་སྦྱོར་ལམ་ལ་ཞུགས་པའི་ཚད་དེ་གང་ཡིན་ཞེས་འདྲི་བར་བྱ་སྟེ། རྫོགས་རིམ་དུ་མ་ཞུགས་པའི་བླ་མེད་ཀྱི་སྦྱོར་ལམ་ལ་ཞུགས་པ་མེད་པ་ནི་རྒྱུད་སྡེ་འདི་སྐོར་དང༌། འོད་གསལ་རྫོགས་ཆེན་གང་གི་ལུགས་ལའང་འདྲ་བའི་ཕྱིར་ཏེ། དེ་ལྟར་ལུང་དང་རིགས་པས་འགྲུབ་པའི་ཕྱིར་རོ་ཞེས་རྟགས་གཉིས་པ་སྤྲོས་པས་ཆོག་པའི་ཕྱིར། དེས་ན་རྣལ་འབྱོར་ལྔ་པོ་གང་རུང་རེ་ལའང་རིམ་པ་གཉིས་ཀྱི་ཆར་གྱུར་ན་གཉིས་རེ་ཡོད་ལ། ཚོགས་སྒྲུབ་སྐོར་དུ་ཞིབ་ཆ་དགོས་པ་མང་བ་སོགས་སྣང་ཡང་འདིར་ཡི་གེ་ཉུང་ངུར་འདོད་པས་བཞག་གོ །ཏིང་ངེ་འཛིན་གོམས་པའི་མཐུས་སློབ་པ་མིན་པའི་ཟུང་འཇུག་མངོན་དུ་གྱུར་པ་ནི་འབྲས་བུའི་མཚན་ཉིད་དོ། །མཚན་ཉིད་དེ་གསུམ་གྱིས་གོ་རིམ་བཞིན་དུ་འབྲེལ་བ་དང༌། དགོས་པ་དང༌། དགོས་པའི་ཡང་དགོས་པ་སྟོན་ཏེ། དེ་ལ་འབྲེལ་བ་ནི། ཀུན་ནས་ཉོན་མོངས་པ་དང་རྣམ་པར་བྱང་བའི་ཆོས་སུ་བརྟགས་པ་ཐམས་ཅད་ཡེ་གདོད་མ་ནས་གཉུག་མའི་སྐུ་གསུང་ཐུགས་གསུམ་གྱི་བདག་ཉིད་དང་རང་རྩལ་ཏུ་མ་གྱུར་བ་མེད་པས་སྔར་རྒྱུ་གཅིག་པ་དང་ཡིག་འབྲུའི་ཚུལ་གྱི་སྐབས་སུ་བྱུང་བ་ལྟར་སྲིད་ཞི་མཉམ་པ་ཉིད་ཀྱི་རང་བཞིན་གྱིས་སངས་རྒྱས་པའི་གཞིའི་དབྱིངས་འོད་གསལ་དོན་དམ་བདེན་པ་དང༌། རང་མདངས་ཡེ་ཤེས་ཀྱི་གསལ་ཆ་འཇའ་ལུས་གསང་ཆེན་རིགས་གཅིག་ཏུ་བྱིན་གྱིས་རླབས་པའི་གཞིའི་དོན་དུ་རྟོགས་པ་ནི་རྒྱུ་ཤེས་པའི་མཚན་ཉིད་དོ་ཞེས་གོང་དུ་ཡང་བཤད་ཟིན་པའི་ཕྱིར་རོ། །འདི་དགོས་འབྲེལ་གྱི་ནང་ནས་འབྲེལ་བ་དང་མཚན་ཉིད་གསུམ་གྱི་ནང་ཚན་ལྟ་བ་གཉིས་སྦྱར་ཏེ་གསུངས་པའི་རྒྱུ་མཚན་དུ་འགྲོའམ་མི་འགྲོ་མི་འགྲོ་ན་ཡང་ཟློས་ཙམ་དུ་ཟད་ལ། འགྲོ་ན་རྒྱུ་མཚན་དེ་བཤད་དགོས་སོ་ཞེ་ན། འབྲེལ་བ་དངོས་ནི། ལྟ་བ་དང་ཏིང་ངེ་འཛིན་གཉིས་གཙོ་བོར་བདག་གཅིག་འབྲེལ་དང༌། དེ་དང་འབྲས་བུ་གཉིས་ཕལ་ཆེར་དེ་བྱུང་འབྲེལ་དུ་འོང་མོད་ཀྱི། ལྟ་བ་དང་ཏིང་ངེ་འཛིན་དེ་བྱུང་འབྲེལ་ཡིན་པའང་ཡོད་དོ། །དེ༵་ལྟ་ནའང་ལྟ་བ་ལ་འབྲེལ་བར་གསུངས་པ་ནི༵་དཔེར་ན་ཕར་ཕྱིན་ཐེག་པར་སེམས་བསྐྱེད་ཐེག་ཆེན་ལམ་གྱི་སྡོམ་པར་བཞག་པ་བཞིན་ཏེ། བསྐྱེད་རྫོགས་གང་ཡིན་ཀྱང་འབྲས་བུ་སྐུ་གཉིས་ཀྱི་དངོས་རྒྱུར་འགྲོ་བ་ལ་ལྟ་བས་ལམ་དང་འབྲས་བུའི་ཚུལ་གྱི་སྤྱིར་ཤར་བ་ཞིག་དགོས་པས་ལྟ་བ་ནི་ལམ་གོང་འོག་གི་འབྲེལ་སྦྱོར་བྱེད་ཡིན་ཏེ། ཤེས་རབ་ཀྱིས་དཀྱུ་ས་མ་མཐོང་ན་ཏིང་འཛིན་གྱི་རྟ་རྒྱུག་པ་མི་འོང་བའི་དོན་ནོ། །ལྷའི་རྣམ་པ་ལམ་དུ་གང་བྱས་ཐམས་ཅད་ཀྱང་སངས་རྒྱས་ང་རྒྱལ་ཆེན་པོའི་ཏིང་ངེ་འཛིན་དུ་འགྲོ་བ་ནི་ལྟ་བས་བྱང་ཆུབ་ཆེན་པོའི་ཡེ་ཤེས་ཀྱི་ཚུལ་ལ་ངེས་པ་རྙེད་དེ་བསྒོམ་པའི་ལག་རྗེས་ཡིན་གྱི། ཞི་བ་དགོད་ཁ་ཡལ་ལེ་བ་དང༌། ཁྲོ་བོ་སྡང་མིག་ཅེ་རེ་བ་ཙམ་ཞིག་བསྒོམ་པས་བླ༵་ན༵་མེད༵་པའི༵་སངས་རྒྱས་ཀྱི་གཟུགས་སྐུ་མི་འགྲུབ་པ་བཞིན་དུ་བསྐྱེད་རྫོགས་ཀྱི་ཁོག་གཞན་ལ་ཡང་རིགས་པ་སྦྱར་བར་བྱ་སྟེ་སླའོ། །དེའི་ཕྱིར་ལྟ་བ་ལ་སང༵ས་རྒྱས༵་སུ༵་འགྲུབ༵་པའི༵་དོན༵་དུ༵་འབྲེ༵ལ་བ༵་ཞེས་བྱའོ༵། །དེའང་དང་པོ་ལྟ་བ་རྟོགས་པ་ཁོ་ནས་མི་ཆོག་གི །དེ་རྟོགས་པའི་དགོས་པ་ཅི་འདྲ་བ་ཞིག་བྱེད་པ་ནི། ཀུན་ནས་ཉོན་མོངས་པ་དང་རྣམ་པར་བྱང་བའི་ཆོས་དང༌། རྣལ་འབྱོར་པ་ལ་ཡེ་ཤེས་གསལ་བ་དང་དངོས་གྲུབ་འགུགས་པའི་སྒོ་ནས་སྒྲོན་མ་དང་ལྕགས་ཀྱུ་ཞེས་པའི་སྒྲས་རྒྱུད་ནས་བསྔགས་པ་སྨན་ཤ་ལྔ་དང་བདུད་རྩི་ཆེན་པོ་ལྔ་ལ་སོགས་པར་བཏགས་པ་ཐམས་ཅད་ཡེ་ནས་གཞིའི་ཆོས་སྐུ་རིག་སྟོང་མཉམ་རྫོགས་ཆེན་པོའི་རོལ་རྩལ། དུ་མ་རོ་གཅིག་གི་ངང་ལ་སངས་རྒྱས་པའི་མཉམ་པ་ཆེན་པོ་མི་མཉམ་པ་མེད་པའི་དོན་ལ་རིམ་གཉིས་ཀྱི་རྟོགས་པ་ཀློང་དུ་གྱུར་པའི་སྒོ་ནས་བླང་དོར་མེད་པར་སྤྱོད་པ་སྟེ། དེའང་ལས་དང་པོ་པ་དང་གྲུབ་པ་དང་ཐམས་ཅད་མཁྱེན་པ་ལ་སོགས་པའི་རིམ་པའམ། ཡེ་ཤེས་ཅུང་ཟད་ཕེབས་པ་དང༌། ཡེ་ཤེས་ལ་ཅུང་ཟད་དབང་ཐོབ་པ་ལ་སོགས་པའི་ངེས་པས་རང་གི་རྒྱུད་ཚོད་དང་བསྟུན་ཏེ་སྤྱོད་པར་གསུངས་སོ། །འདིའི་སྤྱོད་པ་ཞེས་གསུངས་པ་ནི་ཉེ་རྒྱུའི་སྤྱོད་པ་ཁོ་ན་ལ་མི་བྱ་ཞིང༌། བདུད་རྩི་ལྔ་ལ་ལོངས་སྤྱོད་པ་རྐྱང་པ་ཡང་མིན་ཏེ། གཞུང་དེའི་དེ་མ་ཐག་པར། འཇུག་པའི་མཚན་ཉིད་དོ་ཞེས་ཏིང་ངེ་འཛིན་གྱི་མཚན་ཉིད་དུ་གསུངས་པའི་ཕྱིར་རོ། །དེའི་ཕྱིར་ཡེ་ནས་སངས་རྒྱས་པའི་མཉམ་པ་ཆེན་པོ་ཞེས་གསུངས་པའི་ནུས་པ་དེ་རྩལ་དུ་འདོན་པ་གལ་ཆེ་བས་དེ་འདྲའི་མཉམ་པ་ཉིད་ཀྱི་དོན་དེ་སྤྱོད་ཡུལ་དུ་བྱེད་པའི་དོན་ཡིན་ཞིང༌། དེའི་ཡན་ལག་རྣམས་ཀྱང་དེར་བསྡུའོ། །ཏིང་ངེ་འཛིན་དེ་ནི་བླ་ན་མེད་པའི་སངས་རྒྱས་སུ་འགྲུབ་པའི་རྒྱུ་ཡིན་པའི་ཕྱིར་ལྟ་བ་རྟོགས་པའི་དགོས་པ་སྟེ། སངས་རྒྱས་སྒྲུབ་པའི་ལམ་གྱི་རྩེ་མོ་གསང་སྔགས་བླ་མེད་ཀྱི་རྣལ་འབྱོར་ཐུན་མོང་མ་ཡིན་པ་ཐམས་ཅད་རིམ་པ་གཉིས་ཀྱི་ཕྱོགས་སུ་འདུ་བར་རྒྱུད་ལས། སངས་རྒྱས་རྣམས་ཀྱིས་ཆོས་བསྟན་པ། །རིམ་པ་གཉིས་ལ་ཡང་དག་བརྟེན། །བསྐྱེད་པ་ཡི་ནི་རིམ་པ་དང༌། །རྫོགས་པ་ཡི་ནི་རིམ་ཉིད་དོ། །ཞེས་གསུངས་པའི་ཕྱིར་རོ། །གལ་ཏེ་དེ་ལྟ་མིན་པར་ཚིག་འདིས་བསྐྱེད་རྫོགས་ཕྱོགས་རེ་བ་ཙམ་ཞིག་བསྟན་པ་ཡིན་ན་བླ་མེད་ལུགས་ཀྱི་ཐུན་མོང་མ་ཡིན་པའི་རྣལ་འབྱོར་གྱི་ཆིངས་རྫོགས་པར་མི་འགྱུར་རོ། །ཀླུའི་ཞབས་ཀྱིས། ཐབས་འདི་རྫོགས་པའི་སངས་རྒྱས་ཀྱིས། །སྐས་ཀྱི་རིམ་པ་ལྟ་བུར་གསུངས། །ཞེས་རིམ་གཉིས་གོ་རིམ་བཞིན་དུ་བགྲོད་དགོས་པར་གསུངས་པ་བཞིན་ནོ། །དགོས་པའི་ཡང་དགོས་པ་ནི། ཀུན་ནས་ཉོན་མོངས་པ་དང༌། རྣམ་པར་བྱང་བའི་ཆོས་སྤྱི་དང༌། སྨན་ལྔ་དང་བདུད་རྩི་ལྔ་ལ་སོགས་པ་རྣལ་འབྱོར་པའི་ལོངས་སྤྱོད་ཀྱི་ཁྱད་པར་དུ་བཏགས་པ་ཐམས་ཅད་ཡེ་ནས་སངས་རྒྱས་པའི་མཉམ་པ་ཆེན་པོའི་ངང་དུ་བླང་དོར་བསལ་བཞག་མེད་པར་འཇོག་པའི་ལྟ་བ་དང་ཏིང་ངེ་འཛིན་རྣམས་གོང་ནས་གོང་དུ་ནུས་པ་བརྟན་ཞིང་གསལ་སྣང་འཕེལ་བའི་ལག་རྗེས་སུ། རྩ་རྒྱུད་སྙིང་པོར། ཡེ་ཤེས་ཐིག་ལེ་དེ་ཉིད་ལ། །ཡེ་ཤེས་ཐིག་ལེ་ཉིད་སྣང་བ། །བསམ་གྱིས་མི་ཁྱབ་མཐའ་ཡས་མཆོག །ཅེས་སོགས་གསུངས་པ་བཞིན་ཏུ་ནམ་ཞིག་ན་འབད་རྩོལ་མེད་པར་ཡོངས་སུ་དག་པའི་ཡེ་ཤེས་ཀྱི་ངང་གཅིག་ཏུ་ལྷུན་གྱིས་གྲུབ་པའི་ཕྱིར། མ་དག་པའི་གནས་སྐབས་ནས་ཀྱང་སྲིད་པའི་འཁོར་བ་ཉིད་ཡེ་ནས་བླ་ན་མེད་པར་སངས་རྒྱས་པའི་བསྒྲུབ་གཞིའི་རང་བཞིན་དེ། འདིར་ལམ་བསྒོམ་སྟབས་ཀྱིས་དག་པ་གཉིས་ལྡན་མྱ་ངན་ལས་འདས་པའི་མཚན་ཉིད་དམ་བདག་ཉིད་ཅན་གྱི་སྐུ་དང་ཡེ་ཤེས་ཕྲིན་ལས་དང་བཅས་པའི་ངོ་བོ་ཉིད་དུ༵་ལྷུན༵་གྱིས༵་གྲུབ༵་པ༵་སྟེ། ལྷུན་གྱིས་གྲུབ་པར་བསྟན་པའི་ཚིག་སྔ་མ་ལམ་མཐར་ཕྱིན་པའི་སྒོ་ནས་བསྔགས་པ་དང༌། འདིས་གཞི་དུས་ཀྱི་སྐྱེ་འཆི་བར་དོ་གསུམ་བླ་ན་མེད་པའི་འབྲས་བུ་ཆོས་སྐུར་བསྒྱུར་བའི་སྒོ་ནས་བསྔགས་པ་སྟེ། ལྡོག་ཆའི་དབང་གིས་ཡིན་ནོ། །དེ་ནི་འབྲས་བུའི་མཚན་ཉིད་དོ། །མདོར་ན་སྐུ་གསུང་ཐུགས་མི་ཟད་པ་རྒྱན་གྱི་འཁོར་ལོ་གང་ལ་མངོན་སུམ་པར་གྱུར་པ་ནི་དགོས་པའི་ཡང་དགོས་པའོ། །མི་ཟད་པ་ཞེས་པ་རྒྱུན་གྱིས་རྟག་པའི་དོན་དང༌། རྒྱན་ཞེས་པ་ཆོས་ཀྱི་དབྱིངས་རྒྱས་པར་འགེང་བའི་རྒྱན་ཏེ་རྒྱ་མ་ཆད་པའི་དོན་དང༌། འཁོར་ལོ་ཞེས་པ་གདུལ་བྱའི་རྒྱུད་ཀྱི་སྒྲིབ་པའི་དྲྭ་བ་མཐའ་དག་གཅོད་པའི་དོན་ནོ་ཞེས་བླ་མ་རྣམས་གསུང་ངོ༌། །གསུམ་པ་ནི། དེ་ལ་བསྙེན་པ་དང༌། ཉེ་བའི་བསྙེན་པ་དང༌། སྒྲུབ་པ་དང༌། སྒྲུབ་པ་ཆེན་པོའི་དོན་ཏེ་བསྙེན་སྒྲུབ་ཡན་ལག་བཞི་ཡོངས་སུ་རྫོགས་པར་ལྷུན་གྱིས་གྲུབ་པར་གྱུར་བའི་རྣལ༵་འབྱོར༵་ལ༵་བརྩོན༵་པར༵་བྱའོ༵། །སྤྱིར་བསྐྱེད་རྫོགས་ཀྱི་ལམ་ཐམས་ཅད་བསྙེན་སྒྲུབ་ཡན་ལག་བཞིའི་ཆིངས་སུ་བཅིང་བ་ནི་ཟབ་མོ་གཏེར་གྱི་ཕྱོགས་ལ་ཡོངས་གྲགས་ཡིན་པར་མ་ཟད། སྔ་འགྱུར་གྱི་རྒྱུད་སྡེ་ཆེན་མོ་རྣམས་དང༌། ན་རོ་ཏ་པའི་འདུས་པ་རྒྱུད་ཕྱི་མའི་འགྲེལ་བ་དང༌། བྱང་ཆུབ་སེམས་དཔའི་འགྲེལ་བ་སྐོར་གསུམ་ལ་སོགས་པར་ཡོངས་སུ་གྲགས་པའི་ཤིང་རྟ་ཆེན་པོའི་ལུགས་སྲོལ་ཡིན་ཏེ། བསྙད་པས་ཆོག་གོ །དེ་ལ་བསྙེན་པ་ནི། བྱང་ཆུབ་ཀྱི་སེམས་ཤེས་པ་སྟེ། དང་པོ་སྟོང་ཉིད་བྱང་ཆུབ་སྟེ། །ཞེས་པ་ལྟར་བསྒོམ་མོ། །དེའང་ཆོས་ཐམས་ཅད་ཡེ་ནས་རང་བཞིན་རྣམ་དག་གི་སངས་རྒྱས་སམ་ཆོས་ཉིད་ཤིན་ཏུ་རྣམ་པར་དག་པའི་དབྱིངས་མཐའ་བཞིའི་མཚན་མ་དང་བྲལ་བའི་གཤིས་སུ་གཉིས་སྣང་གི་སྤྲོས་པ་ཐམས་ཅད་ཉེ་བར་ཞི་བའི་རང་བཞིན་གྱིས་མཉམ་པར་བཞག་པ་ནི་བྱང་ཆུབ་ཀྱི་སེམས་ཞེས་བྱའོ། །དེ་ལྟ་བུའི་ངང་དུ་འཇོག་ཚུལ་ཡང་ཤེས་བྱའི་ཡིན་ལུགས་དེ་ཉིད་ལམ་གྱིས་གསར་དུ་བསྒྲུབ་ཅིང་གཉེན་པོས་བཅོས་སུ་མེད་དེ། རྗེ་བཙུན་འཕགས་པས། འདི་ལ་བསལ་བྱ་ཅི་ཡང་མེད། །བཞག་པར་བྱ་བ་ཅུང་ཟད་མེད། །ཡང་དག་ཉིད་ལ་ཡང་དག་ལྟ། །ཞེས་དང༌། རྩ་བའི་རྒྱུད་ལས། དངོས་རྣམས་ཉིད་ནི་དངོས་མེད་པས། །རྟོགས་པའི་དབང་བསྒྱུར་ཉིད་འཛིན་ཡིན། །ཞེས་པ་ལྟར་ཡེ་ཇི་བཞིན་པའི་ངང་ལ་བསལ་བཞག་སྤྲོ་བསྡུ་མེད་པར་ལྷོད་དེ་བཞག་པར་བྱའོ། །འདི་ཡང་སློབ་དཔོན་ཆེན་པོ་འཇམ་དཔལ་བཤེས་གཉེན་ཕྱི་མས་སྒྲུབ་ཐབས་ཀུན་བཟང་དུ་ཤུ་ནྱ་ཏའི་སྔགས་དོན་རྣམ་ཐར་སྒོ་གསུམ་ལ་སྦྱར་ཏེ་གསུངས་པ་དང་གནད་གཅིག་གོ །ཉེ་བའི་བསྙེན་པ་ནི། སྟོང་ཉིད་དེའི་ངང་ནས་ཆོ་ག་སུམ་བསྐྱེད་ལྟ་བུའི་རིམ་པས་བདག་ཉིད་ལྷར་ཤེས་ཤིང་བསྒོམ་པ་སྟེ། དེའང་ཆོས་ཐམས་ཅད་ཡེ་ནས་སངས་རྒྱས་པའི་རང་བཞིན་ཞེས་གསུངས་པ་ནི་སྔར་བཤད་པ་ལྟར་ཕུང་པོ་ལྔ་དེ་བཞིན་གཤེགས་པའི་དཀྱིལ་འཁོར་གྱི་བསྒྲུབ་གཞིར་གནས་པའི་དོན་ཏུ་བྱའོ། །དེ་ལྟར་གནས་པས་ན་བདག་ཉིད་ཀྱང་ཡེ་ནས་དག་པ་ལྷའི་རྟེན་གྱི་རང་བཞིན་ཡིན་གྱི། ད་ལྟ་ལམ་གྱིས་གསར་དུ་སྒྲུབ་པ་ནི་མ་ཡིན་པར་རྟོགས་པ་སྟེ་ཀྱང་གི་སྒྲས་ནི། གདགས་གཞི་ཕུང་པོ་རྣམས་གཞིའི་དཀྱིལ་འཁོར་དུ་གནས་པར་མ་ཟད། བཏགས་པའི་ཆོས་གང་ཟག་བདག་ཉིད་ཀྱང་ཞེས་སྦྱར་བར་བྱའོ༵་། །སྒྲུབ་པ་ནི་ཡུམ་བསྐྱེད་པ་སྟེ། དེའང་ཡུམ་ཆེན་མོ་ཞེས་པ་གསང་བ་སྙིང་པོར་སྣང་བའི་དབྱིངས་དང་ཞེས་སོགས་གསུང་བའི་ཐད་ཀྱི་འགྲེལ་བ་དགོངས་རྒྱན་དུ། རྟོགས་པའི་དུས་ན་ནམ་མཁའ་ལ་སོགས་པའི་འབྱུང་བ་ལྔ་དང་དག་པའི་དུས་སུ་དབྱིངས་ཕྱུག་མ་ལ་སོགས་པའི་རིགས་ཀྱི་བཙུན་མོ་ལྔ་ཡིན་པས་དེ་སྐད་ཅེས་བརྗོད་པར་གསུངས་སོ། །རྟོགས་པའི་དུས་སུ་ནམ་མཁའ་ལ༵་ས་འབྱུང་བ་བཞི་དང༌། དག་པའི་ཚེ་སྣང་བ་ཆེན་པོ་ཆོས་ཉིད་ཀྱི་དབྱིངས་ནས་ཡུམ་ཆེན་མོ་ཡེ་ཤེས་ཀྱི་ས་ཆུ་མེ་རླུང་བཞིར་སྣང་ཞིང༌། དག་པ་དང་མ་དག་པའི་དུས་དེ་དང་དེའི་བརྟན་གཡོའི་བྱ་བ་རབ་འབྱམས་སུ་བྱེད་པའི་སྐྱེད་ཡུམ་ཡེ་ནས་ཡིན་པའི་ཕྱིར། དེ་ཉིད་བཙུན་མོ་རྣམ་པ་ལྔར་ང་རྒྱལ་བྱའོ། །སྒྲུབ་པ་ཆེན་པོ་ནི། ཐབས་ཡབ་དང་ཤེས་རབ་ལྷ་མོ་རྣམས་སྙོམ་འཇུག་གི་བདེ་བས་རོལ་བ་སྟེ། དེའང་རོལ་བ་ཙམ་ཞིག་མིན་གྱི་སྲས་འབྱིན་བྱེད་པའི་རིམ་པའོ། །ཇི་ལྟར་བྱེད་ན་ཡུམ་ཆེན་མོ་ལྔ་ཤེས་རབ་ཀྱི་བརྡ་དང་ཡུམ་གྱི་མཁའ་སྟོང་པ་ཉིད་མཚོན་བྱེད་ཀྱི་དབྱིངས་ལས་ཕུང་པོ་ལྔའི་དག་པ་སངས་རྒྱས་རིགས་ལྔར་སྦྱར་བའི་ཐབས་ཀྱི་ཆ་ཡབ་སྨོན་པ་མེད་པར་ཡེ་ནས་ཟུང་དུ་འབྲེལ་བ་ལས་བྱང་ཆུབ་སེམས་ཀྱི་སྤྲིན་སྤྲུལ་པ་ལྕམ་དྲལ་ཏེ་སེམས་དཔའ་དང་སེམས་དཔའ་དང་འཇོམ་བྱེད་ཡབ་ཡུམ་ལ་སོགས་པར་གྱུར་པའི་རང་བཞིན་ནི་ཞེས་པ་དེ་རྣམས་ཀྱི་དོན་དམ་བདེན་པ་ལ་མི་བྱའི་མེའི་རང་བཞིན་ཚ་བ་ཞེས་པ་ལྟར་ལྷ་ཚོགས་དེ་ཐམས་ཅད་བདེ་སྟོང་གི་རྣམ་རོལ་ཚུལ་གཅིག་ཏུ་འོག་ནས་འཆད་པར་སྦྲེལ་ཏོ། །སྔོན་གྱི་བླ་མ་རྣམས་ལ་ཡོངས་སུ་གྲགས་པའི་སྒྱུ་འཕྲུལ་དཀྱིལ་ཆོག་རྣམས་སུ་སྲས་འབྱིན་དེ་ལྟར་བྱ་བ་ཞིག་ཞི་བའི་སྐབས་ནས་མི་འབྱུང་ཡང༌། ཁྲོ་བོའི་སྒྲུབ་ཐབས་ལས་བྱུང་བས་དེ་གཉིས་རིགས་པ་མཚུངས་པར་མ་ཟད་གཞུང་འདི་ལ་གསུངས་པ་འདིས་ཀྱང་ཤེས་སོ། །ལྷ་ཐམས་ཅད་ཀྱང་རང་གི་སེམས་ཉིད་ཡེ༵་ནས༵་རང་བཞིན་རྣམ་དག་ཏུ་སང༵ས་རྒྱས༵་པའི༵་དོན༵་ལ༵་སླར་ཡང་འབྲས་བུ་ལམ་བྱེད་ཀྱི་ངང་ལ་ཁྱད་པར་ཅན་བསྐྱེད་དེ། སྒྱུ༵་མ༵་ལྟ་བུའི་ཡེ་ཤེས་ཀྱི་ལུས་ཅན་དབྱིངས་ཀྱི་བཙུན་མོ་ལྔ་ལ༵་སྒྱུ༵་མ༵་ལྟ་བུའི་ཕུང་པོ་རྣམ་པར་དག་པ་རིགས་ལྔའི་མགོན་པོ་རྣམས་སྙོམ་འཇུག་གི་དགྱེས་པས་རོལ༵་ཅིང༵་། དེ་ལ་བྱུང་བའི་བདེ༵་མཆོ༵ག་ཡེ་ཤེས་སྒྱུ༵་མའི༵་རྒྱུ༵་ལ༵་འདོད་དགུར་སྤྱོད་པའི་དུས་ཉིད༵་ན༵་མཚ༵ན་མ༵་མེད༵་པའི༵་དོན༵་མི༵་དམི༵གས་མཁའ༵་དང༵་སྙོམ༵ས་པ༵འི་དེ་བཞིན་ཉིད་ཀྱི་ཀློང༵་དུ༵་ཐ་དད་མེད་པར་རྟོགས་པས་དབང་བསྒྱུར༵་ནས༵་ལྷུན༵་གྱིས༵་གྲུབ༵་པའི་ཏིང་ངེ་འཛིན་ལ་འཇུག་པ་སྟེ། ཞེས་ཡབ་ཡུམ་སྲས་གསུམ་དང་རྣམ་ཐར་སྒོ་གསུམ་སྦྱར་བ་ནི་སྤྱིར་རྣམ་ཐར་སྒོ་གསུམ་ལ་རྒྱུ་འབྲས་ངོ་བོ་གསུམ་ལ་སྦྱར་བ་དང༌། བྱ་བྱེད་ལས་གསུམ་ལ་སྦྱར་བ་གཉིས་ལས་འདིར་ཕྱི་མའི་དབང་དུ་མཛད་པར་སྣང་ཞིང༌། སྲས་བྱ་བར་བཞག་པ་ནི་བདེ་ཆེན་བསྐྱེད་པའི་བྱ་བ་ལ་བྱས་ནས་དེའི་སྣང་ཆ་སྲས་ཀྱི་རྣམ་པར་ཤར་བ་སྟེ། ཁ་ཅིག་ལོང་སྤྱད་བྱ་སྤྱོད་བྱེད་སྤྱད་པའི་བྱ་བ་གསུམ་པོ་གོ་རིམ་བཞིན་དུ་བྱ་བ་དང་བྱེད་པ་དང་ལས་གསུམ་ལ་འདོད་པ་ནི་དཔལ་ལྡན་ཕྱོགས་གླང་གི་བཞེད་པ་མིན་ནོ། །སྤྱིར་སྲས་བསྐྱེད་ཙམ་ཞིག་བྱ་བར་བཞག་ན་ཀུན་བཏུས་རྩ་འགྲེལ་དང་མི་མཐུན་ཡང༌། སྨོས་མ་ཐག་པ་ལྟར་བྱས་ན་འགྲིགས་པར་མངོན་ནོ། །དེར་ཡང་མ་ཟད་གསུང་བཟླས་བརྗོད་ཀྱི་རྣལ་འབྱོར་སྤྲོ་བསྡུའི་ཏིང་ངེ་འཛིན་ལ་བརྟེན་ནས་བདུད་རྣམ་པ་བཞིའང་བཏུལ་ནས་དངོས་གྲུབ་བསྡུ་བ་སོགས་རྒྱལ་བའི་མཛད་པ་ཡོངས་སུ་རྫོགས་པར་སྒྲུབ་པའི་བར་གྱིས་ཆོ་གའི་ཡན་ལག་རྣམས་མཐར་ཕྱིན་པའི་དོན་འགྲུབ་པའོ། །ཐུན་མོང་མ་ཡིན་པ་རྫོགས་རིམ་དང་སྦྱར་ན། བླ་མེད་ཀྱི་སྦྱོར་ལམ་དྲོ་བ་ནས་བཟོད་པའི་བར་ནི་བསྙེན་པ་ཡིན་ཏེ། བྱང་ཆུབ་ཀྱི་སེམས་འོད་གསལ་ཤེས་པའམ་དངོས་གནས་ལ་མཐོང་ཞིང༌། བདག་ཉིད་ཡེ་ཤེས་སྒྱུ་མའི་ལྷ་སྐུར་དངོས་གནས་སུ་སོང་བ་མེད་པའི་ཕྱིར་རོ། །ཆོས་མཆོག་ནི་ཉེ་བར་བསྙེན་པ་སྟེ། འཇའ་ཚོན་ལྟ་བུའི་སྒྱུ་མ་བདེ་བའི་མཆོག་གི་ལྷ་སྐུར་དངོས་སུ་སོང་བའི་ཕྱིར་རོ། །དེ་ནས་སླར་ཡང་རིག་མའི་འདོད་ཡོན་ཁྱད་པར་ཅན་ལ་བརྟེན་ནས་དོན་གྱི་འོད་གསལ་སྐྱེད་པ་ནི་སྒྲུབ་པ་སྟེ། ཡུམ་སྐྱེད་པ་མཆོག་ཡིན་པའི་ཕྱིར་རོ། །སླར་ཡང་ཟུང་འཇུག་རྡོ་རྗེ་ལྟ་བུའི་གཟུགས་སྐུར་བཞེངས་ཏེ། ཕྱོགས་དུས་མཉམ་པ་ཉིད་ཀྱི་སྤྲུལ་པའི་གཟུགས་རབ་འབྱམས་སུ་སྣང་བ་ནི་སྒྲུབ་པ་ཆེན་པོ་སྟེ། ཐབས་ཀྱི་སྐུ་དང་ཤེས་རབ་ཀྱི་ཐུགས་ཟུང་འབྲེལ་ཏུ་གྱུར་པའི་ཕྱིར་རོ། །དེ་ཡང་སྐུ་མཐར་ཐུག་དང༌། ཡེ་ཤེས་མཐར་ཐུག་ལ་ཨེ་ཝཾ་དང་ཕ་མ་ཞེས་པའི་སྒྲས་བསྙད་པ་རྒྱུད་ན་མང་བས་དེ་ལ་བརྟེན་པའོ། །རྒྱས་པར་ཧེ་རུ་ཀ་གལྤོ་ལས་ཤེས་པར་བྱའོ། །བསྙེན་སྒྲུབ་བཞིའི་སྒྲ་བཤད་ལ་མི་འདྲ་བའི་བཤད་ཚུལ་འགའ་ཞིག་ཡོད་པ་སོགས་ཀྱང་གཞན་ལས་ཤེས་སོ། །དེ་དག་ཐམས་ཅད་ཀྱང༌། བདག་ཉིད་ཆེན་པོ་པྲ་ཛྙཱ་རསྨིས། གསང་བ་སྙིང་པོ་དཔལ་ལྡན་འདུས་པ་དང་སོགས་བསྐྱེད་པ་མ་ཧཱ་ཡོ་ག་ཇི་བཞིན་ཏུ། ཞེས་གསུངས་པའི་རྡོ་རྗེའི་ཚིག་ལ་དཔྱད་ནས་ཅུང་ཟད་འབྲུ་སྣོན་བྱས་པའོ། །བཞི་པ་ནི། ཆོས་ཐམས་ཅད་གདོད་མ་ཉིད་ནས་ངོ་བོ་རྣམ་པར་དག་པའི་ཆོས་ཉིད་ཡིད་བཞིན་གྱི་གཞལ་ཡས་ཁང་རྒྱ་ཡོངས་སུ་མ་ཆད་པའི་འཁོར་ལོར་གནས་པ་ལ། ཡེ་ནས་ཇི་ལྟར་ཡོད་པ་དེ་བཞིན་ཏུ་དེ་དང་རྣམ་པ་འདྲ་བའི་ཐབས་ཀྱིས་ལམ་དུ་བྱས་ཏེ་བླ་ན་མེད་པའི་ཡེ་ཤེས་ཀྱི་དཀྱིལ་འཁོར་དུ་འཇུག་པའང༌། ཞེས་པའི་འང་སྒྲས་བསྙེན་བསྒྲུབ་ཡན་ལག་བཞི་དང་སྦྱར་ནས་རིམ་པ་གཉིས་བཤད་དུ་ཡོད་པར་མ་ཟད་ཅེས་བཤད་རྒྱུར་མངོན་ནོ། །དེ་ལ་རེ་ཞིག་དང་པོར་གསང་སྔགས་ཐབས་ཀྱི་ཐེག་པའི་རྒྱུད་སྡེ་དང་བསྟན་བཅོས་ཀྱི་གཞུང་གདམ་ངག་ཏུ་འཆར་བའི་ཐོས་པ་མང་དུ་ཚོལ་བ་ནི་མིག་ཕྱེ་བ་སྟེ། གདོང་གཡོགས་མ་ཕྱེ་བར་དཀྱིལ་འཁོར་བལྟ་བ་མེད་པ་ཇི་འདྲ་བ་དེ་འདྲའོ། །དེ་ལྟར་ཐོས་པ་བསགས་པའི་མཐུས་རྒྱུད་དོན་ཕྱིན་ཅི་མ་ལོག་པར་དོན་སྤྱིའི་གོ་བ་རྣམ་དག་གི་སྟེང་ནས་རྟོགས་པ་ནི་དཀྱིལ་འཁོར་མཐོང་བ་སྟེ། འདི་ལ་རིགས་པ་ཕྲ་མོ་ཞིག་ཀྱང་ཡོད་དོ། །རྟོགས་པ་ནས་དོན་ཇི་ལྟ་བ་བཞིན་དུ་ཏིང་ངེ་འཛིན་གྱིས་གོམས་པར་བྱེད་པ་ནི་དཀྱིལ་འཁོར་མཐོང་ནས་དེ་ཉིད་དུ༵་ཞུགས་པའོ། །ཞུགས་ནས་སྒོམ་སྟོབས་ཀྱིས་བདེ་བ་ཆེན་པོའི་ཏིང་འཛིན་མངོན་དུ་གྱུར་པ་ནི་དངོས་གྲུབ་ཐོབ་པའོ། །དེ་ལྟར་ན་ཚུལ་འདི་ལ་ནི་བསྐྱེད་པ་ནས་རྫོགས་པ་ཆེན་པོའི་བར་གྱི་ཡེ་ཤེས་ཀྱི་ཚུལ་ཆེན་པོ་མཐར་ཕྱིན་པའི་དོན་ཏོ། །ཡི་གེ་འཁོར་ལོ་ཚོགས་ཆེན་གྱིས་ཞེས་པ་རོང་འགྲེལ་ཏུ་ཡེ་ཤེས་དང་མཚན་མའི་ཡི་གེ་གཉིས་མཛད་ཅིང༌། ཕྱོགས་བཅུ་མུན་སེལ་ཏུ། ཡི་གེའི་སྤྲིན་འཕྲོ་བས་ཡི་གེ་འཁོར་ལོ་ཚོགས་ཆེན་ཞེས་སྤྲུལ་པ་ཆོ་འཕྲུལ་གྱི་ཡི་གེ་དང༌། ཆོས་སྟོན་པ་བརྡའི་ཡི་གེ་རྣམ་པར་འཕྲོ་བ་ལ་མཛད། གཙང་ཊཱིཀ་ཏུ་ནི་ཡི་གེ་ཞེས་པ་ཆོས་ཐམས་ཅད་མིང་ཚིག་ཡི་གེས་བཏགས་ཙམ་མ་གཏོགས་རང་བཞིན་གྱིས་གྲུབ་པ་མེད་དེ། དེ་ལྟར་མེད་བཞིན་དུ་སྣང་ཆ་སྐུ་དང་ཡེ་ཤེས་ཀྱི་འཁོར་ལོ་བསོད་ནམས་དང་ཡེ་ཤེས་ཀྱི་ཚོགས་ཆེན་ཡེ་རྫོགས་སུ་བཞུགས་པ་ཞེས་ཀྱང་བཤད་མོད། འོན་ཀྱང་ཡི་གེ་ནི་ཨ་ཀྵ་རའི་སྒྲ་ལས་མི་འགྱུར་བ་ལ་འཇུག་པས་མཚན་བརྗོད་ལས། ཡི་གེ་མཆོག་སྟེ་རྣམ་པར་དག །ཅེས་པ་འགྲེལ་ཆེན་དྲི་མེད་འོད་དུ་དྲངས་པ་ན། མི་འགྱུར་མཆོག་སྟེ་རྣམ་པར་དག །ཅེས་འབྱུང་བ་ལྟར་འཕོ་འགྱུར་གཉིས་སུ་མེད་པའི་དཀྱིལ་འཁོར་གྱི་འཁོར་ལོའི་ཚོགས་ཆེན་པོ་འཕྲོ་བའི་དོན་ཏེ། ས་དེ་ལ་ཞེས་པ་མི་སློབ་ཟུང་འཇུག་ལ་བྱེད་པས་དེ་ཉིད་དུ་ལྷུན་གྱིས་གྲུབ་པའི་ཚུལ་དུ་འཇུག་པ་སྟེ། ལྷུན་གྱིས་གྲུབ་པའི་ཚུལ་དུ་འཇུག་ལུགས་ནི་ས་བརྒྱད་པ་ཡན་ཆད་ན་འབད་རྩོལ་རགས་པ་ཞི་བའི་ལྷུན་གྱིས་གྲུབ་པའི་ཡེ་ཤེས་ལྟ་བུ་ལ་མི་བྱའི། སྙིགས་དུས་ཀྱི་ཚེ་ཐུང་གཅིག་ལ་སངས་རྒྱས་འགྲུབ་པའི་དོན་ཡིན་ཏེ། སྐྱེས་བུ་བློ་རྩལ་རབ་ཀྱིས་ཕུང་ཁམས་ཐམས་ཅད་ཡེ་ནས་སངས་རྒྱས་ཀྱི་བསྒྲུབ་གཞིར་གནས་པའི་དོན་རིག་ནས་གོམས་པ་དྲག་རྡལ་ལམ་ཤུགས་དྲག་ཏུ་འགྲོ་བ་ཡིན་ནོ། །དེའི་ཕྱིར་གཞི་དུས་ནས་ཡོད་ཡོད་པའི་ཆོས་ཅིག་ཐབས་མཁས་ཀྱིས་ལམ་དུ་བྱས་ནས་མྱུར་དུ་རྣམ་པ་ཐམས་ཅད་མཁྱེན་པའི་གོ་འཕང་དམ་པར་བགྲོད་པའི་སྔགས་བླ་མེད་ཀྱི་ཐབས་མཁས་ཁྱད་པར་ཅན་འདི་ནི། སྐྱེ་བོ་བློ་གྲོས་དམན་པ་ཕལ་གྱིས་བྱ་བ་མ་ཡིན་ནོ། །དེའི་རྒྱུ་མཚན། ཕལ་གྱིས་ཐོས་ཏེ་ཇི་ལྟར་བསམ་ཀྱང་གནས་འདི་ཤིན་ཏུ་རྟོགས་པར་དཀའ་བས་ན་བདེན་ཞིང་མི་བསླུ་བ་དང་ཟབ་ཅིང་གཏིང་དཔག་དཀའ་བའི་དོན་ཡིན་པར་མི་ཤེས་པས་ཡིད་ཆེས་པར་མི་འགྱུར་རོ། །ཡིད་མ་ཆེས་པ་དང་སྐྱེ་བོ་ཕལ་ཆེར་གྱི་བློ་ལ་གོ་དཀའ་ཞིང་བདེན་པ་དང་ཟབ་པ་མ་ཤེས་པའི་རང་ཉམས་དང་སྦྱར་ནས། གཞན་ཀུན་ཀྱང་དེ་དང་འདྲ་སྙམ་ནས་དེ་ཡང་རྫུན་ནོ་ཞེས་སྐྱེས་བུ་རབ་ལ་བསྐུར་བ་འདེབས་ཤིང་སུན་འབྱིན་པའི་བློ་སྐྱེ་བར་འགྱུར་བས་རབ་ཏུ་གསང་བར་བྱ་བ་ཡིན་པའི་ཕྱིར་དེ་བཞིན་གཤེགས་པས་གསང་བའི་ཐེག་པ་ཞེས་བཀའ་སྩལ་ཏོ། །དེ་བས་ན་ཆོས་ཐམས་ཅད་ཡེ་ནས་སངས་རྒྱས་པའི་དོན་ལ་མ་ནོར་བར་རྟོགས་པའི་བློ་གྲོས་མ་སྐྱེས་ཀྱི་བར་དུ་མོས་པ་ཆུང་ངུ་ཐེག་པ་འོག་མ་བས་འགྲོ་བའི་དོན་བྱས་ན་སློབ་མའི་ཉམས་དང་མ་སྦྱར་བར་ཅི་ཐོད་ཐོད་དུ་ཆོས་བསྟན་པས་གདུལ་བྱ་ཆུད་ཟོས་སུ་འགྱུར་ཏེ། འཕགས་པ་ལྷས། གང་ཞིག་གང་གང་ལ་དགའ་བ། །དེ་ཡིས་དེ་ཉིད་སྔར་སྤྱད་བྱ། །ཉམས་པར་གྱུར་པར་དམ་ཆོས་ཀྱི། །སྣོད་ནི་ཅིས་ཀྱང་མ་ཡིན་ནོ། །ཞེས་སོ། །རྒྱུ་མཚན་དེ་ཉིད་ཀྱི་ཕྱིར་གདུལ་བྱ་ཆུད་མི་ཟ་བའི་ཕྱིར་རམ་ཆེད་དུ། སློབ་དཔོན་གྱིས་འཁོར་བའི་སྐྱོན་དང་མྱ་ངན་ལས་འདས་པའི་ཡོན་ཏན་དང༌། གཞན་ཡང་ཐུན་མོང་གི་ཐེག་པས་རྒྱུད་སྨིན་པར་བྱས་ནས་གདོད་ཐུན་མོང་མ་ཡིན་པའི་ཐེག་པ་ལ་གཞུག་པའི་སྒོ་མཐའ་དག་ལ་མཁས་པར་བྱ་བ་ཡིན་གྱི། མདོ་སྔགས་ཀྱི་ཐེག་པའི་ཕྱོགས་འགའ་ཡང་མི་ཤེས་པས་སློབ་དཔོན་གྱིས་བཟུང་དུ་མི་རུང་བར་རྒྱུད་སྡེ་རྒྱ་མཚོ་ནས་འབྱུང་ངོ༌། །གཉིས་པ་ལ། མདོར་བསྟན་པ་དང༌། རྒྱས་པར་བཤད་པ་གཉིས་ལས། དང་པོ་ནི། དེ་ལྟར་ལྟ་བའི་ཁྱད་པར་གྱིས་དེ་དང་འབྲེལ་བའི་དཀའ་ཐུབ་དང་བརྟུལ་ཞུགས་ཀྱང་ནང་ཕན་ཚུན་ཁྱད་འཇུག་པའི་བྱེ་བྲག་གི་དབྱེ་བ་མང་དུ་ཡོད་པར་འགྱུར་རོ། །དེ་རྒྱས་པར་ཕྱེ་ན་སྨོས་ཀྱིས་མི་ལྡང་ཡང༌། འདིར་ནི་དངོས་དང་ཞར་ལ་འོང་བ་གཉིས་སུ་བཤད་པར་བཞེད་དོ། །གཉིས་པ་ལ། ཞར་ལས་བྱུང་བ་དང༌། སྐབས་དོན་དངོས་གཉིས། དང་པོ་ནི། དེ་ལ་དཀའ་ཐུབ་མེད་པ་ནི། འཇིག་རྟེན་ཕྱལ་བ་དང་མུར་ཐུག་པོ། དེའི་རྒྱུ་མཚན། དེ་དག་ནི་ཐར་པ་ཐོབ་པའི་ཆེད་དུ་འཇིག་རྟེན་ཕྱི་མ་ཕན་ཆད་ཀྱི་དོན་ལ་སེམས་པའི་སྒོ་ནས་དཀའ་ཐུབ་སྒྲུབ་པ་མེད་པའི་ཕྱིར་ཏེ། འདི་སྣང་གི་ཚེ་གང་པོ་ཙམ་ཞིག་གཙོ་བོར་འཛིན་པའི་ཕྱིར་རོ། །གཉིས་པ་ནི། དཀ༵་འ་ཐུབ༵་ཡོད༵་པ༵་གང་ཞེ་ན། དེ་ནི༵་རྣམ༵་པ༵་བཞི༵་སྟེ༵་། འཇིག་རྟེན་གྱི་དཀའ་ཐུབ་དང༌། འཇིག་རྟེན་ལས་འདས་པའི་དཀའ་ཐུབ་གསུམ་མོ། །དེ་ལས་དང་པོ་ནི་རྒྱང༵་འཕེ༵ན་དང༵་མུ༵་སྟེག༵ས་པ༵་སྟེ༵། རྒྱང་འཕེན་ཚེ་དང་ནོར་ལ་སོགས་པ་སྒྲུབ་པའི་གསང་སྔགས་ཀྱི་ཀུན་ཏུ་སྤྱོད་པ་ལ་གནས་པའི་ཕྱིར་དང༌། མུ་སྟེགས་རང་གིས་འདོད་པའི་ཐར་བ་དོན་དུ་གཉེར་ཏེ་མེ་ལྔ་དང་ཕྱུགས་ཀྱི་བརྟུལ་ཞུགས་སོགས་ལུས་ལ་གདུང་བ་སྐྱེད་པའི་དཀའ་ཐུབ་སྤྱོད་པ་མང་བའི་ཕྱིར་རོ། །འོན་ཀྱང་རྣམ་འགྲེལ་ལས། བྱམས་སོགས་རྨོངས་དང་འགལ་མེད་ཕྱིར། །ཉེས་པ་ཤིན་ཏུ་ཚར་གཅོད་མིན། །ཞེས་དང༌། འདི་ཡིས་ལྕི་ཉིད་ཟད་འགྱུར་གྱི། །གཟུགས་ཅན་མིན་ཕྱིར་སྡིག་ལྕི་མིན། །ཞེས་སོགས་ཀྱི་སྐབས་སུ་ཇི་སྐད་གསུངས་པ་ལྟར། ལུགས་དེ་དག་ལ་སྲིད་པའི་རྩ་བ་དང་འཛིན་སྟངས་དངོས་སུ་འགལ་བའི་ལམ་བསྒྲུབ་པ་མེད་པས་ཇི་ལྟར་བྱས་ཀྱང་འཇིག་རྟེན་ཏེ་སྲིད་པའི་ར་བ་ལས་མི་འདའ་བའི་ཚུལ། མ་ཏི་ཙི་ཏྲས། ཁྱོད་ཀྱི་བསྟན་ལས་ཕྱིར་ཕྱོགས་པའི། །སྐྱེ་བོ་མ་རིག་གིས་ལྡོང་བ། །སྲིད་རྩེའི་བར་དུ་སོང་ནས་ཀྱང༌། །སྡུག་བསྔལ་ཡང་འབྱུང་སྲིད་པ་བསྒྲུབས། །ཞེས་སོ། །དེའི་ཕྱིར་འཇིག་རྟེན་པའི་དཀའ་ཐུབ་དང་ཞེས་སྨོས་སོ། །ཕྱི་མའི་ཕྱོགས་ལ། ཉན་ཐོས་ཀྱི་དཀའ་ཐུབ་དང༌། བྱང་ཆུབ་སེམས་དཔའི་དཀའ་ཐུབ་དང༌། བླ་ན་མེད་པའི་དཀའ་ཐུབ་བོ། །དོན་ནི་དེར་ཟད་དེ། སླར་ཡང་དེ་གཞུང་དུ་བསྟན་པ་ནི། དེ་ལ་ཕྱལ་བ་ནི་རྒྱུ་འབྲས་ལ་རྨོངས་པའི་ཕྱིར་དཀའ་ཐུབ་མེད་པའོ། །མུར་ཐུག་པ་ནི་ཆད་པར་ལྟ་བའི་ཕྱིར་དཀའ་ཐུབ་མེད་པའོ། །རྒྱང་འཕེན་པ་ནི་ཚེ་འདིའི་ཁྱད་པར་སྒྲུབ་པའི་ཕྱིར་གཙང་སྦྲ་ལ་སོགས་པའི་དཀའ་ཐུབ་ཅན་ནོ། །མུ་སྟེགས་ནི་བདག་རྟག་པ་ཞིག་ཡོད་པ། དེ་དག་པར་བྱ་བའི་ཕྱིར་ལུས་སུན་འབྱིན་ཅིང་མེ་ལྔ་བརྟེན་པ་ལ་སོགས་པའི་དཀའ་ཐུབ་དང་བརྟུལ་ཞུགས་ལོག་པར་སྤྱོད་པའོ། །ཞེས་པ་རྣམས་གོ་སླའོ། །ཉན་ཐོས་ཀྱི་དཀའ་ཐུབ་ནི། འདུལ་བ་ལུང་ལས། སྡིག་པ་ཅི་ཡང་མི་བྱ་སྟེ། །དགེ་བ་ཕུན་སུམ་ཚོགས་པར་སྤྱད། །རང་གི་སེམས་ནི་ཡོངས་སུ་འདུལ། །འདི་ནི་སངས་རྒྱས་བསྟན་པ་ཡིན། །ཞེས་འབྱུང་སྟེ། ལྷག་པ་ཚུལ་ཁྲིམས་ཀྱི་བསླབ་པའི་དབང་དུ་མཛད་ནས་ཉེས་སྤྱོད་ཀྱི་སྒོ་འཕར་གཅོད་པ་སྡིག་པ་ཅི་ཡང་མི་བྱ་བ་དང༌། ལྷག་པ་ཏིང་ངེ་འཛིན་གྱི་བསླབ་པའི་དབང་དུ་མཛད་ནས་སྤོང་བ་བསམ་གཏན་གྱི་འཁོར་ལོ་ལ་གནས་པ་དགེ་ཆོས་ཕུན་སུམ་ཚོགས་སྤྱོད་པ་དང༌། ལྷག་པ་ཤེས་རབ་ཀྱི་བསླབ་པའི་དབང་དུ་མཛད་ནས་ཉོན་མོངས་པའི་འཆིང་བ་གཅོད་པ་རང་སེམས་ཡོངས་སུ་འདུལ་བ་སྟེ། སངས་རྒྱས་ཀྱི་བསྟན་པ་རྟོགས་པའི་ཆོས་རྣམས་ཀྱི་འགགས་སྡོམ་ནི་དེ་ཙམ་དུ་འདུས་སོ་ཞེས་གསུངས་པའོ། །སྐབས་འདིའི་དྲང་མཚམས་ནི། དགེ་བ་དང་མི་དགེ་བའི་ཆོས་ཐམས་ཅད་ཀུན་རྫོབ་དང་དོན་དམ་པ་གཉིས་ཆར་བདེན་གྲུབ་ཏུ་སོ་སོ་ནས་ཡོད་པར་ལྟ་བ་དང༌། ངེས་འབྱུང་གི་བསམ་པ་ཁྱད་པར་ཅན་གྱིས་ཡིད་དྲངས་ཏེ། སོ་སོ་ཐར་པའི་ཚུལ་ཁྲིམས་ཡང་དག་པར་བླངས་ནས། དགེ་བ་ནི་སྤྱོད༑ མི་དགེ་བ་ནི་སྤོང་བའི་དཀའ་ཐུབ་དང་བརྟུལ་ཞུགས་སྤྱོད་པ་སྟེ། གཙོ་བོར་རྐང་པ་དང་པོའི་ཚིག་དེའི་དྲང་མཚམས་སུ་འགྲོའོ། །དགེ་བ་དང་སྡིག་པ་བདེན་པ་གཉིས་ཀ་ན་གྲུབ་པར་བལྟ་ཞེས་པའི་སྐབས་སུ་རྩོད་ཚུལ་མང་ཡང༌། ཉན་ཐོས་སྡེ་པའི་དབང་དུ་བཏང་ན་གང་དང་ཡང་འགལ་བ་མི་སྣང་ངོ༌། །བྱང་ཆུབ་སེམས་དཔའི་དཀའ་ཐུབ་ནི། བྱང་ཆུབ་སེམས་དཔའི་སྡོམ་པ་ཉི་ཤུ་པ་ལས། རྐྱེན་དུ་འཚམ་པར་དོན་མི་བྱེད། །རྫུ་འཕྲུལ་བསྡིགས་ལ་སོགས་མི་བྱེད། །སྙིང་རྗེར་ལྡན་ཞིང་བྱམས་ཕྱིར་དང༌། །སེམས་དགེ་བ་ལ་ཉེས་པ་མེད། །ཅེས་འབྱུང་སྟེ། སེམས་ཅན་ཐམས་ཅད་སྲིད་པའི་སྡུག་བསྔལ་ལས་བསྒྲལ་འདོད་ཀྱི་སྙིང་རྗེ་ཆེན་པོའི་བློས་ཟིན་ན་ཆོས་ཐམས་ཅད་ཐེག་པ་ཆེན་པོར་འགྱུར་རོ། །ཆོས་ལ་སྒྲ་འཇུག་པའི་ཡུལ་བཅུ་ཡོད་པར་བཤད་ཀྱང༌། སྐབས་འདིར་ལམ་ལ་ཆོས་ཞེས་གསུངས་པའོ། །དེའི་ཕྱིར་རྣམ་པའམ་ལྟར་སྣང་དུ་དགེ་བ་དང་མི་དགེ་བ་གང་སྤྱོད་ཀྱང་བྱང་སེམས་ཀྱི་སྡོམ་པ་ཉམ་པར་མི་འགྱུར་ཏེ། བྱང་ཆུབ་སེམས་དཔའི་སྡོམ་པ་ནི་མདོར་ན་སྙིང་རྗེ་ཆེན་པོས་གཞི་བཟུང་ནས་སྤྱོད་དེ། གསང་ཆེན་ཐབས་ལ་མཁས་པའི་མདོར་དེད་དཔོན་སྙིང་རྗེ་ཆེན་པོའི་བྱུང་བ་བཞིན་ནོ། །དེ་བས་ན་ཚིག་འདི་ནི་རྒྱལ་བའི་སྲས་ཀྱི་སྤྱོད་པ་སྦྱིན་སོགས་ཐབས་དང་ཤེས་རབ་ཐམས་ཅད་བསོད་ནམས་དང་ཡེ་ཤེས་ཀྱི་ཚོགས་ཞེས་བྱ་བའི་མིང་འཐོབ་པ་ནི་རྩ་བ་སྙིང་རྗེ་ཆེན་པོ་ལ་རག་ལས་པས། དེས་ཟིན་ཞིང་སེམས་ཅན་གྱི་དོན་དུ་འགྱུར་ན་སྲོག་གཅོད་དང་བསྡིགས་བསྐྲད་ལ་སོགས་པ་རྣམས་ཀྱང་དགེ་བའི་ཚོགས་སུ་འགྱུར་ན་གཞན་ལྟ་ཡང་ཅི་སྨོས་ཞེས་པའི་དོན་ཡིན་གྱི། གསོད་གཅོད་རྒྱལ་བའི་སྲས་ཀྱི་དཀའ་ཐུབ་ཀྱི་སྡོམ་དུ་བསྟན་པ་ལྟ་བུ་གཏན་མིན་ཏེ། རྒྱས་པར་དཔལ་ལྡན་ཞི་བ་འཚོའི་སྡོམ་པ་ཉི་ཤུ་པའི་འགྲེལ་བ་ལས་ཤེས་དགོས་པར་ཡོད་དོ༵། །བླ་ན་མེད་པ་སྔགས་ཀྱི་དཀའ་ཐུབ་ནི། དམ་ཚིག་ཆེན་པོའི་མདོ་ལས། སངས་རྒྱས་ཐེག་པར་རབ་ངེས་ན། །ཉོན་མོངས་འདོད་ལྔ་ཀུན་སྤྱོད་ཀྱང༌། །པདྨ་ལ་ནི་འདམ་བཞིན་ཏེ། །དེ་ལ་ཚུལ་ཁྲིམས་ཕུན་སུམ་ཚོགས། །ཞེས་གསུངས་ཏེ། དེའང་རྡོ་རྗེ་སྙིང་པོ་རྒྱན་ལས། འདོད་ཆགས་རྣམ་པ་བཞིའི་གཉེན་པོར་སྔགས་ཀྱི་རྒྱུད་སྡེ་ཆེན་པོ་བཞི་གསུངས་པར་བཤད་པ་ལྟར། སྔགས་ཀྱི་ཆེད་དུ་བྱ་བའི་གདུལ་བྱ་ནི་ཉོན་མོངས་ལ་ལམ་བྱེད་ཀྱི་ཐབས་མཁས་ལ་འཇུག་ནུས་པའི་གང་ཟག་ཁྱད་པར་ཅན་ཁོ་ནར་ངེས་ཤིང༌། དེ་དག་གིས་ནི། ཆོས་ཐམས་ཅད་ཡེ་ནས་མཉམ་པ་ཉིད་དུ་རྟོགས་པའི་ཤེས་རབ་དང༌། གདུལ་བྱ་སྙིང་རྗེས་མི་གཏོང་བའི་ཐབས་ཆེན་པོ་ཅན་ཉིད་ཀྱི་ཕྱིར། གཞན་དོན་དུ་འགྱུར་ན་མངོན་སྤྱོད་ལ་སོགས་པའི་དུས་སུ་སྟོངས་ཀྱི་སེམས་ཞེ་སྡང་སྐྱེད་པར་གསུངས་པ་ལྟ་བུའི་དམིགས་བསལ་བའི་སྐབས་ན་སྙིང་རྗེ་ནི་བསྟེན་དུ་མེད་ལ། ཞེ་སྡང་ནི་སྤང་དུ་མེད་པ་སྟེ་ཆེད་དུ་མ་སྤང་ཡང་ཤིང་ལས་བྱུང་བའི་སྲིན་བུས་ཤིང་ཉིད་ཟ་བ་བཞིན་དུ་ཐར་པ་མཆོག་གི་བར་ཆད་དུ་གྱུར་པའི་ཆོས་རྣམས་སྦྱོང་བའི་ལྷན་ཅིག་བྱེད་རྐྱེན་དུ་འགྱུར་བ་ནི། དཔལ་མཆོག་དང་པོར་རྒྱུའི་ཀུན་སློང་སྙིང་རྗེ་ཆེན་པོས་དྲངས་ནས་དུས་ཀྱི་ཀུན་སློང་དྲག་ཤུལ་གྱི་སེམས་སུ་བྱས་ཏེ་ལས་འགའ་ཞིག་རྩོམ་པར་གསུང་བ་བཞིན་ནོ། །དེ་བས་ན་རྒྱུའི་ཀུན་སློང་ནི་ཇི་ལྟར་ཡང་སྙིང་རྗེ་ཆེན་པོས་གཞི་བཟུང་བ་ཞིག་དགོས་སོ་ཞེས་བསྟན་པའི་ཕྱིར། དེ་ལྟར་ཏེ་མཉམ་ཉིད་ཀྱི་གནས་ལུགས་རྟོགས་པའི་ཐབས་མཁས་ཀྱི་ཚུལ་མ་རྟོགས་པ་རྣམས་ལ་ཐུགས་རྗེ་ཆེན་པོ་མི་འབྱུང་བ་ནི་མ་ཡིན་ཏེ། ཞེས་གསུངས་སོ། །མདོར་ན་ཇི་ལྟར་ལྟ་བས་ཡེ་ནས་རང་བཞིན་རྣམ་པར་དག་པར་རྟོགས་པ་བཞིན་དུ་དེའི་ཚོད་དང་སྦྱར་བའི་དཀའ་ཐུབ་དང་བརྟུལ་ཞུགས་ཀྱང་དེ་ལྟར་རྣམ་པར་དག་པར་སྤྱོད་པ་བྱའི། རང་གི་ཉམས་དང་མ་སྦྱར་བའི་སྤྱོད་པ་དེ་འདྲ་བ་དང་པོ་ནས་རྩོམ་པ་བཀག་སྟེ། ཞི་འཚོའི་དེ་ཉིད་གྲུབ་པར། ཀེ་ཏ་ཀའི་མེ་ཏོག་གླང་པོ་ཆེ་གཞན་གྱིས་ཟོས་པ་ལས་བྲུན་དང༌། ཕལ་གྱི་གླང་པོ་ཆེས་ཟོས་ན་གླ་རྩི་ལྟ་བུར་འགྱུར་བའི་དཔེ་གྲངས་མཚམ་བཞིན་ནོ། །དེས་ན་སྤྱོད་པ་རྒྱུད་ཚོད་དང་སྦྱར་བར་བྱ་སྟེ༑ འདི་ནི་སྔགས་བླ་མེད་ཁོ་ནའི་སྤྱོད་པ་ཐུན་མིན་མིན་གྱི། རྒྱུད་སྡེ་འོག་མའི་སྲོལ་ཀྱང་ཡིན་ནོ། །གསུམ་པ་མཇུག་གི་དོན་ལ། གདམ་ངག་གང་ལ་གཏད་པའི་སྣོད་དང༌། བརྩམ་ཟིན་པའི་མཐའ་བྱང་གདགས་པ་གཉིས་ལས། དང་པོ་ནི། བི་མ་མི་ཏྲའི་ཁོག་གཞུང་ལས། བློ་རྣོ་ཟབ་བརྟན་མྱུར་བ་དང༌། །ཞེས་སོགས་གསུངས་པ་བཞིན། སྣོད་ཀྱི་གང་ཟག་ཁྱད་པར་ཅན་ཞིག་ལ་མན་ངག་ལེགས་པར་གཏད་པ་མེད་ན། སྣོད་མིན་གྱི་རྒྱུད་ལ་མི་འབྱོར་བས་གསང་སྔགས་འདྲེ་ལ་གཏད་པ་ལྟ་བུའི་ཚུལ་དུ་འགྱུར་ཞིང་གདམ་པའི་སརྒ་ལོག་པ་དང༌། བརྒྱུད་པའི་བྱིན་བརླབ་ཟབ་མོ་ཉམ་དུ་འགྲོ་བས་ན་འདིར་གསུངས་བ། ལྟ༵་བ༵འི་ཕྲེང༵་བ༵་ཞེས་བྱ་བའི་གསང་བ་ཆེན་པོའི་མན་ངག་འདི༵་ནི། མི་མཁས་པའི་བྱིས་པ། རྟོག་གེ་ངན་པའི་གདོན་གྱིས་བརླམས་ཤིང༌། རྡོ་རྗེའི་ལམ་ལ་མོས་པ་ཆུང་བ་རྣམས་ཀྱི་སྤྱོད་ཡུལ་མིན་ནོ། །འོ་ན་སུ་ཞིག་གི་ཡུལ་དུ་འགྱུར་ཞེ་ན། ཇི་ལྟར་དམུས༵་ལོང༵་གི་མིག་གི་སྒྲིབ་བྱེད་རང༵་བྱེ་ནས༵་ཡོངས་སུ་དག་པའི་མིག༵་རྙེད༵་པ་ལྟར། ཟབ་མོའི་ཆོས་ལ་འཇུག་པ་ཡངས་པའི་ཤེ༵ས་རབ༵་དང༌། བཤེས་གཉེན་དམ་པ་བསྟེན་པའི་ཐབ༵་ས་མཁས་པ་སོགས་ཚོགས་ཀྱི༵་བསགས་རྒྱབ་མང་ཞིང༌། དེ་འདྲའི་རྣམ་དཔྱོད་ཀྱི་སྒྱུ་རྩལ༵་འཆ༵ང་བ༵་འི་སྐྱེས༵་བུ་མཆོག་རྣམས་ཀྱི་ཡུལ་ཏུ་འགྱུར་རོ། །འོན་ཀྱང་སྙིགས་མའི་དུས་སུ་སྐལ་བ་ཕུན་སུམ་ཚོགས་པའི་སློབ་མ་དེ་ལྟ་བུ་ཨུ་དུམ་ཝ་ར་ལྟར་རྙེད་པར་དཀའ་མོད་ཀྱི། བརྒྱ་ལམ་དུ་ཡོད༵་ན༵་དེ་དག་དང་འཕྲ༵ད་ནས་དོན་དུ་འགྱུར༵་པར་ཤོག་ཅིག༵་ཅེས་སངས་རྒྱས་གཉིས་པ་དེས་ཐུགས་ཤིན་ཏུ་བརྩེ་བས་སྨོན་ལམ་དུ་མཛད་པའོ། །གཉིས་པ་ནི། ལྟ་བའི་ཕྲེང་བ་ཞེས་བྱ་བ་སྒྱུ་འཕྲུལ་དྲ་བའི་རྩ་བའི་རྒྱུད་ཀྱི་མན་ངག །གསང་སྔགས་ཐམས་ཅད་ཀྱི་སྤྱིའི་ཆིངས། ལུང་ཐམས་ཅད་ཀྱི་མདོ་འགགས། ཐེག་པ་མཐའ་དག་གི་སྣང་བྱེད། བསྟན་པ་ཐམས་ཅད་ཀྱི་སྤྱི་སྡོམ། དཔའ་བོ་དང་རྣལ་འབྱོར་མ་རྣམས་ཀྱི་གསང་བའི་མཛོད་ཆེན་པོ་རྫོགས་སོ་སྟེ་བརྩམས་ཟིན་པའི་དོན་ནོ། །དེ་ལྟར་ན་སངས་རྒྱས་གཉིས་པ་པདྨཱ་ཀ་རའི་རྡོ་རྗེའི་གསུང་འཁྲུལ་པ་དང་རྙོགས་པའི་བསླད་པ་མེད་པ་འདི་ཉིད་གངས་ཅན་དུ་བྱུང་བའི་རྒྱ་གཞུང་རྣམས་ཀྱི་སྔ་ཤོས་དང་སྒྲུབ་བརྒྱུད་ཤིང་རྟ་ཆེན་པོའི་རིང་ལུགས་ཡོངས་སུ་རྫོགས་པའི་ཆུ་འགོ་ལྟ་བུར་གྱུར་པ་ཡིན་པས། ཕྱོགས་ཞེན་གྱིས་ཀུན་ནས་བསླངས་པའི་བླང་དོར་གདོན་གྱིས་རང་རྒྱུད་མི་དཀྲུགས་པར་ཐར་འདོད་བློ་གྲོས་རྣམ་པར་ཡངས་པ་རྣམས་ཀྱིས་སྐལ་པ་དང་ལྡན་པའི་གང་ཟག་རྣམས་ལ་ལེགས་པར་འཆད་ཅིང་རྒྱས་པར་སྤེལ་བས་རྒྱལ་བ་དགྱེས་པའི་ཕྲིན་ལས་ཕྱོགས་མེད་དུ་བསྐྱང་བར་མཛད་འཚལ་ལོ། །བདག་འདྲ་དམ་པའི་རྣམ་ཐར་གྱིས་དབུལ་ཡང༌། །འཁྲུལ་མེད་དམ་པའི་མགོན་གྱིས་རྗེས་བཟུང་ཞིང༌། །རིས་མེད་ཟབ་དགུའི་ཆོས་ཀྱི་ནོར་བུ་རྙེད། །བསྟན་འགྲོར་ཕན་འདོད་ལྷག་བསམ་ཟོལ་མེད་པས། །འདི་ཡང་སངས་རྒྱས་རྔ་ཡབ་གླིང་པ་ཡི། །ཁྱད་འཕགས་གཞུང་གི་རྒྱུན་བཟང་སྤེལ་བའི་བློས། །གདམས་ཁྲིད་གཅེས་བསྡུས་མཛོད་ཆེན་ནང་ནོར་དུ། །བཞུགས་ཕྱིར་ལེགས་བཤད་དབྱིག་གི་སྙིང་པོ་བསྡུས། །སྐྱེ་བར་ཟབ་རྒྱས་མདོ་སྔགས་ཆོས་མཛོད་འཛིན། །རྒྱལ་དབང་པདྨའི་ཕྲིན་ལས་མཁའ་ཁྱབ་སྒྲུབ། །མཐའ་ཡས་འགྲོ་ཀུན་གདོད་མའི་ས་མཆོག་ཏུ། །སྒྲོལ་བའི་རྣམ་ཐར་ཆོས་དབྱིངས་མཐར་ཁྱབ་ཤོག །འདིའང་གདོད་མའི་མགོན་པོ་བདེ་བ་ཆེན་པོ་དང་ཞལ་མི་གཉིས་པ་རྗེ་བླ་མ་པདྨ་འོད་གསལ་མདོ་སྔགས་གླིང་པའི་ཞལ་སྔ་ནས་རྩ་བའི་ལུང་དང་གནད་དོན་བཀྲོལ་བའི་ལྗགས་བཤད་རྗེས་གནང་དང་བཅས་པ་སྩལ་ནས་གདམས་མཛོད་དུ་འཁོད་པར་གྱིས་ཤིག་པའི་བཀའ་ལུང་སྤྱི་བོར་ཕེབས་པར་བརྟེན། རིས་མེད་ཀྱི་རྣལ་འབྱོར་སྤྱོད་པ་བློ་གྲོས་མཐའ་ཡས་པའི་སྡེས་བླ་མའི་ཞལ་ལུང་མདོར་བསྡུས་ཏེ་བཀོད་པ་དགེ་ལེགས་འཕེལ༎ ༎
