༄༅། བཅོམ་ལྡན་འདས་དཔལ་རྡོ་རྗེ་སེམས་དཔའ་ལ་ཕྱག་འཚལ་ལོ། ༈ །ཤེས་བྱའི་ཡུལ་ལ་རྣམ་པར་བརྟག །ཐར་པའི་ལམ་ལ་རབ་ཞུགས་ཏེ། །འབྲས་བུ་རྣམ་མཁྱེན་ཐོབ་བྱའི་ཕྱིར། །རིན་ཆེན་སྤུངས་པ་བསྟན་པར་བྱ། ༈ །མཚན་ཉིད་དྲུག་ལྡན་ བརྒྱུད་པ་ལྡན་ཞིང་གདམས་ངག་ཆེ། །དབང་དང་ལྡན་ཞིང་བྱིན་རླབས་ཆེ། །ཤེས་རབ་ལྡན་ཞིང་སྙིང་རྗེ་ཆེ། །ཐོས་པ་ལྡན་ཞིང་ཉམས་མྱོང་ཆེ། །ལྟ་བ་ལྡན་ཞིང་སྒོམ་གྱིས་ཟིན། །ཚིག་ལ་མཁས་ཤིང་དོན་ལ་གནས། །ཞེས་པ་སྟེ། བླ་མ་ནི། །འདུ་ཤེས་བཞི་ ལ་ཐུན་མོང༌། ཁྱད་པར། ཡང་དག །ལོག་པའི་འདུ་ཤེས་བཞི། དེ་རེ་རེ་ལའང་བཞི་བཞི་སྟེ། དང་པོ་ལ། འཁོར་བ་ཆུ་བོ། ཆོས་གྲུ། བླ་མ་མཉན་པ། ཉམས་ལེན་ཆུ་བརྒལ་བའི་ཚུལ་ལོ། །གཉིས་པ་ལ། རང་ནད་པ། བླ་མ་སྨན་པ། ཆོས་ལ་སྨན། ཉམས་ལེན་ནད་གསོ་བའི་འདུ་ཤེས་སོ། །གསུམ་པ་ལ། རང་དུག་ལྔའི་ནད་ཀྱིས་ཟིན་པ། བླ་མ་སངས་རྒྱས། གདམས་ངག་བདུད་རྩི། ཉམས་ལེན་དུག་ལྔ་སེལ་བའི་འདུ་ཤེས། བཞི་པ་ལ། རང་རྒྱལ་པོ། བླ་མ་ཚོང་པ། ཆོས་ཚོང་ཟོང་། ཉན་བཤད་ཚོང་གི་འདུ་ཤེས་ཏེ། དེ་རྣམས་ལས་ཕྱི་མ་བཞི་སྤང་ལ། སྔ་མ་རྣམས་དང༌། ལྡན་ པའི་དྲན་པས་བསྟེན། །སྣོད་ཀྱི་སྐྱོན་ཁ་སྦུབས། ཞབས་བརྡོལ། དུག་ཅན། གསུམ་སྤངས་ལ་བསྙེན། །མཐོང་ལུགས་སངས་རྒྱས་ཐམས་ཅད་འདུས་པ། རྡོ་རྗེ་འཛིན་པའི་ངོ་བོ། དཀོན་མཆོག་གསུམ་གྱི་རྩ་བ། དཔེ་ནོར་བུ། དོན་སངས་རྒྱས་ལས་ཀྱང་ལྷག་པར་བལྟ་བ་སྟེ། ལྔ་ལྡན་སྒོ་ནས་བལྟ། ༈ །དལ་འབྱོར་མི་ལུས་རིན་པོ་ཆེ༔ ཤིན་ཏུ་རྙེད་དཀའ་ཐོབ་དུས་སུ༔ དོན་མེད་སྟོང་ལོག་མ་བྱེད་པར༔ གཏན་གྱི་སོ་ནམ་ད་རེས་གྱིས༔ ཕྱི་ཡི་མི་རྟག་ དབྱར་དགུན། དུས་བཞི། ཉིན་མཚན། སྐྱེ་འགག་གནས་གསུམ། རྒྱུ་ཟག་པ་དང་མ་ཟད་པ་ཡང་མི་རྟག་པ་སྟེ། རྣམ་པ་ལྔ༔ ནང་གི་མི་རྟག་ ངེས་པར་འཆི། ནམ་འཆིའི་ཆ་མེད། འཆི་བའི་ཚེ་ཅིས་ཀྱང་མི་སྟོངས་པ་སྟེ། རྣམ་པ་གསུམ༔ ཚུལ་བདུན་ ཅིས་ཀྱང་འཆི། སྐྱབས་གནས་མེད། ལམ་རྒྱུས་མེད། ཅིས་ཀྱང་མི་བཟློག །མགྱོགས་པར་འགྲོ། གཟུར་ས་མེད། ཤིན་ཏུ་ཉམ་ང་བར་འགྲོ་དགོས་པ་སྟེ། ལྡན་པས་བསྒོམ་བྱ་ཞིང༔ ཕན་ཡོན་ རྒྱུད་ལ་དད་པ་སྐྱེ། བླ་མ་ལ་མོས་གུས་འབྱུང༌། གདམས་ངག་ལ་ངེས་ཤེས་སྐྱེ། ཉམས་ལེན་ལ་འཇུག་ནུས། སྟོང་ཉིད་རྟོགས་སླ་བ་སྟེ། ལྔ་ལྡན་སྒོ་ནས་འབད༔ ཡུལ་དང་ལུས་དང་སེམས་གསུམ་ལ༔ སྐྱེ་བོ་མུ་སྟེགས་ཉན་རང་གིས༔ གང་ཟག་ཆོས་ཀྱི་བདག་ཏུ་བཟུང༔ བདག་འཛིན་མ་གྲུབ་བདག་མེད་སྟོང༔ རྩ་བའི་སྡུག་བསྔལ་བཞི་ལ་གནས༔ ཡན་ལག་སྡུག་བསྔལ་དྲུག་ལ་སྤྱོད༔ ཉིང་ལག་སྡུག་བསྔལ་བགྲང་ལས་འདས༔ རང་སེམས་སྟོང་པ་ཉིད་དུ་སྤྲད༔ དགེ་སྡིག་གཉིས་ཀྱིས་རྒྱུ་བྱས་ནས༔ འབྲས་བུ་བདེ་སྡུག་གཉིས་སུ་འབྱིན༔ དགེ་སྡིག་སོ་མཚམས་ལེགས་པར་དབྱེ༔ སྡིག་པ་སྟོབས་བཞིའི་སྒོ་ནས་བཤགས༔ ༈ ༔སངས་རྒྱས་ཆོས་དང་དགེ་འདུན་ལ༔ ཉན་རང་བྱང་སེམས་གསང་སྔགས་གསུམ༔ རང་དང་གཞན་དང་གཉིས་དོན་དུ༔ སྐྱབས་འགྲོ་རྣམ་གསུམ་འདས་པའི་ཆོས༔ སངས་རྒྱས་ཐོབ་འདོད་བསམ་པའི་དོན༔ སྒྲུབ་ཅིང་ཉམས་སུ་ལེན་པ་དང༔ ཆོས་ཉིད་བློ་འདས་འདུས་མ་བྱས༔ སེམས་བསྐྱེད་གསུམ་ཡིན་ཐེག་ཆེན་ཆོས༔ སྡིག་པ་རྣམ་གསུམ་མི་བྱ་ཞིང༔ དགེ་བ་རྣམ་གསུམ་སྤྱད་པར་བྱ༔ གནས་ལུགས་རང་ངོ་ཤེས་པའོ༔ སྡོམ་གསུམ་སངས་རྒྱས་བསྟན་པ་ཡིན༔ བྱམས་པའི་སེམས་ལ་ཞེ་སྡང་མེད༔ སྙིང་རྗེའི་སེམས་ལ་འདོད་ཆགས་མེད༔ བྱང་ཆུབ་སེམས་ལ་གཏི་མུག་མེད༔ སེམས་གསུམ་སངས་རྒྱས་ཐུགས་དགོངས་ཡིན༔ གནོད་སྦྱོར་ལོག་པ་ཚུལ་ཁྲིམས་ཏེ༔ ཕན་འདོད་ཞུགས་པ་ཏིང་འཛིན་ཡིན༔ ཐབས་ཤེས་ཟུང་འཇུག་ཤེས་རབ་སྟེ༔ བསླབ་གསུམ་སྡེ་སྣོད་གསུམ་ཡིན་ནོ༔ ༈ ༔རང་བྱུང་ངོ་བོ་ཉིད་ཀྱི་གཞི༔ རང་བཞིན་སྐྱེ་མེད་སྟོང་པར་གནས༔ རོལ་པ་འགག་མེད་སྣང་བ་འཆར༔ ངོ་བོ་གཉིས་མེད་མཐའ་བྲལ་ལོ༔ ངོ་མ་ཤེས་ལ་ལོག་པར་བརྟགས༔ ཉོན་མོངས་ལྔ་དང་ལས་གསུམ་གྱིས༔ རིགས་དྲུག་ལུས་བླངས་སྡུག་བསྔལ་མྱོང༔ མ་རིག་ནས་བརྩམ་རྒ་ཤིས་འཁོར༔ ཡིན་ཚུལ་ཅིར་ཡང་མ་ངེས་པས༔ སྣང་ཚུལ་ཅིར་ཡང་འཆར་བར་བྱེད༔ སྣང་ཚད་སེམས་ཉིད་རང་སྣང་ཡིན༔ རང་གྲོལ་རྟོགས་པ་བལྟ་བའོ༔ སོ་མ་རང་ཐང་ལྷུག་པར་བཞག༔ རྟོག་མེད་རང་གསལ་རྗེན་པར་འཆར༔ སྣང་སེམས་དབྱེར་མེད་ངོ་བོ་གཅིག༔ གསལ་སྟོང་དབྱེར་མེད་བསྒོམ་པ་ཡིན༔ རྐྱེན་སྣང་ཕྲ་རགས་མཐོ་དམན་ཀུན༔ རྟོག་པ་སེམས་ཀྱི་ཆོ་འཕྲུལ་ཡིན༔ ཞུམ་དང་ཁེངས་པ་མེད་པ་རུ༔ ཞེན་མེད་རང་གྲོལ་སྤྱོད་པ་ཡིན༔ གང་བྱེད་རང་སོ་ངང་ལ་གནས༔ དེ་ཡི་ངོ་བོ་རང་གསལ་བ༔ དེ་ཀ་སྟོང་པ་ཉིད་ཡིན་ཏེ༔ སྟོང་པ་འགག་པ་མེད་པ་ཡིན༔ མ་གྲུབ་མི་རྟོག་མི་འགག་པ༔ རྣམ་དག་ལྷུན་གྲུབ་བདེན་གསུམ་དང༔ ལྟ་སྒོམ་སྤྱོད་པ་འབྲས་བུའི་ཆོས༔ ཤེས་བྱ་ཀུན་ལྡན་ལ་ཟློ་ཡིན༔ ༈ ༔ངེས་པ་གཉིས་ཟད་ གཞིའི་ངེས་པ་ཟད་དེ་གནས་ལུགས་མངོན་དུ་གྱུར། ལམ་གྱི་ངེས་པ་ཟད་དེ་ལམ་མཐར་ཕྱིན་པའོ། །སྐུ་རུ་སྨིན༔ ཇི་ལྟ་ཇི་སྙེད་མཁྱེན་གསུམ་རྒྱས༔ རྟེན་ཅན་རྟེན་མེད་ངོ་བོ་ཉིད༔ གཅིག་དང་དུ་མ་བྲལ་བར་བཞུགས༔ ༈ ༔དགེ་བ་འདི་ཡི་བསོད་ནམས་ཀྱིས༔ ནམ་མཁའ་མཉམ་པའི་སེམས་ཅན་རྣམས༔ ཡང་དག་དོན་དང་རབ་མཇལ་ནས༔ རྫོགས་པའི་སངས་རྒྱས་མྱུར་ཐོབ་ཤོག༔ རིན་ཆེན་སྤུངས་པ་རྫོགས་སོ༔ ༈ ཕྱི་བཤད་རིན་ཆེན་སྤུངས་པའི་བསྡུས་དོན་བཞུགས་སོ། །བླ་མ་དམ་པ་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །ཕྱི་བཤད་རིན་པོ་ཆེ་སྤུངས་པ་འདི་ལ་དོན་གསུམ་སྟེ། ཀླད་ཀྱི་དོན་དང༌། གཞུང་གི་དོན་དང༌། མཇུག་གི་དོན་ནོ། །དང་པོ་ལ་གཉིས་ཏེ། བཅོམ་ལྡན་འདས་དཔལ་ ཕྱག་འཚལ་བ་དང༌། བཤད་པར་དམ་བཅའ་བའོ། །དེ་ལ་བཞི་སྟེ། ཤེས་བྱའི་ ཤེས་བྱ་ལ་བརྟགས་པ་དང༌། ཐར་པའི་ ཐར་པའི་ལམ་ལ་འཇུག་པ་དང༌། འབྲས་བུ་རྣམ་མཁྱེན་ འབྲས་བུ་ཐོབ་ཚུལ། མན་ངག་གི་མཚན་ རིན་ཆེན་སྤུངས་པ་ བསྟན་པའོ། །གཉིས་པ་གཞུང་དོན་ལ་གསུམ། དྲང་དོན་རྒྱུད་བསྐུལ་གྱི་གདམས་པ་དང༌། ཐུན་མོང་གི་མན་ངག་དང༌། ཁྱད་པར་རྫོགས་པ་ཆེན་པོའི་གདམས་ངག་གོ །དང་པོ་ལ་དྲུག་སྟེ། བླ་མ་བསྟེན་པ་དང༌། དལ་འབྱོར་བརྟག་པ་དང༌། མི་རྟག་པ་བསྒོམ་ཚུལ་དང༌། བདག་འཛིན་གཞོམས་པ་དང༌། སྡུག་བསྔལ་བསམ་པ་དང༌། རྒྱུ་འབྲས་དཔྱད་པའོ། །དང་པོ་ལ་བཞི། མཚན་ཉིད་དྲུག་ བླ་མའི་མཚན་ཉིད་དང༌། འདུ་ཤེས་བཞི་ བསྟེན་ཐབས་དང༌། སྣོད་ཀྱི་ མཉན་ཚུལ་དང༌། མཐོང་ལུགས་ མཐོང་ལུགས་སམ་བཀུར་ཐབས་སོ། །གཉིས་པ་དལ་འབྱོར་བརྟགས་པ་ལ་བཞི། དལ་འབྱོར་མི་ལུས་ དལ་འབྱོར་ངོས་བཟུང་བ་དང༌། ཤིན་ཏུ་དཀོན་ ཐོབ་པར་དཀའ་བར་བསམ་པ་དང༌། དོན་མེད་སྟོང་ལོག་ ཡལ་བར་མི་དོར་བ་དང༌། གཏན་གྱི་སོ་ནམ་ དོན་ཡོད་པར་བྱ་བའོ། །གསུམ་པ་མི་རྟག་པ་བསྒོམ་པ་ལ་བཞི། ཕྱིའི་མི་རྟག་ ཕྱིའི་མི་རྟག་པ་ལྔ་དང༌། ནང་གི་མི་རྟག་ ནང་གི་མི་རྟག་པ་གསུམ་དང༌། ཚུལ་བདུན་བསྟན་ཚུལ་ཇི་ལྟར་བསྒོམ་པ་དང༌། ཕན་ཡོན་ལྔ་ལྡན་ ཕན་ཡོན་གྱི་སྒོ་ནས་འབད་པའོ། །བཞི་པ་བདག་འཛིན་གཞོམ་པ་ལ་བཞི། ཡུལ་དང་དུས་དང་ གང་ལ་བདག་ཏུ་འཛིན་པ་དང༌། སྐྱེ་བོ་མུ་སྟེགས་ གང་གིས་བདག་ཏུ་འཛིན་པ་དང༌། གང་ཟག་དང་ཆོས་ཀྱི་བདག་ ཇི་ལྟར་བདག་ཏུ་འཛིན་པ་དང༌། བདག་འཛིན་མ་གྲུབ་ བདག་འཛིན་བཀག་ལ་བདག་མེད་དུ་བསྟན་པའོ། །ལྔ་པ་སྡུག་བསྔལ་བསམ་པ་ལ་བཞི། རྩ་བའི་སྡུག་བསྔལ་ རྩ་བའི་སྡུག་བསྔལ་བཞི་བསྟན་པ་དང༌། ཡན་ལག་སྡུག་བསྔལ་ ཡན་ལག་གི་སྡུག་བསྔལ་དྲུག་བསྟན་པ་དང༌། ཉིང་ལག་གི་སྡུག་བསྔལ་ ཉིང་ལག་གི་སྡུག་བསྔལ་གྲངས་ལས་འདས་པར་བསྟན་པ། སྡུག་བསྔལ་དེ་དག་སེམས་སུ་ངོ་སྤྲད་པའོ། །དྲུག་པ་རྒྱུ་འབྲས་དཔྱད་པ་ལ་བཞི། དགེ་སྡིག་ རྒྱུ་ངོས་བཟུང་བ་དང༌། འབྲས་བུ་བདེ་སྡུག་ འབྲས་བུ་སྐྱེ་ཚུལ་བསྟན་པ་དང༌། དགེ་སྡིག་སོ་མཚམས་དབྱེ་བ་དང༌། སྡིག་པ་སྟོབས་བཞི་ སྡིག་པ་བཤགས་པའོ། ༈ །སྤྱི་དོན་གཉིས་པ་ཐུན་མོང་སྤྱིའི་གདམས་ངག་ལ་ལྔ་སྟེ། སྐྱབས་འགྲོ་བྱ་བ་དང་། སེམས་བསྐྱེད་པ་དང་། སྡོམ་པ་བཟུང་བ་དང་། བྱང་སེམས་བསྒོམ་པ་དང་། བསླབ་པ་གསུམ་ལ་བློ་སྦྱོང་བའོ། །དང་པོ་ནི། སངས་རྒྱས་ཆོས་དང་ སྐྱབས་འགྲོ་བྱ་བ་དང༌། གཉིས་པ་ནི། སངས་རྒྱས་ཐོབ་འདོད་ སེམས་བསྐྱེད་པ་དང་། གསུམ་པ་སྡོམ་པ་བཟུང་བ་ལ་བཞི། སྡིག་པ་རྣམ་གསུམ་ སོ་ཐར་སྡོམ་པ་དང༌། དགེ་བ་རྣམ་གསུམ་ བྱང་སེམས་སྡོམ་པ་དང༌། གནས་ལུགས་རང་ངོ་གསང་སྔགས་ཀྱི་སྡོམ་པ་དང༌༑ སྡོམ་གསུམ་སངས་རྒྱས་ སྡོམ་གསུམ་སངས་རྒྱས་ཀྱི་བསྟན་པར་བསྟན་པའོ། །བཞི་པ་བྱང་སེམས་བསྒོམ་པ་ལ་བཞི། བྱམས་པའི་སེམས་ བྱམས་པ་དང༌། སྙིང་རྗེའི་སེམས་ སྙིང་རྗེ་དང༌། བྱང་ཆུབ་ཀྱི་སེམས་ བྱང་ཆུབ་ཀྱི་སེམས་བསྒོམ་པ། དེ་གསུམ་སངས་རྒྱས་ཀྱི་ཐུགས་དགོངས་སུ་བསྟན་པའོ། །ལྔ་པ་བསླབ་པ་སྦྱངས་པ་ལ་བཞི། གནོད་སྦྱོར་ལོག་པ་ ལྷག་པ་ཚུལ་ཁྲིམས་ཀྱི་བསླབ་པ་དང༌། ཕན་འདོགས་ཞུགས་པ་ ལྷག་པ་ཏིང་ངེ་འཛིན་གྱི་བསླབ་པ་དང༌། ཐབས་ཤེས་ཟུང་འཇུག་ ལྷག་པ་ཤེས་རབ་ཀྱི་བསླབ་པ་དང༌། བསླབ་པ་སྡེ་སྣོད་གསུམ་ དེ་གསུམ་སྡེ་སྣོད་གསུམ་དུ་བསྟན་པའོ། ༈ །སྤྱི་དོན་གསུམ་པ་ཁྱད་པར་རང་གི་མན་ངག་ལ་གསུམ་སྟེ། གཞི་དང༌། ལམ་དང༌། འབྲས་བུའོ། །དང་པོ་ལ་གཉིས་ཏེ། གནས་ལུགས་དང༌། འཁྲུལ་ལུགས་སོ། །གནས་ལུགས་ལ་བཞི་སྟེ། རང་བྱུང་ངོ་བོ་ཉིད་ རང་བྱུང་ངོ་བོ་ཉིད། རང་བཞིན་སྐྱེ་མེད་ རང་བཞིན་སྐྱེ་མེད། རོལ་པ་འགགས་མེད་ རོལ་པ་འགགས་མེད་དང༌། ངོ་བོ་གཉིས་མེད་ ངོ་བོ་མཐའ་བྲལ་དུ་བསྟན་པའོ། །འཁྲུལ་ལུགས་ལ། ངོ་མ་ཤེས་ རྒྱུ་དང༌། ཉོན་མོངས་ རྐྱེན་དང༌། རིགས་དྲུག་ལུས་ འབྲས་བུ་དང༌། མ་རིག་ནས་རྩ་བ་ འཁོར་ཚུལ་ལོ། །གཉིས་པ་ལམ་གྱི་མན་ངག་ལ་ལྔ་སྟེ། མཐའ་ལ་མི་གནས་པའི་ལྟ་བ་དང༌། རིག་པ་རང་ལུགས་ཀྱི་བསྒོམ་པ་དང༌། འཁྲུལ་པ་རང་གྲོལ་གྱི་སྤྱོད་པ་དང་། འགྱུར་མེད་གནད་དུ་བཏང་བ་དང༌། ལ་ཟླའི་མན་ངག་གོ །དང་པོ་ལྟ་བ་ལ་བཞི་སྟེ། ཡིན་ཚུལ་ཅིར་ཡང་ གནས་ལུགས་དང༌། སྣང་ཚུལ་ཅིར་ཡང་ འཆར་ལུགས་དང༌། སྣང་ཚད་སེམས་ཉིད་ ཡིན་ལུགས་དང༌། རང་གྲོལ་རྟོགས་པ་ གྲོལ་ལུགས་སོ། །གཉིས་པ་བསྒོམ་པ་ལ་བཞི་སྟེ། སོ་མ་རང་བབས་ སེམས་ཀྱི་བཞག་ཐབས་དང༌། རྟོག་མེད་རང་གསལ་ ཉམས་ཀྱི་སྐྱེ་ཚུལ་དང༌། སྣང་སེམས་དབྱེར་མེད་ སྣང་བ་བསྒོམ་དུ་ངོ་སྤྲད་པ་དང༌། གསལ་སྟོང་དབྱེར་མེད་ རྟོག་པ་བསྒོམ་དུ་ལ་བཟླ་བའོ། །གསུམ་པ་སྤྱོད་པ་ལ་བཞི་སྟེ། རྐྱེན་སྣང་ ཉམས་སྣང་ངོས་བཟུང་བ། རྟོག་པ་ལ་སེམས་ དེ་སེམས་སུ་བསྟན་པ་དང༌། ཞུམ་དང་ཁེངས་པ་ རོ་སྙོམས་པ་དང༌། ཞེན་མེད་རང་གྲོལ་ ཞེན་མེད་བསྐྱང་བའོ། །བཞི་པ་ལམ་གནད་དུ་བཏང་བ་ལ་བཞི་སྟེ། གང་བྱེད་རང་སོ་ གནས་པས་སོ་གཟུང་བ་དང༌། དེའི་ངོ་བོ་ གསལ་བས་མདངས་འདོན་པ་དང༌། དེ་མ་སྟོང་ སྟོང་པར་རྒྱས་གདབ་པ་དང༌། སྟོང་པ་འགག་པ་ དབྱེར་མེད་དུ་གནད་ལ་དབབ་པའོ། །ལྔ་པ་ལ་བཟླ་བ་ལ་བཞི་སྟེ། མ་གྲུབ་པའི་ གནས་ལུགས་ངོ་སྤྲོད་པ་དང་། རྣམ་དག་ལྷུན་གྲུབ་ དེ་གཞི་རུ་ལ་བཟླ་བ་དང༌། དེའི་ངོ་བོ་ ལམ་དང་འབྲས་བུ་ལ་ལ་བཟླ་བ་དང༌། ཤེས་བྱ་ཀུན་ལྡན་ ཤེས་བྱ་ཀུན་ལྡན་དུ་ལ་བཟླ་བའོ། །གསུམ་པ་འབྲས་བུ་ལ་བཞི་སྟེ། ངེས་པ་གཉིས་ རྟེན་སྐུ་བསྟན་པ་དང༌། ཇི་ལྟ་ཇི་སྙེད་ བརྟེན་པ་ཡེ་ཤེས་པ་དང༌། རྟེན་ཅན་རྟེན་མེད་ འགྲོ་དོན་མཛད་ཚུལ་དང༌། གཅིག་དང་དུ་མ་ དགོངས་པ་གཞུག་ལུགས་སོ། །མཇུག་གི་དོན་ལ་གཉིས་ཏེ། དགེ་བ་འདི་ དགེ་བ་བསྔོ་བ་དང༌། རིན་ཆེན་སྤུངས་པ་ མཚན་རྫོགས་ཚུལ་ལོ། །ཕྱི་བཤད་རིན་ཆེན་སྤུངས་པའི་བསྡུས་དོན་རྫོགས་སོ།། ༈ །།ནང་བཤད་གསང་སྔགས་ལམ་རིམ་གྱི་ལོ་རྒྱུས་བཞུགས་སོ༑ ༑ན་མོ་རཏྣ་གུ་རུ། གསང་སྔགས་ལམ་རིམ་འདི་མཛད་པ་པོ་སློབ་དཔོན་ཆེན་པོ་པདྨ་འབྱུང་གནས་ཡིན་ཏེ། ཨོ་རྒྱན་པདྨ་འབྱུང་གནས་ཀྱིས། ཞེས་པས། དེ་ནི་སྐུ་ཆེ་བའི་ཡོན་ཏན་དཔག་ཏུ་མེད་པ་དང་ལྡན་ཞིང་། མན་ངག་རྩོམ་པའི་ཚད་དུ་ཕྱིན་པའོ། །དང་པོ་ནི། ནུབ་ཕྱོགས་ཨོ་རྒྱན་དུ་སྐུ་མེ་ཏོག་པདྨོ་ལས་བརྫུས་ནས་བྱོན་པས་མངལ་གྱི་སྒྲིབ་གཡོགས་མེད་དེ། སྐྱེ་འཆི་དབང་དུ་འདུས་པས་སྐུ་ལ་འདས་གྲོང་མི་མངའ་འོ། །གསུང་གི་ཆེ་བ་ཆོས་ཀྱི་སྒོ་མོ་སྣ་ཚོགས་སུ་སྟོན་ཅིང་ངག་ལ་ནུས་པ་ཐོབ་པས་དྲེགས་པ་ཅན་ཐམས་ཅད་བཀའ་ཉན་པ་ལ་སོགས་པའོ། །ཐུགས་ཀྱི་ཆེ་བ་ཤེས་བྱའི་དཀྱིལ་འཁོར་མ་ལུས་པར་མཁྱེན་ཅིང་ཟག་པ་མེད་པའི་དགོངས་པ་དང་ལྡན་ཏེ་རྣམ་པར་མི་རྟོག་པའི་ཏིང་ངེ་འཛིན་ལ་བཞུགས་པ་ནི་མཁྱེན་པ་གཉིས་དང་ལྡན་ནོ། །སྐུ་གསུང་ཐུགས་མི་ཟད་པ་རྒྱན་གྱི་འཁོར་ལོ་དང་ལྡན་ཏེ། དགོས་འདོད་འབྱུང་ཞིང་རེ་བ་སྐོང་བས་ཡོན་ཏན་གྱི་ཆེ་བའོ། །ཉོན་མོངས་པ་དང་རྣམ་པར་རྟོག་པ་ཐམས་ཅད་རང་སར་ཞི། ཚེ་དང་ནུས་པ་ཕྱོགས་མེད་དུ་རྒྱས་ཏེ། མི་དང་མི་མ་ཡིན་པ་ཐམས་ཅད་དབང་དུ་འདུས་ཤིང༌། ལོག་ལྟ་གདུག་པ་ཅན་ཐམས་ཅད་ཚར་གཅོད་པ་ནི་ཕྲིན་ལས་ཀྱི་ཆེ་བའོ། །དེ་དག་ནི་ཆེ་བའི་ལོ་རྒྱུས་ཟུར་ཙམ་བརྗོད་ཀྱིས། རྒྱས་པར་གཞན་དུ་བལྟའོ། །གཉིས་པ་མན་ངག་རྩོམ་པར་མཐར་ཕྱིན་པ་ནི། རབ་ཆོས་ཉིད་བདེན་པའི་དོན་མཐོང་བ་སྟེ་དངོས་པོའི་གནས་ལུགས་ཇི་བཞིན་རྟོགས་པའོ། །འབྲིང་ཡི་དམ་ལྷའི་ཞལ་གཟིགས་པ་སྟེ་ལོངས་སྤྱོད་རྫོགས་པ་སྐུའི་ཞིང་ཁམས་མངོན་སུམ་གཟིགས་ཤིང༌། ཡི་དམ་ཐམས་ཅད་ཀྱི་ཞལ་མཐོང༌། ཐ་མ་མཚན་མའི་ལས་ལ་གཉེར་ཐོབ་པ་སྟེ། ཞི་རྒྱས་དབང་དྲག་གི་ལས་ཐམས་ཅད་ཐོགས་པ་མེད་པར་གྲུབ་བོ། །དེ་ལྟ་བུའི་བླ་མ་སྤྲུལ་སྐུ་པདྨ་འབྱུང་གནས་ཀྱིས་མཛད་པའོ། ། ཨྠིྀ། གདམས་ངག་རིན་པོ་ཆེ་གསལ་བའི་སྒྲོན་མ་འདི། དུས་གསུམ་སངས་རྒྱས་ཀུན་གྱི་གཤེགས་ཤུལ། སྡེ་སྣོད་ཀུན་གྱི་ལྡེ་མིག ༑རྒྱུད་ལུང་ཀུན་གྱི་སྙིང་པོ། མན་ངག་གནད་ཀྱི་ཉིང་ཁུ། ངེས་དོན་གནད་ཀྱི་ཊཱི་ཀ །སངས་རྒྱས་ཀུན་གྱི་ཐུགས་དགོངས། རྡོ་རྗེ་སེམས་དཔའི་གསང་ལམ། མུན་པ་སེལ་བའི་སྒྲོན་མེ། འཛུར་ས་མེད་པའི་འཕྲང་དགུགས། སངས་རྒྱས་ལག་ཏུ་གཏོད་པ། འཁོར་བ་དོང་ནས་སྤྲུགས་པའི་གསང་སྔགས་ལམ་གྱི་རིམ་པ་རིན་པོ་ཆེ་གསལ་བའི་སྒྲོན་མེ་ཞེས་བགྱི་བ་སྟོན་པར་བགྱིད་དོ། ༈ །གསང་སྔགས་ལམ་གྱི་རིམ་པ་རིན་པོ་ཆེ་གསལ་བའི་སྒྲོན་མེ་ཞེས་བྱ་བ་བཞུགས་སོ། །རྒྱ་གར་སྐད་དུ། མནྟྲ་སྱ་པ་ཐ་ཀྲ་མ་ནཱ་མ། བོད་སྐད་དུ། གསང་སྔགས་ལམ་གྱི་རིམ་པ་ཞེས་བྱ་བ། རྩ་བ་གསུམ་ལ་གུས་པས་ཕྱག་འཚལ་ལོ། །གཟུང་འཛིན་ གཉིས་བཅོམ་ མཁྱེན་པ་ གཉིས་ལྡན་ རྟག་ཆད་ གཉིས་འདས་ གཞི་དང་འབྲས་བུའི་ གཉིས་དཔལ་མཆོག །སངས་རྒྱས་ཀྱི་དགོངས་པ་ མ་ལུས་འདུས་ཤིང་ སྤྱོད་ཡུལ་ ཡང་དག་ འཁྲུལ་པའི་ ཟོལ་མེད་པ། །བཅོམ་ལྡན་བདེ་གཤེགས་དྲུག་པ་ཀུན་བཟང་ལ། །གུས་གདུང་དད་པའི་སྒོ་ནས་ཕྱག་འཚལ་ལོ། ༈ །བདག་ཉིད་རྣལ་འབྱོར་ཨོ་རྒྱན་པད་འབྱུང་གིས། །རྡོ་རྗེ་ཐེག་པ་ཚེ་གཅིག་ཐོབ་བྱེད་ལམ། ཕྱི་རབས་རྣལ་འབྱོར་ལས་ལྡན་དོན་ཆེད་དུ། ཉམས་ལེན་ལམ་རིམ་རིན་ཆེན་སྒྲོན་མེ་བྲི། ༈ །གནས་ལུགས་འཁྲུལ་ལུགས་ངོ་ཡིས་བཟུང་བ་ནི༔ གཉིས་མེད་ལྷུན་གྲུབ་ཀ་དག་ཡེ་སངས་རྒྱས༔ སྙིང་པོ་མ་རིག་ལས་ཉོན་ལོག་སྡུག་བསྔལ༔ ༈ ཕུན་སུམ་ཚོགས་ལྔས་བསྒྲུབ་གཞི་བཟུང་བ་ནི༔ བྲག་ནགས་དུར་ཁྲོད་ཕུག་སོགས་སྐྱོན་ཡོན་བརྩི༔ དད་གུས་དབང་དམ་ཤེས་རབ་གདིང་དང་ལྡན༔ ཚེས་སྐར་གཟའ་བཟང་ཡར་ངོའི་སྔ་གོང་འཇུག༔ ཟས་སྨན་མཆོད་སྐོང་སྒྲུབ་པའི་རྫས་རྣམས་གསོས༔ མ་དྲེད་རྩེ་གཅིག་དཀའ་ཐུབ་མཐར་འདོན་ནོ༔ ༈ རྣལ་འབྱོར་སྲུང་བའི་འཁོར་ལོའི་རིམ་པ་ནི༔ ལུས་ཅན་རྣམས་ཀྱིས་བྱང་ཆུབ་མཆོག་ཐོབ་བསྒོམ༔ ལྷ་མཚན་རྡོ་རྗེ་འཁོར་ལོའི་མཚན་མས་བསྲུང༔ ལུས་ངག་སེམས་གསུམ་སྐུ་གསུང་ཐུགས་ཡིག་བསམ༔ སྤྱན་རས་འཇམ་དཔལ་གང་ཡིན་ཡི་དམ་བསྒོམ༔ རྡོ་རྗེ་ས་ར་གུར་དང་མདའ་མེ་བསྒོམ༔ བྱ་བྱེད་ངོ་བོ་མི་དམིགས་ངང་བཞག་གོ༔ ༈ གནད་ལྔའི་སྒོ་ནས་ལམ་ལ་འཇུག་པ་ནི༔ ཐུན་མོང་ཁྱད་པར་བླ་མེད་གསུམ་སྐྱབས་འགྲོ༔ དཀར་དམར་ནག་གསུམ་བརྒྱ་བཟླས་མཆོད་དང་བསྐང༔ དལ་འབྱོར་ལས་འབྲས་མི་རྟག་ཉེས་དམིགས་བསྒོམ༔ སྒྲུབ་དང་མཆོད་པའི་མཎྜལ་གཉིས་སུ་འབུལ༔ སྣ་ཚོགས་ལམ་ཁྱེར་གཏོར་མས་ཚོགས་གཉིས་བསགས༔ རྒྱུན་གཏོར་གསུམ་བདུན་བཅུ་གཅིག་ལ་སོགས་བཏང༔ རྣམ་དག་བདུན་དང་ཕྱི་ནང་གསང་མཆོད་དབུལ༔ དེ་རྣམས་གསང་སྔགས་ལམ་གྱི་སྔོན་འགྲོ་ཡིན༔ ༈ བསླབ་པ་གསུམ་ལ་ལམ་གཞི་བཅའ་བ་ནི༔ མི་མཐུན་སྤོང་སེམས་མཉམ་འཇོག་རང་ངོ་ཤེས༔ ཤཱི་ལ་ས་མཱ་ཏི་པྲཛྙ་ཤིཀྵ་ཎ༔ སྣོད་སྦྱོར་ལོག་ཞུགས་ཁྱད་པར་ཐབས་ཀྱིས་ཟིན༔ བདེན་པའི་དོན་མཐོང་གོམས་འདྲིས་དེ་མི་འདའ༔ གཉེན་པོའི་དབང་གིས་བརྒྱད་ཁྲི་བཞི་སྟོང་གསུངས༔ སེམས་གསུམ་སྒོ་ནས་ལམ་ལ་གཟུད་པ་ནི༔ ཡིད་འོང་ཕ་མར་ཤེས་ཤིང་བདེ་ལྡན་བསྒོམ༔ སེམས་ཅན་ཕ་མ་སྡུག་བསྔལ་བྲལ་བར་བསམ༔ སངས་རྒྱས་ཐོབ་འདོད་བསམ་པ་ལེན་དང་སྒྲུབ༔ སེམས་གསུམ་སྡེ་གསུམ་བསླབ་གསུམ་ཉམས་གསུམ་ཐུགས༔ ༈ ཁྱད་པར་གསང་སྔགས་ལམ་གྱི་ངོ་བོ་ནི༔ བུམ་གསང་ཡེ་ཤེས་བདེ་ཆེན་དབང་བཞི་བསྐུར༔ ༈ ཕྱི་ནང་སྣོད་བཅུད་གཞལ་ཡས་ལྷ་རུ་བསྐྱེད༔ སྐྱེ་འགགས་གཉིས་མེད་རང་བྱུང་མཐའ་ལས་འདས༔ ༈ གནད་གསུམ་གཅུན་པས་རྟོག་མེད་རང་གསལ་འཆར༔ ༈ སོ་སྤྱི་བོགས་གསང་ཡང་དག་ཁོ་ན་སྤྱོད༔ ༈ མཉམ་རྗེས་བཟའ་བཅའ་བསྲུང་བའི་དམ་ལྔ་དབོག༔ དེ་རྣམས་གསང་སྔགས་ལམ་གྱི་ངོ་བོ་ཡིན༔ ༈ རྣལ་འབྱོར་ལྔ་ཡིས་ཉམས་སུ་བླང་བ་ནི༔ རྟོག་མེད་ཆོས་སྐུར་ཉལ་ལ་འཆི་རྟོག་སྦྱང༔ འགག་མེད་ལོངས་སྐུར་ལངས་ཏེ་སྐྱེ་རྟོག་སྦྱང༔ དབང་བཞིས་བདུད་རྩིར་བརླབས་ལ་བཟའ་བཏུང་རོལ༔ སྣ་ཚོགས་ཐབས་ཀྱིས་བཟུང་ལ་ལམ་དུ་བཏང༔ དྲོད་སྲོད་སྐྱེས་ཤར་འདའ་ཀ་བར་དོ་བསྟིམ༔ ༈ ལམ་དེའི་གེགས་བཞིའི་སྐྱོན་བསལ་བསྟན་པ་ནི༔ བྱིང་རྒོད་ཉམས་གསུམ་བཅོས་ལ་ལམ་དུ་བཏང༔ ནད་ཀྱི་གཞི་རྒྱུ་རྐྱེན་འབྲས་ངོས་བཟུང་བཅོས༔ སྤྲོས་བྲལ་ངང་བཞག་རྒྱུན་ཆད་སྐྱེ་དང་སེལ༔ གནོད་པའི་ཟས་བཟའ་འཁྲིག་བཅད་ལྟོས་མེད་བསྐྱུར༔ དཀར་དམར་ནག་པོའི་ཧཱུྃ་འཕྲོས་སྟེང་འོག་གདོན༔ རང་ལས་གཞན་མེད་ཡ་ང་ཁུ་འཕྲིག་བཅད༔ དམ་ཉམས་གྲོགས་གནས་ཟས་ཀྱི་གྲིབ་བཞི་བསལ༔ ༈ ལམ་དེའི་བོགས་འདོན་གནད་དུ་གཏོང་བ་ནི༔ སྐྱོན་གྱི་རྩ་བ་བདག་འཛིན་བློ་ཡིས་བཏང༔ ཡོན་ཏན་གཞི་མ་ཉམས་ལེན་དྲག་པོ་བྱ༔ རང་གི་ལམ་ལ་བོགས་མེད་གཞན་གྱིས་འདོན༔ ཉོན་མོངས་གནད་ཀྱིས་གཟུང་ལ་ལམ་དུ་བཏང༔ རྣམ་རྟོག་ལམ་དུ་བཏང་ལ་བོགས་སུ་འདོན༔ དེ་རྣམས་ལམ་གྱི་ཆ་རྐྱེན་ཡན་ལག་ཡིན༔ ༈ ལམ་དེ་མཐར་ཕྱིན་མངོན་གྱུར་འབྲས་བུ་ནི༔ སྐུ་ལྔ་གསུང་ལྔ་ཐུགས་ཡོན་ཕྲིན་ལས་ལྔ༔ རྟེན་ཅན་རྟེན་མེད་ངོ་བོ་ཉིད་དོན་བྱེད༔ གཅིག་དང་ཐ་དད་གཉིས་སུ་མེད་པར་བཞུགས༔ གཟིགས་ཚུལ་གྲུབ་དང་མ་གྲུབ་བདེན་བརྫུན་བརྟག༔ ༈ ཚེ་འདིར་སངས་རྒྱས་ཐོབ་བྱེད་ཉེ་བའི་ལམ༔ རང་མཚན་མངོན་དུ་སྟོན་པ་ངེས་པའི་ཆོས༔ ཐབས་དང་ཤེས་རབ་གཉིས་ལྡན་ཤིང་རྟའི་སྲོལ༔ ཕམ་བྱེད་རྒྱལ་འགྱུར་བླ་མེད་ཐེག་མཆོག་གོ༔ ༈ འདི་སྦྱར་དགེ་བ་དྲི་མེད་རྣམ་དག་གིས༔ ལུས་ཅན་ཐམས་ཅད་སྨིན་གྲོལ་ལམ་ཞུགས་ཏེ༔ རང་ཆས་ཤེས་རྟོག་སྒྲིབ་དག་ཉེས་ཟད་ནས༔ ཡང་དག་རྫོགས་པའི་སངས་རྒྱས་མྱུར་ཐོབ་ཤོག༔ གསང་སྔགས་ལམ་གྱི་རིམ་པ་རིན་པོ་ཆེ་གསལ་བའི་སྒྲོན་མེ་ཞེས་བྱ་བ༔ ཨོ་རྒྱན་པདྨ་འབྱུང་གནས་ཀྱིས་མཛད་པ་རྫོགས་སོ༔ ༈ ནང་བཤད་གསང་སྔགས་ལམ་རིམ་ས་བཅད་བཞུགས་སོ། །དཔལ་ཀུན་ཏུ་བཟང་པོ་ལ་ཕྱག་འཚལ་ལོ། །ནང་བཤད་གསང་སྔགས་ལམ་གྱི་རིམ་པ་འདི་ལ་དོན་གསུམ་སྟེ། ཀླད་ཀྱི་དོན་དང༌། གཞུང་གི་དོན་དང༌། མཇུག་གི་དོན་ནོ། །དང་པོ་ལ་གཉིས་ཏེ། གཉིས་བཅོམ་གཉིས་ལྡན་ ཕྱག་འཚལ་བ་དང༌། བདག་ཉིད་རྣལ་འབྱོར་ རྩོམ་པར་དམ་བཅའ་བའོ། །གཞུང་དོན་ལ་གསུམ་སྟེ། གཞི། ལམ། འབྲས་བུའོ། །གཞུང་གི་སྤྱི་དོན་དང་པོ་ལ་གཉིས་ཏེ། གནས་ལུགས་ གནས་ལུགས་དང་། སྙིང་པོ་མ་རིག་ འཁྲུལ་ལུགས་སོ། །གཞུང་གི་སྤྱི་དོན་གཉིས་པ་ལ་གསུམ་སྟེ༑ ལམ་གྱི་སྔོན་དུ་འགྲོ་བ་དང༌། ལམ་རང་གི་ངོ་བོ་དང༌། ལམ་གྱི་ཆ་རྐྱེན་བསྟན་པའོ། །དང་པོ་ལ་གསུམ་སྟེ། ཕུན་སུམ་ཚོགས་པ་ལྔས་སྒྲུབ་པའི་གཞི་བཟུང་བ་དང༌། བསྲུང་བའི་འཁོར་ལོའི་རིམ་པ་བསྟན་པ་དང༌། གནད་ལྔའི་སྒོ་ནས་ལམ་ལ་འཇུག་པར་བསྟན་པའོ། །དང་པོ་ལ་ལྔ་སྟེ། ཕུན་སུམ་ཚོགས་ལྔས་ གནས་གང་དུ་དང༌། དད་གུས་ གང་ཟག་གང་གིས་དང༌། ཚེས་སྐར་ དུས་གང་ལ་དང༌། ཟས་སྨན་ རྫས་གང་བསགས་པ་དང༌། མ་དྲེད་རྩེ་གཅིག་ ཐབས་ཇི་ལྟར་བསྒྲུབ་པའོ༑ ༑གཉིས་པ་སྲུང་འཁོར་ལ་དྲུག་སྟེ། རྣལ་འབྱོར་སྲུང་བའི་ བྱང་ཆུབ་སེམས་ཀྱི་སྲུང་བ་དང༌། ལྷ་མཚན་རྡོ་རྗེ་ མཚོན་ཆ་ལྔའི་སྲུང་བ་དང༌། ལུས་ངག་ཡིད་ ཡི་གེ་གསུམ་གྱི་སྲུང་བ་དང༌། སྤྱན་རས་འཇམ་དཔལ་ ཞི་དྲག་གསུམ་བརྩེགས་ཀྱི་སྲུང་བ་དང༌། རྡོ་རྗེ་ར་བ་ རང་བྱུང་ལྷུན་གྱིས་གྲུབ་པའི་སྲུང་བ་དང༌། བྱ་བྱེད་ དོན་དམ་མཚན་མ་མེད་པའི་སྲུང་བའོ། །གསུམ་པ་གནད་ལྔས་ཐེག་ཆེན་ལམ་དུ་འཇུག་པ་ལ་ལྔ་སྟེ། གནས་ལྔའི་ སྐྱབས་འགྲོས་ཆོས་ཀྱི་གཞི་གཏིང་བ་དང༌། དཀར་དམར་ནག་ ཡིག་བརྒྱས་སྒྲིབ་པ་སྦྱང་བ་དང༌། དལ་འབྱོར་ལས་འབྲས་ ལས་འབྲས་མི་རྟག་པ་སྒོམ་པ་ལ་བློ་སྦྱངས་པ་དང༌། སྒྲུབ་དང་མཆོད་ མཎྜལ་དང་ སྒྲུབ་དང་མཆོད་ གཏོར་མ་བཏང་བས་ཚོགས་བསགས་པ་དང༌། རྣམ་དག་བདུན་ མཆོད་པ་ཕུལ་ལ་ཚོགས་ཡོངས་སུ་སྤེལ་བའོ། །གཉིས་པ་ལམ་རང་གི་ངོ་བོ་ལ་གཉིས་ཏེ། ཐུན་མོང་སྤྱི་ཡི་ལམ་དང༌། ཁྱད་པར་གསང་སྔགས་ཀྱི་ལམ་མོ། །དང་པོ་ལ་གཉིས་ཏེ། བསླབ་པ་གསུམ་ལ་ལམ་གྱི་གཞི་བཅའ་བ་དང་། སེམས་གསུམ་གྱིས་ལམ་གྱི་ཁ་བསྒྱུར་བའོ། །དང་པོ་ལ་ལྔ་སྟེ། བསླབ་པ་གསུམ་ ངོ་བོ་ངོས་བཟུང་བ་དང༌། ཤཱི་ལ་ས་མཱ་ མཚོན་བྱའི་སྒྲ་དོན་བཤད་པ་དང༌། གནོད་སྦྱོར་ལོག་ མཚན་ཉིད་ཀྱི་རང་ལྡོག་བསྟན་པ་དང༌། བདེན་པའི་དོན་མཐོང་ མཚན་གཞི་སྐྱེ་ཚུལ་དང༌། གཉེན་པོའི་དབང་གིས་ གཉེན་པོའི་སྒོ་ནས་རྣམ་གྲངས་སུ་འགྱུར་ཚུལ་ལོ། །གཉིས་པ་སེམས་གསུམ་གྱིས་ལམ་གྱི་ཁ་བསྒྱུར་བ་ལ་བཞི་སྟེ། སེམས་གསུམ་སྒོ་ནས་ བྱམས་པ་དང༌། སེམས་ཅན་ སྙིང་རྗེ་དང༌། སངས་རྒྱས་ཐོབ་འདོད་ བྱང་ཆུབ་ཀྱི་སེམས་དང༌། སེམས་གསུམ་སྡེ་གསུམ་ དེ་དག་སྡེ་སྣོད་གསུམ་དུ་ངོས་བཟུང་ལ་བསླབ་པ་གསུམ་དུ་བསྟན་པའོ། །གཉིས་པ་གསང་སྔགས་རང་གི་ལམ་ལ་ལྔ་སྟེ། ཁྱད་པར་གསང་སྔགས་ རྒྱུད་སྨིན་པར་བྱེད་པ་དབང་དང༌༑ ཕྱི་ནང་ སྨིན་པ་གྲོལ་བར་བྱེད་པ་ལ་ལྟ་བ་དང༌། གནད་གསུམ་གཅུན་པས་ ལྟ་བ་གོམས་པར་བྱེད་པ་ལ་བསྒོམ་པ་དང༌། སོ་སྤྱིའི་བོགས་ དེ་དང་མི་འབྲལ་ཞིང་རྩལ་སྦྱོང་བ་ལ་སྤྱོད་པ་དང༌། མཉམ་རྗེས་བཟའ་བཅའ་ དེ་དག་ལས་མི་འདའ་བར་བྱེད་པ་ལ་དམ་ཚིག་གོ །གཉིས་པ་ལྟ་བ་ལ་གཉིས་ཏེ། ཕྱི་ནང་སྣོད་བཅུད་ བསྐྱེད་རིམ་སྦྱང་བ་དང༌། སྐྱེ་འགག་གཉིས་མེད་ རྫོགས་རིམ་བསྟན་པའོ། །གསུམ་པ་དེའི་ཆ་རྐྱེན་བསྟན་པ་ལ་གསུམ་སྟེ། རྣལ་འབྱོར་ལྔའི་ལམ་ཉམས་ལེན་གྱི་གནད་བསྟན་པ་དང༌། གེགས་སེལ་བཞིས་ལམ་གྱི་རྐྱེན་བསལ་བ་དང༌། བོགས་འདོན་བཞིས་ལམ་མཐར་འདོན་པའོ། །དང་པོ་ལ་ལྔ་སྟེ། རྟོག་མེད་ ཉལ་བའི་རྣལ་འབྱོར་དང༌། འགག་མེད་ ལྡང་བའི་རྣལ་འབྱོར་དང༌། དབང་བཞིས་ ཟ་བའི་རྣལ་འབྱོར་དང༌། སྣ་ཚོགས་ རྒྱུན་གྱི་རྣལ་འབྱོར་དང༌། དྲོད་སྲོད་ དུས་གནད་ཀྱི་རྣལ་འབྱོར་རོ། །གཉིས་པ་ལམ་གྱི་གེགས་བསལ་བ་ལ་བཞི་སྟེ། བྱིང་རྒོད་ ཏིང་ངེ་འཛིན་ཉམས་ཀྱི་གེགས་བསལ་བ་དང༌། ནད་ཀྱི་ ལུས་ན་ཚ་སྡུག་བསྔལ་གྱི་གེགས་བསལ་བ་དང༌། རང་ལས་གཞན་མེད་ སེམས་རྣམ་རྟོག་ལྷ་འདྲེའི་གེགས་བསལ་བ་དང༌། དམ་ཉམས་ རྐྱེན་སྣང་གློ་བུར་གྲིབ་ཀྱི་གེགས་བསལ་བའོ། །གསུམ་པ་ལམ་གྱི་བོགས་འདོན་པ་ལ་གསུམ་སྟེ། ལམ་དེའི་ བོགས་མི་འབྱུང་བའི་སྐྱོན་བསལ་བ་དང༌། ཡོན་ཏན་ འབྱུང་བའི་གཞི་བཅའ་བ་དང༌། བོགས་ཇི་ལྟར་འདོན་པའི་ཐབས་སོ། །དེ་ལ་གཉིས་ཏེ། རང་ལ་གནས་པ་དེ་ལ་བོགས་འདོན་པ་དང༌། སྔར་མེད་གསར་དུ་སྐྱེས་པའི་ཉོན་མོངས་པ་རྣམ་རྟོག་བོགས་སུ་འདོན་པའོ། །དེ་ལ་གཉིས་ཏེ། ཉོན་མོངས་ ཉོན་མོངས་བོགས་སུ་འདོན་པ་དང༌། རྣམ་རྟོག་ རྣམ་རྟོག་བོགས་སུ་འདོན་པའོ། །གཞུང་གི་སྤྱི་དོན་གསུམ་པ་ཉམས་སུ་བླངས་ནས་མཐར་ཕྱིན་པའི་འབྲས་བུ་ལ་བཞི་སྟེ། སྐུ་ལྔ་ འབྲས་ཆོས་ངོས་བཟུང་བ་དང༌། རྟེན་ཅན་རྟེན་མེད་ འགྲོ་དོན་བྱེད་ཚུལ་དང༌། གཅིག་དང་ཐ་དད་ དགོངས་པ་ལ་བཞུགས་ལུགས་དང༌། གཟིགས་ཚུལ་ གདུལ་བྱ་ལ་ཐུགས་རྗེས་འཛིན་ཚུལ་ལ་གཞན་གྱི་ལོག་རྟོག་དགག་པའོ༑ ༑སྤྱི་དོན་གསུམ་པ་མཇུག་གི་དོན་ལ་གསུམ་སྟེ། ཚེ་འདིར་ ཆོས་ཀྱི་ཆེ་བ་བསྟན་པ་དང༌། འདི་སྦྱར་དགེ་བ་ དགེ་བ་ཐམས་ཅད་མཁྱེན་པར་བསྔོ་ཞིང༌། གསང་སྔགས་ལམ་གྱི་རིམ་ མཚན་རྫོགས་ཚུལ་དང་བཅས་པ་བསྟན་པའོ། །ནང་བཤད་གསང་སྔགས་ལམ་རིམ་གྱི་བསྡུས་དོན་རྫོགས་སོ། ༈ །གསང་སྔགས་ལམ་རིམ་གྱི་སྙིང་པོ་སེམས་ཉིད་གཅེར་སྟོན་ལག་ཁྲིད་ལྔ་པ་བཞུགས་སོ། །བླ་མ་དམ་པ་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །གུ་རུ་པདྨ་འབྱུང་གནས་ཀྱིས་མཛད་པའི་མན་ངག །སེམས་ཉིད་གཅེར་སྟོན་གྱི་ལག་ཁྲིད་འདི་ལ་ལྔ་སྟེ། དང་པོ་རང་སེམས་ལྟ་རུ་གཞུག་པ། གཉིས་པ་མ་མཐོང་ན་སེམས་ངོ་བསྟན་པ། གསུམ་པ་མཐོང་བའི་དོན་ངོ་སྤྲོད་པ། བཞི་པ་དེའི་རྒྱབ་སྐྱོར་བ། ལྔ་པ་ལ་བཟླ་བའོ། །དང་པོ་ནི། རང་གི་སེམས་གང་ཡིན་ལྟོས་ལ། ལན་གཅིག་ཁྱེར་ལ་ཤོག །བྱས་པས། སེམས་སྟོང་པ་ཉིད་ལྟ་རྒྱུ་མེད་པ་གཅིག་འདུག་ཟེར་ན། མ་མཐོང་བ་ཡིན། སྣང་བ་ཐམས་ཅད་སེམས་སུ་འདུག་ཟེར་ནའང་། སེམས་ཀྱི་སྤྱི་རིག་ཙམ་མཐོང་བ་ཡིན་གྱིས། ད་ལྟ་བའི་རང་སེམས་ངོས་མ་ཟིན་པ་ཡིན། ནམ་མཁའ་དང་ས་གཞི་རི་བོ་ལ་སོགས་པ་གང་ནས་བཙལ་ཡང་མི་རྙེད་དོ། །གཉིས་པ་སེམས་ངོ་བསྟན་པ་ནི། སེམས་ཀྱི་སྤྱི་རིག་ཙམ་གྱིས་མཐོང་བའི་གོ་མི་ཆོད་དོ། །ད་ལྟ་གང་ཤར་དེ་ངོས་ཟིན་དགོས་པ་ཡིན། ད་ཐབས་སྒྲ་དེ་སེམས་སུ་ཤར་བ་ཡིན་ཏེ། ངས་སེམས་གང་ཡིན་ལྟོས་ཤིག་བྱས་པ་ཁྱོད་ཀྱི་སེམས་སུ་ཤར་བ་ཡིན་ཏེ། ཁྱོད་རང་གང་ཡིན་ལྟ་བ་པོ་དེ་རང་སེམས་ཡིན་ནོ། །བྱས་ལ། མ་དང་བུའི་དཔེས་ངོ་སྤྲད་དོ། །དེ་ཡང་སེམས་ལ་ལྟོས་ཤིག་ཅེས་པའི་སྒྲ་དེས་སེམས་ལ་ཚུར་བལྟས་སམ། སེམས་ཀྱིས་སྒྲ་ཕར་བཟུང་ངམ༑ སྒྲ་སེམས་དབྱེར་མེད་དུ་འདུག་ལྟོས་ཤིག །བྱས་ལ་དྲི། སྔ་མ་གཉིས་པོ་ལྟར་ཟེར་ན། མ་གོ་བ་ཡིན། དབྱེར་མེད་དུ་འདུག་ཟེར་ན། གོ་བ་ཡིན་པས་ངོ་སྤྲད་པར་བྱའོ། །དེ་བཞིན་དུ་གཟུགས་ལ་སོགས་པ་ཡུལ་ལྔ་ཀ་སེམས་སུ་ངོ་སྤྲད་དོ། །གསུམ་པ་མཐོང་བའི་དོན་ངོ་སྤྲོད་པ་ལ་གསུམ་སྟེ། རྣམ་པ་རང་གྲོལ། གཤིས་ཡེ་གྲོལ། ངོ་བོ་གཉིས་མེད་དུ་ངོ་སྤྲད་པའོ། །དང་པོ། གང་ཡིན་ལྟ་བ་པོའི་སེམས་དེ་རང་དུ་འདུག་གམ། གཞན་དུ་འདུག་དྲི། གཞན་དུ་འདུག་ཟེར་ན། མ་གོ་བ་ཡིན། རང་དུ་འདུག་ཟེར་ན། ངོ་མཐོང་བ་ཡིན་པས་ངོ་སྤྲད་དོ། །གཞན་མ་ཡིན་པར་ཐག་བཅད་ལ། རང་སྣང་ལ་གཞུག་པས་རང་གྲོལ་དུ་ཤེས་འགྲོ་བ་ཡིན་ནོ། །རང་དུ་ཤེས་པས་གཞན་འཛིན་གྱི་རྟོག་པ་མི་སྐྱེ་བ་ནི་རང་གྲོལ་གྱི་གནད་ཡིན་ནོ། །གཉིས་པ། ཡེ་གྲོལ་ངོ་སྤྲད་པ་ནི། ལྟ་བ་ལྟར་སྣང་བའི་སེམས་དེ་སྣང་བ་རང་རྒྱུད་དུ་བདེན་པ་གཅིག་འདུག་གམ། ངོ་བོ་སྟོང་པར་འདུག་དྲི། རང་རྒྱུད་དུ་འདུག་ཟེར་ན། མ་གོ་བ་ཡིན། སྟོང་པར་འདུག་ཟེར་ན། ཚུར་བཀུག་ནས་སྟོང་པར་འདུག་གམ། ཡལ་བའི་རྗེས་ལ་སྟོང་པར་འདུག་གམ། དེ་ཀ་རང་སྟོང་པར་འདུག་དྲི། སྔ་མ་གཉིས་པོ་ལྟར་ཟེར་ན། མ་གོ་བ་ཡིན། དེ་ཀ་རང་སྟོང་པར་འདུག་ཟེར་ན། ངོ་བོ་མཐོང་བ་ཡིན་པས་ངོ་སྤྲད་དོ། །སྣང་བ་རང་རྒྱུད་པ་བདེན་པ་མིན་པར་ཐག་བཅད་ལ་སྟོང་པ་ཉིད་ལ་གཞུག་པས་ཡེ་གྲོལ་ཤེས་འགྲོ་བ་ཡིན་ནོ། །སྟོང་པར་ཤེས་པས་སྣང་བ་ལ་བདེན་འཛིན་མི་སྐྱེ། དེ་ཀ་ཡེ་གྲོལ་གྱི་གནད་ཡིན་ནོ། །གསུམ་པ། གཉིས་མེད་ངོ་སྤྲད་པ་ནི། སྟོང་པ་ཅང་མེད་ཅིག་འདུག་གམ། གསལ་རིག་དང་བཅས་པ་གཅིག་འདུག་གམ་དྲི། ཅང་མེད་གཅིག་འདུག་ཟེར་ན། མ་གོ་བ་ཡིན། གསལ་རིག་ཅན་འདུག་ཟེར་ན། གསལ་བ་དང་སྟོང་པ་ཐ་དད་དུ་འདུག་གམ། གཅིག་ཏུ་འདུག་དྲི། ཐ་དད་དུ་འདུག་ཟེར་ན། མ་གོ་བ་ཡིན། གཅིག་ཏུ་འདུག་ཟེར་ན། ངོ་བོ་མཐོང་བ་ཡིན་པས་ངོ་སྤྲད་དོ། ། སྣང་སྟོང་ཐ་དད་དུ་མེད་པར་གཉིས་མེད་ལ་གཞུག་པས་གཉིས་མེད་དུ་ངོ་ཤེས་འགྲོ་བ་ཡིན་ནོ། །གཉིས་མེད་ལ་གནས་པས་གཉིས་འཛིན་གྱི་རྟོག་པ་མི་སྐྱེ། དེ་ཀ་ལ་གཉིས་མེད་རྟོགས་ཟེར་བ་ཡིན་ནོ། །བཞི་པ་རྒྱབ་སྐྱོར་ལ་གསུམ་སྟེ། རང་གྲོལ་གྱི་རྒྱབ་སྐྱོར་བ། རྣམ་པའི་ཆོས་ཐམས་ཅད་ཀྱི་གཤིས་ལ་ཡེ་ཡིན་ཡེ་གྲོལ་ལ་འདའ་བ་མེད་དེ། བྱ་གར་འཕུར་ཡང་ནམ་མཁའ། མི་གར་ཕྱིན་ས་སྟོང་ལས་མི་འདའ་བ་བཞིན་དུ་དང་། ཡེ་གྲོལ་གྱི་རྒྱབ་སྐྱོར་བ། ཆོས་ཐམས་ཅད་ཀྱི་གཤིས་ལ་ཡེ་ཡིན་ཡེ་གྲོལ་ལས་འདའ་ས་མེད་དེ། གསེར་གླིང་ན་གསེར། ཆུ་གླིང་ན་ཆུ་ལས་འདའ་ས་མེད་པ་ལྟ་བུ་དང༌། གཉིས་མེད་ཀྱི་རྒྱབ་སྐྱོར་བ་ལ། ཆོས་ཐམས་ཅད་ཀྱི་ངོ་བོ་ལ་གཉིས་མེད་ལས་འདའ་ས་མེད་དེ། ཆུ་ལ་གཤེར་བ། མེ་ལ་ཚ་བ་ལས་འདའ་ས་མེད་པ་ལྟ་བུ་ནི། ལྟ་བའི་སྐབས་སུ་ཤེས་པར་བྱའོ། །ལྔ་པ། ལ་བཟླ་བ་ལ་གསུམ་སྟེ། ཀ་དག །ལྷུན་གྲུབ། ཀུན་གྱི་བདག་ཉིད་དུ་ལ་བཟླ་བའོ། །དང་པོ་ལ་གསུམ་སྟེ། སེམས་ཀྱི་རྩལ་སྣང་འགག་མེད་དུ་ཤར་བ་ལ་གཞན་གྱི་རྟོག་པ་ཀ་ནས་དག་པ་ནི་ཡེ་ཤེས་ཀ་དག །ཤར་ཚད་ཀྱི་རང་བཞིན་སྟོང་པ་ལ་དངོས་པོ་རང་རྒྱུད་ཀྱིས་བདེན་པ་ཀ་ནས་དག་པ་ནི་དབྱིངས་ཀ་དག །ངོ་བོ་སྣང་སྟོང་གཉིས་མེད་ལ་ཐ་དད་ཀྱི་གཉིས་ཆོས་ཀ་ནས་དག་པ་ནི་གཉིས་མེད་ཀ་དག་གོ །གཉིས་པ་ལྷུན་གྲུབ་ལ་གསུམ་སྟེ། སེམས་ཀྱི་རྩལ་སྣང་སྣ་ཚོགས་སུ་འཆར་བ་ལ་གཞན་ནས་བཙལ་མི་དགོས་ཏེ། སེམས་རང་ཤར་བ་ནི་ཆོས་ཅན་ལྷུན་གྲུབ། ཤར་ཚད་ཀྱི་རང་བཞིན་སྟོང་པ་ལས་གཞན་ནས་བཙལ་མི་དགོས་ཏེ། ཤར་བ་པོ་རང་སྟོང་པ་ནི། ཆོས་ཉིད་ལྷུན་གྲུབ། ངོ་བོ་སྣང་སྟོང་གཉིས་མེད་ལ་བཙལ་མི་དགོས་ཏེ། གང་སྣང་བ་དེ་དག་རང་གཉིས་སུ་མེད་པ་ནི་གཉིས་མེད་ལྷུན་གྲུབ་བོ། །གསུམ་པ་ཀུན་གྱི་བདག་ཉིད་དུ་གནས་པ་ནི། ཆོས་ཐམས་ཅད་ནི་རང་གི་སེམས་ཡིན་ལ། སེམས་ཀྱི་རྣམ་པ་ཅིར་ཡང་མ་འགགས་པ། གཤིས་ཅིར་ཡང་མ་གྲུབ། ངོ་བོ་ཅིར་ཡང་མ་གྲུབ་པ། གཞི་སྟོང་པ། ངོ་བོ་ཅིར་ཡང་མི་རྟོག་པ། དེ་ཆོས་ཐམས་ཅད་ཀྱི་བདག་ཉིད་དོ། །དེ་ཀ་གཞི་སྣང་སྟོང་གཉིས་མེད་ཀྱི་བདག་ཉིད་དུ་བསྟན་ཏེ། རྣམ་པ་ཅིར་ཡང་མ་འགགས་པ། གཞི་སྣང་བ་གཤིས་ཅིར་ཡང་མ་གྲུབ་པ། གཞི་སྟོང་པ་ངོ་བོ་ཅིར་ཡང་མི་རྟོག་པ། གཞི་གཉིས་མེད་ཀྱི་བདག་ཉིད་དོ། །དེ་ཀ་ལྟ་བ་བསྐྱེད་རྫོགས་གཉིས་ཀྱི་བདག་ཉིད་ཀྱང་ཡིན་ཏེ། མ་འགགས་པ་བསྐྱེད་རིམ། མ་གྲུབ་པ་རྫོགས་རིམ། མི་རྟོག་གཉིས་མེད་དོ། །བསྒོམ་པ་ཞི་གནས་ལྷག་མཐོང་གི་བདག་ཉིད་ཀྱང་ཡིན་ཏེ། མ་འགགས་པ་ལྷག་མཐོང༌། མ་གྲུབ་པ་ཞི་གནས་སོ། །མི་རྟོག་པ་གཉིས་མེད་དོ། །འབྲས་བུ་སངས་རྒྱས་སྐུ་གསུམ་གྱི་བདག་ཉིད་ཀྱང་ཡིན་ཏེ། མ་འགགས་པ་ལོངས་སྐུ། མ་གྲུབ་པ་ཆོས་སྐུ། མི་རྟོག་པ་སྤྲུལ་སྐུའོ། །མཁྱེན་པ་གཉིས་ལྡན་གྱི་ཡེ་ཤེས་ཀྱང་ཡིན་ཏེ། མ་འགགས་པ་ཅི་སྙེད་པ། མ་གྲུབ་པ་ཅི་ལྟ་བ། མི་རྟོག་པ་གཉིས་མེད་དུ་མཁྱེན་པའོ། །སྤངས་རྟོགས་ཕུན་སུམ་ཚོགས་པ་ཡང་ཡིན་ཏེ། མ་འགགས་པ་རྟོགས་པ། མ་གྲུབ་པ་སྤངས་པ་ཕུན་སུམ་ཚོགས་པའོ། །དེ་ལྟར་ཤེས་པས་འཁོར་འདས་སྐྱོན་ཡོན་ལྷ་འདྲེ་ལ་སོགས་པ་མཉམ་པ་ཉིད་དུ་ལ་བཟླ་བ་ཡིན་ཏེ༑ སྣང་བ་མ་འགག་པར་རང་སྣང་དུ་མཉམ། མ་གྲུབ་པ་སྟོང་པར་མཉམ། ངོ་བོ་མི་རྟོག་པར་གཉིས་མེད་དུ་མཉམ་མོ། །དེ་ཙམ་ན་སྣང་ཚད་གཞི་དངོས་པོའི་གནས་ལུགས། ལྟ་བ་བསྐྱེད་རྫོགས་ཀྱི་ཡོ་ལངས། སྒོམ་པ་ཞི་གནས་ལྷག་མཐོང་གི་རང་བཞིན། འབྲས་བུ་སངས་རྒྱས་སྐུ་གསུམ་གྱི་དཀྱིལ་འཁོར་ལས་མི་འདའ་སྟེ། རང་སྣང་རང་ངོ་ཤེས་ནས་གཞན་སྣང་གི་རྟོག་པ་མེད། རང་སྟོང་རང་ངོ་ཤེས་ནས་དངོས་པོ་རང་རྒྱུད་པའི་བདེན་འཛིན་བྲལ། གཉིས་མེད་རང་ངོ་ཤེས་ནས་ཐ་དད་གཉིས་འཛིན་མེད། དེ་ཙམ་ན་སྟོང་པ་རྒྱུ་འབྲས་སུ་འཆར་བ་ཐམས་ཅད་མཁྱེན་པའི་ཁྱད་ཆོས། རྒྱུ་འབྲས་སྟོང་པར་རྟོགས་པས་མཉམ་པ་ཉིད་ཆེན་པོར་ལ་བཟླའོ། །ཆོས་ཐམས་ཅད་སྣང་ཙམ་དུ་ཡོད་པ་ཀུན་རྫོབ་བདེན་པའི་ཕོ་སྒོ། བདེན་མེད་དངོས་སུ་གྲུབ་པ་མེད་པ་དོན་དམ་བདེན་པའི་རང་གཞུང༌། ཆོས་ཐམས་ཅད་ཡོད་པ་ཡང་མ་ཡིན། མེད་པ་ཡང་མ་ཡིན། རྟག་པ་ཡང་མ་ཡིན། ཆད་པ་ཡང་མ་ཡིན། སྐྱོན་ཡེ་དག །ཡོན་ཏན་ཡེ་རྫོགས། མཐའ་ཡེ་བྲལ། གཉིས་མ་གྲུབ་པའི་གཉིས་མེད། ཟུང་མ་རྙེད་པའི་ཟུང་འཇུག །མཐའ་མ་རྙེད་པའི་མཐའ་བྲལ། བཟང་ངན་མ་རྙེད་པའི་ཀུན་ཏུ་བཟང་པོ། བཙལ་མི་དགོས་པའི་ལྷུན་གྲུབ། འཁྲུལ་པ་མེད་པའི་ཀ་དག་གོ །བར་ལག་ཐམས་ཅད་རྨིངས་ནས་འགྱེལ། འགྲམ་རྡོ་ཐམས་ཅད་རྩད་ནས་ཐོན། ཐམ་ཐུམ་ཐམས་ཅད་གཅིག་ཏུ་འདྲེས། འཁོར་འདས་ཀྱི་ཆོས་ཐམས་ཅད་གདོད་ནས་རྣམ་པར་གྲོལ་བ་མིན་པ་གཅིག་ཀྱང་མེད་དོ། །སེམས་ཉིད་གཅེར་སྟོན་གྱི་ལག་ཁྲིད་ལྔ་པ། གསང་སྔགས་ལམ་གྱི་རིམ་པ་རིན་པོ་ཆེ་གསལ་བའི་སྒྲོན་མེའི་ཁྲིད་ཆུང༌ ཨྠིྀ། ༈ །གསང་སྔགས་ལམ་གྱི་རིམ་པའི་དོན་ཁྲིད་ངོ་སྤྲོད་དུ་གདབ་པ་རྒད་པོ་མཁར་ཚུགས་དང་རྒན་མོ་མཛུབ་ཚུགས་ཀྱི་གདམས་ངག་བཞུགས་སོ། །ན་མོ་གུ་རུ། སློབ་དཔོན་པདྨ་འབྱུང་གནས་ལ། རྔོག་ཤེས་རབ་རྒྱལ་པོ་ཞེས་བྱ་བ། ལོ་དྲུག་ཅུ་ལོན་པ༑ ཐོས་པ་མ་བྱས་པ། དད་པ་དང་ལྡན་པ་གཅིག་གིས། བྲག་པོ་ཆེའི་དབེན་རྩ་རུ། སློབ་དཔོན་ལོ་གཅིག་བཞུགས་ཏེ། རྔོག་གིས་ཀྱང་ཆོས་ཞུ་མ་བྱས། སློབ་དཔོན་གྱིས་ཀྱང་ཆོས་བཤད་མ་བྱས་པར། ལོ་གཅིག་ན་སློབ་དཔོན་བཞུད་པར་དགོངས་པ་ལ། རྔོག་གིས་གསེར་སྲང་གང་གིས་མཎྜལ་གྱི་མེ་ཏོག་བྱས་ནས་སློབ་དཔོན་ལ་ཕུལ་ཏེ། ཞུས་པ། སློབ་དཔོན་རིན་པོ་ཆེ་ལགས། བདག་ལ་ཐུགས་བརྩེ་བར་དགོངས་འཚལ། བདག་གཅིག་ཏུ་ཐོས་པ་ཆུང༌། གཉིས་སུ་ཤེས་རབ་ཆུང༌། གསུམ་དུ་ན་སོ་རྒས། སློབ་དཔོན་ཆེན་པོ། བདག་ལ་གོ་སླ་བ། སྒྲོ་འདོགས་ཆོད་པ། རྟོགས་པར་སླ་བ། ལྟ་ཁམས་ཆེ་བ། ཉམས་སུ་བླང་སླ་བ། རྒད་པོ་འཆི་ཁ་མའི་གདམས་ངག་ཅིག་ཞུ་འཚལ། ཞུས་པས། སློབ་དཔོན་གྱིས་རྒད་པོའི་སྙིང་གར་ཕྱག་འཁར་བཙུགས་ཏེ་གདམས་པ་ནི་འདིའོ། །ཉོན་ཅིག །རང་རིག་བྱང་ཆུབ་ཀྱི་སེམས་འདི་ལ་ལྟོས་དང༌། གཟུགས་དང་ཁ་དོག་ཏུ་གྲུབ་པ་མེད། ཕྱོགས་དང་རིས་སུ་ཆད་པ་མེད། མཐའ་དང་དབུས་སུ་གྲུབ་པ་མེད། དང་པོ་བྱུང་ས་མེད་དེ་སྟོང༌། བར་དུ་གནས་ས་མེད་དེ་སྟོང༌། ཐ་མ་འགྲོ་ས་མེད་དེ་སྟོང༌། སྟོང་པ་དེ་ག་ངོ་ཤེས་ན། རང་ངོ་རང་གིས་ཤེས་པ་ཡིན། ཆོས་ཀྱི་ཆོས་ཉིད་གོ་བ་ཡིན། སེམས་ཀྱིས་སེམས་མཐོང་བ་ཡིན། དངོས་པོའི་གནས་ལུགས་གཏན་ལ་ཕེབས་པ་ཡིན། རིག་པ་བྱང་ཆུབ་ཀྱི་སེམས་དེ་དངོས་པོ་ཅིར་ཡང་མ་གྲུབ་ཅིང༌། རང་བྱུང་རང་ལ་གནས་པས། གཞན་ནས་བཙལ་དུ་མེད་དེ། ཆོས་ཉིད་དོན་རྟོགས་པར་སླ། ཆོས་ཉིད་ཀྱི་དོན་ལ་རེ་དོགས་ཀྱི་འཛིན་པ་མེད་དེ། སེམས་རྟག་ཆད་ཀྱི་མཐའ་ལས་འདས། སངས་རྒྱ་རྒྱུ་མེད་དེ། སངས་རྒྱས་རང་གི་སེམས། དམྱལ་བར་འགྲོ་རྒྱུ་མེད་དེ། དམྱལ་བ་རང་གི་སེམས། སངས་རྒྱས་ལ་རེ་བ་མེད་དེ། རང་གིས་རང་སངས་རྒྱས། དམྱལ་བར་དོགས་པ་མེད་དེ། རིག་པ་རང་དག །ཆོས་ལ་བྱ་རྒྱུ་མེད་དེ། རིག་པ་ཆོས་ཉིད་རང་གསལ། ལྟ་བ་གནས་ལུགས་ཆེན་པོ་རང་ལ་ཡོད་པས། གཞན་ལ་བཙལ་དུ་མེད་པར་ཐག་བཅད་དོ། །དེ་ལྟར་རྟོགས་ནས་ཉམས་སུ་ལེན་པར་འདོད་ན། རང་གང་ན་འདུག་པ་དེ་ལུས་རི་བོའི་དགོན་གནས་ཡིན་པས། ཕྱི་སྣང་བ་ལ་གང་སྣང་བ་རང་སྣང༌། གསལ་བ་རང་གསལ། སྟོང་པ་རང་སྟོང༌། རང་བྱུང་རང་ཞི། སྤྲོས་པའི་མཐའ་ཐམས་ཅད་དང་བྲལ་བར་བཞག་གོ །དེ་ནས་ཕྱིའི་སྣང་བ་གྲོགས་སུ་ཤར་བས་སྣང་བ་རང་སར་དག །སྣང་བ་རང་སར་དག་པས་སྣང་བ་ལམ་དུ་ཁྱེར་ནས་ཉམས་སུ་བླང་བདེའོ། །ནང་སེམས་ལ་དྲན་པ་ཅི་ཤར་ཅི་འགྱུས་ཀྱང༌། དེའི་ངང་ལ་ངོ་བོ་མེད་དེ་སྟོང་པར་འདུག་པས་འགྱུ་དྲན་རང་སར་དག །སེམས་ལ་གསལ་ཐེབས་ལམ་དུ་སློང་བས་ཉམས་སུ་བླང་བདེའོ། །གསང་བ་ཉོན་མོངས་པ་གང་སྐྱེས་ཀྱང་ཉོན་མོངས་པ་ལ་བལྟས་ན་རྗེས་མེད་དེ། རང་བྱུང་རང་ཞི་ཉོན་མོངས་པ་ལམ་དུ་ཁྱེར་ནས་ཉམས་སུ་བླང་བདེའོ། །དེ་ལྟར་ཉམས་སུ་ལོན་ཙ་ན༑ སྒོམ་པ་ལ་ཐུན་བསྒོམ་བྱར་མེད། ཐམས་ཅད་གྲོགས་སུ་ཤར་ནས་ཉམས་ལ་གསལ་འགྲིབ་མེད། ཆོས་ཉིད་ཀྱི་ཡེ་ཤེས་ལ་རྒྱུན་ཆད་མེད། སྤྱོད་ལམ་ལ་སྤང་བླང་མེད། ཇི་ལྟར་འདུག་ཀྱང་ཆོས་ཉིད་ཀྱི་དོན་དང་འདུ་འབྲལ་མེད་དོ། །དེ་ལྟར་རྟོགས་ཙམ་ན། འབྱུང་བའི་ལུས་རྒས་ཀྱང་བྱང་ཆུབ་ཀྱི་སེམས་ལ་རྒས་རྒྱུ་མེད། དབང་པོ་རྣོ་བརྟུལ་མེད་དོ། །རྣམ་རིག་གི་རྟེན་ལུས་ཞིག་ཀྱང༌། རིག་པ་ཡེ་ཤེས་ཆོས་ཀྱི་སྐུ་ལ་རྒྱུན་ཆད་མེད་པས་མི་འགྱུར་ཞིང་བརྟན་པ་ཐོབ་པས། ཚེ་ལ་རིང་ཐུང་མེད་དོ། །ཆོས་ཉིད་ཀྱི་དོན་ལ་ཕྱོགས་རིས་མེད་པས། དོན་དེ་གོ་བས་ཐོས་པ་ལ་ཆེ་ཆུང་མེད་དོ། །རྒད་པོ་ཡང་དག་པའི་ལམ་དེའི་དོན་ཉམས་སུ་ལོངས་མཛོད་ཅིག །ཉམས་ལེན་རྒྱུད་ལ་འཁོལ་མཛོད། ཚིག་དང་དོན་དུ་མ་བྲལ་བར་མཛོད། བརྩོན་འགྲུས་ཀྱི་གྲོགས་དང་མ་བྲལ་བར་མཛོད། དྲན་པའི་རྩིས་ཟིན་པར་མཛོད། ལོང་གཏམ་དང་འཆལ་གཏམ་ལ་མ་དགའ་བར་མཛོད། ཐ་མལ་གྱི་བློ་དང་བྲལ་བར་མཛོད། བུ་ཚ་བོ་ལ་ཞེན་པ་མ་ཆེ་བར་མཛོད། བཟའ་བཏུང་ལ་སྲེད་པ་མ་ཆེ་བར་མཛོད། ཐ་མལ་པ་ཤི་བ་ལ་སོམས་མཛོད། ཚེ་ལ་ལོང་མེད་བརྩོན་འགྲུས་བསྐྱེད་མཛོད། རྒས་པོ་འཆི་ཁ་མའི་གདམས་ངག་མཛོད་ཅིག །གསུངས་ནས་རྒད་པོའི་སྙིང་ཁར་ཕྱག་འཁར་བཙུགས་ནས་གདམས་པས། རྔོག་ཤེས་རབ་རྒྱལ་པོ་གྲུབ་པ་ཐོབ་བོ། །རྒད་པོ་འཁར་བཙུག་གི་གདམས་པ་སངས་རྒྱས་ངོ་སྤྲོད་ཅེས་བྱ་བ། ཕྱི་རབས་ཀྱི་དོན་དུ་མཁར་ཆེན་བཟའ་མཚོ་རྒྱལ་གྱིས་ཡི་གེར་བཀོད་པའོ། །དགེའོ། ༈ །ན་མོ་གུ་རུ། སློབ་དཔོན་པདྨ་འབྱུང་གནས་རྒྱལ་པོ་ཁྲི་སྲོང་ལྡེའུ་བཙན་གྱིས་སྤྱན་དྲངས་ནས། བྲག་དམར་བསམ་ཡས་ན་བཞུགས་པའི་དུས་སུ། དད་ལྡན་བཟའ་དགེ་ལྡན་མ་ཞེས་བྱ་བ། རྒན་མོ་དད་པ་ཆེ་བ། སློབ་དཔོན་ལ་ཤིན་ཏུ་མོས་པ་གཅིག་ཡོད་པ་དེས། གཡོག་མོ་མར་གོང་བཟའ་རིན་ཆེན་མཚོ་ཞེས་བྱ་བ་ལ། ནངས་པར་ཆབ་ཞོ་ཞལ་བུ་རེ་ལ་རྒུན་འབྲུམ་གྱི་ཕྱེ་ཚུམ་རེ་བཏབ་ནས། སློབ་དཔོན་ལ་དྲང་དུ་བཏང་ངོ༌། །དེ་ནས་སློབ་དཔོན་བསམ་ཡས་སུ་བཞུད་པའི་ལམ་ཁར་རྒན་མོས་ཕྱག་དང་བསྐོར་བ་བྱས་ནས་ཞུས་པ། སློབ་དཔོན་ཆེན་པོ་ལགས། སློབ་དཔོན་ནི་བཞུད་ཁ་མ། རྒན་མོ་འཆི་ཁ་མ། བདག་གཅིག་ཏུ་བུད་མེད་ཀྱི་ལུས་ཐོབ་པས་སྐྱེ་བ་དམན། གཉིས་སུ་བློ་དམན་པས་ཤེས་རབ་དང་ཐོས་པ་ཆུང༌། གསུམ་དུ་ན་སོ་རྒས་ཤེས་པ་རྨུགས་ནས་མིག་མི་གསལ་བས། སློབ་དཔོན་ཆེན་པོས་རྒན་མོ་ལ། ཚེགས་ཆུང་བ་གོ་སླ་བ། ཉམས་སུ་བླང་བདེ་བ་དོན་ཆེ་བ། རྒན་མོ་འཆི་ཁ་མའི་གདམས་ངག་གཅིག་ཞུ་འཚལ། ཞུས་པས། སློབ་དཔོན་གྱི་ཞལ་ནས། རྒན་མོ་ཁྱོད་སུ་ཡིན་གསུངས་པས། རྒན་མོས་ཞུས་པ། ཁོལ་མོ་ངན་མ་འདི། ནངས་པར་གྱི་རིགས་ལ། ཆབ་ཞོ་ཞལ་བུ་རེ་འདྲེན་དུ་བཏང་བ་དེ་ལགས། ཞེས་ཞུས་པས། སློབ་དཔོན་གྱི་ཞལ་ནས། ཁྲི་སྲོང་ལྡེའུ་བཙན་བས་ཀྱང་དད་པ་ཆེ་བ་གཅིག་དེ་ཡིན་ཨང་གསུངས་ནས། རྒན་མོ་དཔོན་གཡོག་ལ་གདམས་པ་བཏབ་པ། རྒན་མོ་སྐྱིལ་ཀྲུང་ཆོས་ལ། ལུས་དྲང་པོར་བསྲང་ལ། དར་ཅིག་སེམས་ཀློད་ལ་ཞོག་ཅིག །གསུངས་ནས། སློབ་དཔོན་གྱིས་རྒན་མོའི་སྙིང་གར་མཛུབ་མོ་བཙུགས་ཏེ་གདམས་པ་བཏབ་པ་ནི་འདིའོ། །རྒན་མོ་ཉོན་མཛོད་ཅིག །ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཀྱི་དགོངས་པ་དང༌། ཁམས་གསུམ་གྱི་སེམས་ཅན་གཉིས་ལ་ཁྱད་ཅི་ཡོད་བྱས་ན། སེམས་རྟོགས་མ་རྟོགས་གཉིས་ལས་མེད། སེམས་ཉིད་ཀྱི་དོན་རྟོགས་པས་སངས་རྒྱས། མ་རྟོགས་པས་འཁོར་བར་འཁྱམས་པ་ཡིན། སངས་རྒྱས་ཀྱི་ཡེ་ཤེས་རང་ལ་ཡོད་ཀྱང་ངོ་མ་ཤེས་པ་ཡིན། དེ་ཡང་སངས་རྒྱས་ཀྱི་རྒྱུ་དེ་སེམས་ཅན་ཐམས་ཅད་ལ་ཡོད། སྒོས་དལ་འབྱོར་ཐོབ་པའི་མི་ལ་ཡོད་དེ། དེ་ཡང་སྐྱེས་པ་ལ་སངས་རྒྱས་ཀྱི་རྒྱུ་ཆེ་ལ། བུད་མེད་ལ་སངས་རྒྱས་ཀྱི་རྒྱུ་ཆུང་བ་མེད། དེས་ན་སྐྱེ་བ་བུད་མེད་དུ་སྐྱེས་ཀྱང་འགལ་བ་མེད་དོ་གསུངས། དེ་ལ་ཡེ་ཤེས་དེ་རྟོགས་ཤིང་ཤེས་པར་བྱེད་པ་ལ། སངས་རྒྱས་ཀྱིས་ཆོས་ཀྱི་སྒོ་མོ་བརྒྱད་ཁྲི་བཞི་སྟོང་གསུངས་ཀྱང༌། དེའི་དོན་ཤེས་པར་བྱེད་པ་ལ། བླ་མའི་གདམས་ངག་གི་ཚིག་གསུམ་དུ་འདུས་པས། བློ་དམན་ཞིང་ཤེས་རབ་ཆུང་ཡང་མི་ལྟོ་གསུངས། དེ་ལྟར་ཆོས་ཉིད་ཀྱི་དོན་སངས་རྒྱས་ཀྱི་དགོངས་པ་ཚིག་གསུམ་ནི། ཕྱི་གཟུང་བའི་ཡུལ་དག་པས་སྣང་བ་རང་སར་གྲོལ༑ ནང་འཛིན་པའི་སེམས་དག་པས་འཛིན་མེད་ཀྱི་རིག་པ་རང་སར་གྲོལ། བར་དུ་འོད་གསལ་ཉམས་དགའ་ནས་རང་ངོ་རང་གིས་ཤེས་པའོ། །དེ་ལ་ཕྱི་གཟུང་བའི་ཡུལ་ཇི་ལྟར་དག་ན། རིག་པ་བྱང་ཆུབ་ཀྱི་སེམས་འདི་ལ། རྣམ་རྟོག་གི་བློས་མ་བསླད་པར་རང་གསལ་དུ་སྣང་བ་འདི་ལ་བཞག་པས། ཡུལ་སྣང་བ་ལ་འཛིན་པ་མ་ཞུགས་པས། སྣང་བ་གང་ལྟར་ཡང་དོན་དུ་བདེན་པར་མི་འཛིན་ཏེ། སྣང་བ་གང་ཡིན། ས་རྡོ་རི་བྲག །རྩི་ཤིང་ནགས་ཚལ། ཁྱིམ་དང་མཁར། ནོར་དང་འབྲུ། དགྲ་དང་གཉེན། སྤུན་དང་གྲོགས། བུ་དང་ཁྱོ་ལ་སོགས་པ་གང་ལའང་བདག་ཏུ་འཛིན་པ་མ་ཞུགས། སྣང་ཡང་དོན་དུ་མི་འཛིན་ཅིང་ཞེན་པ་མེད་པ་དེ། ཕྱི་གཟུང་བའི་ཡུལ་དག་པ་བྱ་བ་ཡིན། གཟུང་བའི་ཡུལ་དག་པ་ཟེར་ནའང་མེད་བཞིན་མི་འགྲོ། ཞེན་པ་དང་འཛིན་པ་མེད་པ། གསལ་ལ་སྟོང་པ། དཔེར་ན་མེ་ལོང་ནང་གི་གཟུགས་བརྙན་ལྟར། སྣང་ཡང་གཟུང་རྒྱུ་དང་འཛིན་པ་མེད་དེ། སྟོང་པ་འདྲ་བས་སྣང་ཡང་རང་སར་སྟོང་པའོ། །ནང་འཛིན་པའི་སེམས་དག་པས་རིག་པ་རང་སར་གྲོལ་བའི་མན་ངག་ནི། སེམས་རྣམ་རྟོག་འགྱུ་བྱེད་ཀྱི་དྲན་པ་དང༌། དུག་ལྔ་གང་སྐྱེས་ཀྱང༌། གང་དྲན་པ་དེ་ལ་བློས་མ་བཟུང་བས། འགྱུ་བ་རང་སངས་སུ་འགྲོ་བས་སྟོང་སྟེ། འགྱུ་བྱེད་ཀྱི་སྐྱོན་གྱིས་མ་གོས་པའོ། །ནང་རིག་པ་དྲི་མ་མེད་པ་ཟེར་ནས། བེམ་པོ་ས་རྡོ་ལྟར་འགྲོ་བ་མེད། རིག་པ་ལ་རྟོག་པའི་དྲི་མས་མ་འཛིན་པ་ལ་ཟེར་བ་ཡིན། དཔེར་ན་རིན་པོ་ཆེ་གསེར་གླིང་དུ་ཕྱིན་པ་འདྲ། གསེར་གྱི་གླིང་དུ་ཕྱིན་ཙམ་ན། ས་རྡོའི་མིང་ཡང་མེད་པ་བཞིན་དུ། རང་རིག་པའི་ཡེ་ཤེས་སུ་གྲོལ་ཙམ་ན། རྣམ་རྟོག་གི་མིང་ཡང་མེད་པ་ཡིན། གསུངས། བར་དུ་འོད་གསལ་ཉམས་དགའ་བའི་མན་ངག་ནི། དེ་ཉམས་སུ་བླངས་པའི་དུས་སུ། རང་རིག་པ་དྲི་མ་མེད་པ་ལ་ས་ལེ་སང་ངེ་ཧྲིག་གེ་བ་དེ་ཉམས་སུ་བླངས་པས། བདེ་བ་གསལ་ལ་མི་རྟོག་པ་གསུམ། ཞེན་འཛིན་དང་བློ་ཡིས་མ་བསླད་པའི་ཉམས་གཅིག་འོང་བས། དེ་སངས་རྒྱས་དགོངས་པ་ཡིན་པས་རང་ངོ་རང་གིས་ཤེས་པའོ། །དཔེར་ན་རང་གི་ཡིན་པ་ལ་འདི་ངའི་ཡིན་ཆེད་དུ་བསྒོམ་མི་དགོས། མ་ཡིན་སྙམ་པའི་རྣམ་པར་རྟོག་པ་གང་ཡང་མི་འཆར་རོ། །དཔེ་དེ་དང་འདྲ་བར་རང་གི་རིག་པ་ཆོས་ཉིད་ཡིན་པར་རང་གིས་ངོ་ཤེས་ཙམ་ན། འཁོར་བའི་ཆོས་ཐམས་ཅད་ཆོས་ཉིད་སྟོང་པ་ཉིད་ཡིན་སྙམ་དུ་བསྒོམ་མི་དགོས། མ་བསྒོམས་ཀྱང་ཆོས་ཉིད་ཀྱི་དོན་དང་མི་འབྲལ་ཏེ། མ་བསྒོམས་པའི་བསྒོམ་པ་བྱ་བ་ཡིན་ནོ། །གསུངས། ཆོས་ཐམས་ཅད་རང་བཞིན་མེད་པར་ཤེས་པ་དེ། ཡེ་ཤེས་ཀྱི་བུ། རིག་པ་ཆོས་ཀྱི་དབྱིངས་ངོ་ཤེས་པས་རང་ངོ་རང་གིས་ཤེས་པ་བྱ་བ་ཡིན་ནོ། །དུས་དེ་ཙམ་ན། སྐྱེ་བ་ལ་མཐོ་དམན་མེད། བྱ་བ་ལ་ཆེ་ཆུང་མེད། བློ་ལ་རྣོ་རྟུལ་མེད། ཤེས་རབ་ལ་ཆེ་ཆུང་མེད། ཐོས་པ་ལ་མང་ཉུང་མེད། ན་སོ་ལ་རྒས་གཞོན་མེད། ཤེས་པ་ལ་གསལ་མི་གསལ་མེད༑ ཚེགས་ཆུང་ལ་གོ་སླ་བ། ཉམས་ལེན་བདེ་ལ་དོན་ཆེ་བ། འཆི་བ་ལ་ཉམས་མི་ང་བ། རྒན་མོ། དེ་ལྟར་ཉམས་སུ་ལེན་མཛོད། ཚེ་ལ་ལོང་མེད་པས་རྩོལ་བ་བསྐྱེད་མཛོད། བུ་ཁྱོའི་ཁོལ་པོར་གླག་མེད་ལོ་སྟོང་བྱེད་པ་ལས༑ རིག་པ་སྐད་ཅིག་བསྒོམ་པས་སྐྱེད་ཆེ། བླ་མས་གདམས་པའི་བརྒྱག་མྱུར་དུ་ཐོན། ཚེ་འདིའི་བྱ་བ་ལ་ཟིན་དུས་མེད་པས་སྒོམ་སྒྲུབ་མཐར་ཕྱུང་མཛོད། རྒན་མོ་འཆི་ཁ་མི་འཚེར་བའི་སུན་མར་གདམས་པ་འདི་ཐོང་མཛོད། ཅེས་གསུངས་ཏེ། རྒན་མོའི་སྙིང་གར་ཕྱག་མཛུབ་བཙུགས་ནས་གདམས་པ་བཏབ་པས། མིང་ཡང་རྒན་མོ་མཛུབ་བཙུག་གི་གདམས་པ་ཞེས་བྱའོ། །རྒན་མོ་དཔོན་གཡོག་གཉིས་གྲོལ་ཏེ་གྲུབ་པ་ཐོབ་བོ། །ཕྱི་རབས་ཀྱི་དོན་དུ་མཁར་ཆེན་བཟའ་མཚོ་རྒྱལ་གྱིས་ཡི་གེར་བཀོད་པའོ། །རྒན་མོ་མཛུབ་བཙུག་གི་གདམས་ངག་རྫོགས་སོ། ༈ །དེ་ལྟར་ཕྱི་ནང་གི་ལམ་གྱི་རིམ་པ་འདི་དག་སློབ་དཔོན་ཆེན་པོ་པདྨ་སཾ་བྷཱ་བའི་རྡོ་རྗེའི་གསུང་འཁྲུལ་མེད། ལོ་ཙཱ་བ་གཉགས་ཛྙཱ་ན་ཀུ་མཱ་ར་དང༌། གནུབས་སངས་རྒྱས་ཡེ་ཤེས་ནས་བཀའ་མའི་བརྒྱུད་པ་བར་མ་ཆད་པ་དང་། གཏེར་ཁ་གོང་འོག་གཉིས་དང༌། གཏེར་ཆེན་རྡོ་རྗེ་གླིང་པ་བཅས་ཀྱི་གཏེར་བྱོན་རྣམས་ཚིག་དོན་འདྲ་བར་བཞུགས་པས་ཤིན་ཏུ་ཡིད་ཆེས་པའི་གནས་སུ་བཟུང་བར་བྱའོ། །འདི་དག་ལ་རང་འགྲེལ་དང༌། གདམས་པའི་ཆ་ལག་སོགས་ཀྱང་རྒྱས་པར་བཞུགས་སོ། །སརྦ་དཱ་མངྒ་ལཾ་བྷ་བ་ཏུ།།
